\podnaslov{Interakcije}
\begin{enumerate}
\item $\sin^2 x + \cos^2 x = 1$
\item $\tan^2 x + 1 = \nicefrac{1}{\cos^2 x}$
\end{enumerate}

\podnaslov{Adicijski izreki}
\begin{enumerate}[resume]
\item $\sin(x+y) = \sin x \cos y + \cos y \sin x$
\item $\cos(x+y) = \cos x \cos y - \sin x \sin y$
\item $\tan(x+y) = \frac{\tan x + \tan y}{1 - \tan x \tan y}$
\item $\sin(x-y) = \sin x \cos y - \cos x \sin y$
\item $\cos(x-y) = \cos x \cos y + \sin x \sin y$
\item $\tan(x-y) = \frac{\tan x - \tan y}{1 + \tan x \tan y}$
\end{enumerate}

\podnaslov{Polovični koti}
\begin{enumerate}[resume]
\item $\sin^2 x = \pol (1 - \cos x)$
\item $\cos^2 x = \pol (1 + \cos x)$
\end{enumerate}

\podnaslov{Vsote in faktorizacija}

\begin{enumerate}[resume]
\item $\sin x + \sin y = 2 \sin(\frac{x+y}{2}) \cos(\frac{x-y}{2})$
\item $\sin x - \sin y = 2 \cos(\frac{x+y}{2}) \sin(\frac{x-y}{2})$
\item $\cos x + \cos y = 2 \cos(\frac{x+y}{2}) \cos(\frac{x-y}{2})$
\item $\cos x - \cos y = -2 \sin(\frac{x+y}{2}) \sin(\frac{x-y}{2})$
\item $\sin x \sin y = \pol (\cos(x-y) - \cos(x+y))$
\item $\cos x \cos y = \pol (\cos (x+y) + \cos(x-y))$
\item $\sin x \cos y = \pol (\sin(x+y) + \sin(x-y))$
\end{enumerate}
