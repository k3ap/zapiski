\naslov{Harmonične funkcije}

\begin{trditev}[verižno pravilo]
  Za $f, g \in \zvodvi{1}$ velja
  \begin{gather*}
	\partial_{\konj{z}} g\circ f
	= \partial_w g(f) \cdot \partial_{\konj{z}} f
	+ \partial_{\konj{w}} g(f) \cdot \partial_{\konj{z}} \konj{f}, \\
	\partial_z g \circ f
	= \partial_w g(f) \cdot \partial_{z} g
	+ \partial_{\konj{w}} g(f) \cdot \partial_z \konj{f}.
  \end{gather*}
\end{trditev}

\begin{definicija}
  Naj bo $\Omega \odpp \R^2$ in $u \in \zvodv{2}{\Omega}$.
  Funkcija $u$ je \pojem{harmonična} na $\Omega$, če je $\Delta u = 0$.
  Funkcija je \pojem{subharmonična}, če je $\Delta u \ge 0$.
\end{definicija}

\begin{definicija}
  Funkcija $u$ zadošča \pojem{lastnosti povprečne vrednosti} (LPV), če za vsak
  $z_0 \in \Omega$ in $r > 0$, pri katerih je $\cl{K}(z_0, r) \subseteq \Omega$,
  velja
  \[
	u(z_0) = \int_0^1 u(z_0 + r e^{2 \pi i t}) dt.
  \]
\end{definicija}

\begin{izrek}
  Funkcija $u \in \zvodv{2}{\Omega}$ je harmonična na $\Omega$ natanko tedaj, ko
  tam zadošča LPV.
\end{izrek}

\begin{proof}
  Naj bo $z_0 \in \Omega$ ter $r > 0$ dovolj majhen.
  Definirajmo $\varphi: [0,r] \to \C$ s predpisom
  \[
	\varphi(s) = \int_0^1 u(z_0 + s e^{2 \pi i t}) dt.
  \]
  Funkcija $\varphi$ je zvezna na $[0,r]$, diferenciabilna na $(0,r)$ in velja
  $\varphi(0) = u(z_0)$.
  Izračunamo lahko
  \[
	\varphi'(s)
	= \int_0^1 \partial_s u(x_0 + s \cos 2 \pi t, y_0 + s \sin 2 \pi t) dt
	= \int_0^1 \sk{\grad u(\vec{r}(t)), (\cos 2 \pi t, \sin 2 \pi t)} dt
  \]
  za $\vec{r}(t) = z_0 + s e^{2 \pi i t}$.
  Drugi vektor v skalarnem produktu je enak normali $\vec{n}(\vec{r}(t))$, torej
  integriramo smerni odvod
  \[
	\varphi'(s)
	= \int_0^1 \frac{\partial u}{\partial \vec{n}} (\vec{r}(t)) dt
	= \inv{2 \pi s} \iint_{K(z_0, s)} \Delta u dS
  \]
  po Greenovi formuli, kjer konstanta pred integralom pride iz spremembe
  spremenljivke.
  Obe smeri implikacije enostavno sledita.
\end{proof}

\vprasanje{Dokaži: funkcija je harmonična natanko tedaj, ko zadošča lastnosti
  povprečne vrednosti.}

\begin{trditev}
  Naj bosta $u, v \in \zvodv{2}{\Omega}$ realni.
  Če je $u + i v$ holomorfna, sta $u$ in $v$ harmonični.
\end{trditev}

\begin{trditev}
  Naj bo $\Omega \subseteq \R^2$ enostavno povezana odprta množica.
  Če je $u$ realna harmonična funkcija, obstaja holomorfna funkcija $f$ na
  $\Omega$, da je $u = \Real f$.
\end{trditev}

\begin{posledica}
  Če je $u$ harmonična in $F$ holomorfna, je $u \circ F$ harmonična.
\end{posledica}

\begin{proof}
  Vemo, da je lokalno $u = \Real f$.
  Sledi $u\circ F = \Real f \circ F = \Real (f \circ F)$.
\end{proof}

\vprasanje{Dokaži: če je $u$ harmonična in $F$ holomorfna, je $u \circ F$
  harmonična.}

\podnaslov{Dirichletov problem}

Označimo enotski disk z $D$ in njegov rob s $\mathbb{T} = \partial D$.
Funkcije na $\mathbb{T}$ lahko identificiramo z $1$-periodičnimi funkcijami na
$\R$ na očiten način.
Dirichletov problem je naslednji:
Če imamo dano funkcijo $f \in \zvezne{\mathbb{T}}$, ali lahko najdemo $u \in
\zvezne{\cl{D}}$, da bo $u$ harmonična na $D$ in da se bo ujemala z $f$ na
$\mathbb{T}$?

\begin{primer}
  Funkcijo $z^{-k}$ razširimo iz $\partial D$ na $D$ kot $\konj{z}^k$.
  Realni del je enak realnemu delu $z^k$, torej je funkcija res harmonična.
\end{primer}

\vprasanje{Kaj je Dirichletov problem na enotskem disku? Kako ga rešiš za
  $z^{-k}$?}

Označimo $e_m(t) = \exp(2 \pi i m t)$ za $m \in \Z$ in $t \in [0,1]$.
Pripadajoča razširitev funkcije $e_m$ na $D$ je funkcija $E_m: D \to \C$,
definirana kot
\[
  E_m(r e^{2 \pi i t}) =
  \begin{cases}
	r^m e^{2 \pi i m t} & m \in \N \\
	r^{-m} e^{2 \pi i m t} & m \in -\N
  \end{cases}
\]
Za $f = \sum_k c_k e_k$ dobimo razširitev $\sum_k c_k E_k$.
Ta prehod lahko zapišemo na eleganten način.
Funkcije $e_k$ tvorijo ortonormiran sistem: če je $f = \sum_k c_k e_k$, je
\[
  c_k = \int_0^1 f(\tau) \konj{e_k(\tau)} d\tau.
\]
Domnevamo, da je
\[
  F(r e^{2 \pi i t})
  = \sum_k \int_0^1 f(\tau) \konj{e_k(\tau)} r^{\abs{k}} e^{2 \pi i k t} d\tau
  = \int_0^1 f(\tau) \sum_k r^{\abs{k}} e^{2 \pi i k (t - \tau)} d\tau.
\]
Če vsoto v zadnjem integralu označimo z $G_r(t - \tau)$, je to enako $(f *
G_r)(t)$.
Z nekaj algebrajske manipulacije lahko ugotovimo, da je
\[
  G_r(s) = P_r(s) := \frac{1 - r^2}{1 - 2 r \cos (2 \pi s) + r^2}.
\]
Tej funkciji pravimo \pojem{Poissonovo jedro}.

\vprasanje{Izpelji predpis za rešitev Dirichletovega problema na enotskem disku.
  Kaj je Poissonovo jedro?}

\begin{lema}
  Za vsak $r \in \zo{0,1}$ je
  \[
	\int P_r(t) dt
	= \inv{\pi} \arctan\!\left( \frac{1+r}{1-r} \tan(\pi t) \right).
  \]
\end{lema}

\begin{trditev}
  Družina $\{P_r\}_r$ je približna enota.
  To pomeni:
  \begin{itemize}
  \item $\int_{-1/2}^{1/2} P_r = 1$
  \item $\int_{-1/2}^{1/2} \abs{P_r} \lesssim 1$ (je omejeno do konstante
	natančno)
  \item za vsak $\delta \in (0, \pol)$ velja
	\[
	  \lim_{r \uparrow 1} \int_{\delta < \abs{t} < \pol} \abs{P_r(t)} dt
	  = 0.
	\]
  \end{itemize}
\end{trditev}

\vprasanje{Definiraj približno enoto.}

\begin{izrek}
  Naj bo $\{P_r\}_r$ poljubna približna enota.
  Tedaj za vsako funkcijo $f \in \zvezne{\mathbb{T}}$ velja
  \begin{itemize}
  \item $P_r * f$ je zvezna na $\mathbb{T}$,
  \item $P_r * f \xrightarrow[r \to 1]{} f$ enakomerno na $\mathbb{T}$.
  \end{itemize}
\end{izrek}

\begin{proof}
  Prva točka:
  Izračunamo
  \[
	P_r * f(t) - P_r * f(t')
	= \int P_r(\tau) (f(t-\tau) - f(t'-\tau)) d\tau
  \]
  Ker je $f$ zvezna, je enakomerno zvezna, zato za $\varepsilon > 0$ obstaja
  $\delta > 0$, da iz $\abs{s_1 - s_2} < \delta$ sledi $\abs{f(s_1) - f(s_2)} <
  \varepsilon$.
  V primeru $\abs{t-t'} = \abs{(t-\tau) - (t'-\tau)} < \delta$ velja
  \[
	\abs{P_r*f(t) - P_r*f(t')}
	\le \int \abs{P_r(\tau)} \abs{f(t-\tau) - f(t'-\tau)} d\tau
	< \varepsilon \int \abs{P_r(\tau)} d\tau,
  \]
  ta integral pa je omejen.

  Druga točka:
  \[
	P_r * f(t) - f(t)
	= \int f(t-\tau) P_r(\tau) d\tau - f(t) \cdot 1
	= \int P_r(\tau) (f(t-\tau) - f(t)) d\tau
  \]
  po prvi lastnosti približne enote.
  Naj bo $\varepsilon > 0$.
  Obstaja $\delta > 0$, da za $\abs{\tau} < \delta$ velja $\abs{f(t-\tau) -
	f(t)} < \varepsilon$.
  Potem integral razdelimo na dva;
  \[
	\int_{\abs{\tau} < \delta} \abs{P_r(\tau)} \cdot \abs{f(t-\tau) - f(t)}
	d\tau
	< \varepsilon \int_{\abs{\tau} < \delta} \abs{P_r(\tau)} d\tau,
  \]
  kar je omejeno po drugi točki definicije, in
  \[
	\int_{\abs{\tau} > \delta} \abs{P_r(\tau)} \cdot \abs{f(t-\tau) - f(t)}
	< M \int_{\abs{\tau} > \delta} \abs{P_r(\tau)} d\tau,
  \]
  ker je razlika $f(t-\tau) - f(\tau)$ zvezna na kompaktu in zato omejena.
  Ta integral konvergira k $0$ za $r \to 1$.
\end{proof}

\vprasanje{Kateri lastnosti veljata za konvolucijo s približno enoto? Dokaži.}

\begin{izrek}
  Naj bo $u$ zvezna na $\cl{D}$ in harmonična na $D$.
  Za $r \in \zo{0,1}$ označimo $u_r: \mathbb{T} \to \C$ s predpisom $u_r(t) =
  u(r e^{2 \pi i t})$.
  Tedaj je $u_r = u_1 * P_r$.
\end{izrek}

\begin{proof}
  Izberimo $\alpha \in D$.
  Naj bo $\psi: \R \to \R$ takšna funkcija, da je
  \[
	e^{2 \pi i t} = \varphi_\alpha(e^{2 \pi i \psi(t)})
  \]
  za Möbiusovo transformacijo $\varphi_\alpha(z) = \frac{z+\alpha}{1 +
	\konj{\alpha}z}$.
  Dodatno zahtevamo, da je $\psi$ zvezna in $\psi(s+1) = \psi(s)+1$.
  Ta funkcija je odvedljiva, njen odvod je $1$-periodična funkcija.
  Ker je $v = u \circ \varphi_\alpha$ harmonična ($u$ harmonična,
  $\varphi_\alpha$ holomorfna), zadošča LPV.
  Velja
  \[
	u(\alpha) = v(0)
	= \int_0^1 v(e^{2 \pi i t}) dt
	= \int_0^1 v(e^{2 \pi i \psi(s)}) \psi'(s) ds,
  \]
  kjer smo zamenjali spremenljivko na $t = \psi(s)$.
  Nadalje je
  \[
	u(\alpha)
	= \int_0^1 v \circ \varphi_{-\alpha}(e^{2 \pi i s}) \psi'(s) ds
	= \int_0^1 u(e^{2 \pi i s}) \psi'(s) ds.
  \]
  Dokazati moramo le še
  \[
	\psi'(s) = \frac{1-r^2}{1 - 2r \cos(2 \pi (t-s)) + r^2}
  \]
  za $\alpha = r e^{2 \pi i t}$.
  Vemo, da je $e_1 \circ \psi = \varphi_{-\alpha} \circ e_1$, kjer je $e_1(t) =
  e^{2 \pi i s}$.
  Če to odvajamo, dobimo
  \[
	2 \pi i e^{2 \pi i \psi(s)} \psi'(s) = \frac{1 - r^2}{(1 - \konj{\alpha}
	  e_1(s))^2} 2 \pi i e^{2 \pi i s}.
  \]
  Iz tega izpeljemo
  \[
	\psi'(s) = \frac{1 - r^2}{(1 - \konj{\alpha} e_1(s))^2}
	\frac{e_1(s)}{\varphi_{-\alpha}(e_1(s))}.
  \]
  Pišimo $z = e_1(s)$.
  Potem je
  \[
	\frac{z}{(1 - \konj{\alpha} z)^2 \varphi_{-\alpha}(z)}
	= \frac{z (1 - \konj{\alpha} z)}{(1 - \konj{\alpha} z)^2 (z - \alpha)}
	= \frac{z}{(z-\alpha)(1 - \konj{\alpha} z)}
	= \frac{1}{(1 - \konj{\alpha} z)(1 - \alpha \konj{z})}
  \]
  upoštevaje $\abs{z} = 1$.
  To je potem enako $1 / \abs{1 - \konj{\alpha}z}^2$, torej ima $\psi'$ res
  želeno obliko.
\end{proof}

\vprasanje{Dokaži, da je harmonična funkcija natančno določena z robnimi
  vrednostmi.}

\begin{lema}
  Naj bo $\Omega \subseteq \R^2$ območje.
  Če je $u \in \zvezne{\Omega}$ in zadošča LPV, tedaj zanjo velja princip
  maksima; če $u$ doseže maksimum v neki točki iz $\Omega$, tedaj je $u$
  konstantna funkcija.
\end{lema}

\begin{proof}
  Brez škode za splošnost privzamemo, da je maksimum dosežen v točki $z_0$, ter
  da je $u(z_0) > 0$ ter realno.
  Vzamemo $K(z_0, r) \subseteq \Omega$ in $\rho < r$.
  Če na LPV uporabimo $\Real$, dobimo
  \[
	u(z_0) = \int_0^1 \Real u(z_0 + \rho e^{2 \pi i t}) dt,
  \]
  oziroma
  \[
	\int_0^1 (u(z_0) - \Real u(z_0 + \rho e^{2 \pi i t})) dt = 0.
  \]
  Ker je integrand povsod pozitiven, mora biti povsod enak $0$, torej je $u(z) =
  u(z_0)$ za vsak $z$ iz krogle $K(z_0, r)$.
  Vzemimo maksimalno množico $M = \{z \in \Omega \such u(z) = u(z_0)\}$.
  Po zgornjem je odprta, hkrati pa je zaprta, ker je množica rešitev enačbe.
\end{proof}

\vprasanje{Dokaži princip maksima za harmonične funkcije.}

\begin{posledica}
  Naj bo $u$ zvezna na $\Omega\odpp\R^2$ in naj zadošča LPV.
  Tedaj je $u$ harmonična.
\end{posledica}

\begin{proof}
  Naj bosta $z_0, r$ taka, da je $K(z_0, r) \subseteq \Omega$.
  Brez škode za splošnost je $z_0 = 0$ in $r = 1$.
  Naj bo $f(t) = u(e^{2 \pi i t})$ ter
  \[
	v(R e^{2 \pi i t}) =
	\begin{cases}
	  f * P_R(t) & R < 1 \\
	  f(t) & R = 1
	\end{cases}
  \]
  Vemo, da je $v$ harmonična na $D$, zvezna na $\cl{D}$ in $\left. v
  \right|_{\partial D} = f$.
  Ker je $w = u-v$ zvezna na $\cl{D}$, doseže $\abs{w}$ maksimum na $\cl{D}$,
  vemo pa, da je $w = 0$ na $\partial D$, torej je $w = 0$ povsod po principu
  maksima.
  Torej je $u = v$ harmonična.
\end{proof}

\vprasanje{Dokaži, da je zvezna funkcija, ki ustreza lastnosti povprečne
  vrednosti, harmonična.}

\begin{izrek}
  Kompleksna funkcija $u \in \zvodvi{2}$ je harmonična natanko tedaj, ko je $u =
  \varphi_1 + \konj{\varphi_2}$ za holomorfni $\varphi_1, \varphi_2$, določeni
  do konstante natančno.
\end{izrek}

\begin{proof}
  Enoličnost: če je $\varphi_1 + \konj{\varphi_2} = \varphi_3 +
  \konj{\varphi_4}$, je $\varphi_1 - \varphi_3 = \konj{\varphi_4 - \varphi_2}$.
  Leva stran je holomorfna, desna antiholomorfna, torej sta konstantni.

  Obstoj: $u = \alpha + i \beta$ za realni harmonični $\alpha, \beta$.
  Potem obstajata holomorfni $f, g$, da je $\alpha = \Real f$ in $\beta = \Real
  g$, torej
  \[
	u = \frac{f + ig}{2} + \konj{\frac{f - ig}{2}}.
  \]
\end{proof}

\vprasanje{Dokaži: kompleksna funkcija je harmonična natanko tedaj, ko je vsota
  holomorfne in antiholomorfne.}

\begin{izrek}
  Naj bo $\Omega$ omejeno območje v $\R^2$ in $f \in \zvodv{2}{\Omega}$ ter $g
  \in \zvezne{\partial \Omega}$.
  Dirichletov problem $\Delta u = f$ na $\Omega$ in $u = g$ na $\partial \Omega$
  ima največ eno rešitev $u \in \zvezne{\cl{\Omega}} \presek \zvodv{2}{\Omega}$.
\end{izrek}

\begin{proof}
  Če sta $u_1, u_2$ rešitvi, je $v = u_1 - u_2$ harmonična na $\Omega$ in zvena
  na zaprtju.
  Ker je $\cl{\Omega}$ kompakt, $\abs{v}$ na njem doseže maksimum; a $v = 0$ na
  $\partial \Omega$, torej je $v = 0$ povsod.
\end{proof}

\vprasanje{Pokaži, da je rešitev Dirichletovega problema na omejenem območju
  enolična, če obstaja.}

\begin{trditev}
  Če je $f$ holomorfna, je $\log \abs{f}^2$ harmonična na množici $\{z \in \C
  \such f(z) \ne 0\}$.
\end{trditev}

\begin{proof}
  Velja $\log f = \log \abs{f} + i \arg f$, torej je $\log \abs{f}$ realni del
  holomorfne funkcije.
\end{proof}

Za $F(z) = \frac{1+z}{1-z}$ velja $P(z) = \Real F(z) = \frac{1 -
  \abs{z}^2}{\abs{1-z}^2}$ in $P(r e^{2 \pi i s}) = P_r(e^{2 \pi i s})$.
Če pogledamo imaginarni del, $Q(z) = \Imag F(z) = \frac{2 \Imag z}{\abs{1
	-z}^2}$, lahko definiramo
\[
  Q_r(t) = Q(r e^{2 \pi i t}) = \frac{2 r \sin(2 \pi t)}{1 - 2r \cos(2 \pi t) +
	r^2}.
\]
Žal $Q_r$ ni približna enota.
Dodatno je
\[
  \lim_{r \uparrow 1} P_r(t) =
  \begin{cases}
	0 & t \ne 0 \\
	\infty & t = 0
  \end{cases}
\]
za $Q$ pa pride
\[
  \lim_{r \uparrow 1} Q_r(t) =
  \begin{cases}
	\cot(\pi t) & t \ne 0 \\
	0 & t = 0
  \end{cases}
\]
kar je posebej grozno, ker ta predpis pri $t \ne 0$ ni integrabilen.
Kljub temu poskusimo definirati $v_r(t) = (f * Q_r)(t)$ in upoštevamo, da je
$Q_r$ liha funkcija.

\begin{trditev}
  Če je $f: \mathbb{T} \to \C$ odvedljiva, tedaj za vsak $t \in \mathbb{T}$
  limita
  \[
	\lim_{\delta \downarrow 0} \int_{\delta \le \abs{\tau} \le \pol} \cot(\pi
	\tau) f(t - \tau) d\tau
  \]
  obstaja.
\end{trditev}

\begin{proof}
  Razdelimo integral na pozitiven in negativen del ter na enem od njih
  uporabimo spremembo spremenljivk, da dobimo
  \[
	I = \int_{\delta}^\pol \cot(\pi \tau) (f(t - \tau) - f(t + \tau)) d\tau.
  \]
  Potem to spremenimo v
  \[
	I = \int_\delta^\pol \tau \cot(\pi \tau) \left( \frac{f(t - \tau) -
		f(t)}{\tau} - \frac{f(t+\tau) - f(t)}{\tau} \right) d\tau.
  \]
  Notranjost oklepaja označimo z $g_t(\tau)$.
  To je zvezna funkcija, ker je $f$ odvedljiva.
  Ker je $\tau \cot(\pi \tau)$ zvezna na $[0,\pol]$, limita integrala za $\delta
  \to 0$ res obstaja.
\end{proof}

\vprasanje{Kaj je $Q_r$? Dokaži, da za odvedljive $f$ obstaja glavna vrednost
  $\cot(\pi \cdot) * f(t)$.}

Funkciji
\[
  Hf(t) = \text{g.v.} \int_{\mathbb{T}} \cot(\pi \tau) f(t-\tau) d\tau
\]
pravimo \pojem{Hilbertova transformiranka $f$}.

\begin{trditev}
  Če je $f: \mathbb{T} \to \C$ odvedljiva, limita
  \[
	\lim_{r\uparrow 1} (f * Q_r)(t)
  \]
  obstaja in je enaka $Hf(t)$.
\end{trditev}

\begin{proof}
  Za vsak $\delta \in (0,1)$ imamo
  \begin{align*}
	v_r(t) = (f * Q_r)(t)
	&= \int_{\abs{\tau} < \delta} \tau Q_r(\tau) \frac{f(t - \tau) - f(t)}{\tau}
	d\tau \\
	&+ \int_{\abs{\tau} \ge \delta} (Q_r(\tau) - \cot(\pi \tau)) f(t - \tau)
	d\tau \\
	&+ \int_{\abs{\tau} \ge \delta} \cot(\pi \tau) f(t - \tau) d\tau,
  \end{align*}
  kjer smo v prvem integralu uporabili dejstvo, da je $Q_r$ lih, da smo lahko
  odšteli $f(t)$.
  Te integrale označimo z $I_1$, $I_2$ in $I_3$.
  Za $\delta \to 0$ gre $I_3 \to Hf(t)$, dokazati moramo, da $I_1$ in $I_2$
  konvergirata k $0$.

  Dokažimo prvo
  \[
	\sup_{r, \tau} \abs{\tau Q_r(\tau)} < \infty.
  \]
  Velja
  \[
	\tau Q_r(\tau) = \frac{2 r \tau \sin(2 \pi \tau)}{1 - 2r \cos(2 \pi \tau) +
	  r^2}
	= \inv{\pi} \frac{xr \sin x}{1 - 2 r \cos x + r^2}
  \]
  za $x = 2 \pi \tau$.
  Za $r \to 1$ je $x r \sin x \approx r \sin x \cdot r \sin x$, in zato
  \[
	\tau Q_r(\tau) \approx \frac{(r \sin x)^2}{1 - 2r \cos x + r^2}
	\approx \frac{\abs{z - \konj{z}}^2}{\abs{1 - z}^2}
	\approx \frac{\abs{w - \konj{w}}^2}{\abs{w}^2}
	= \abs{\frac{w}{\abs{w}} - \frac{\konj{w}}{\abs{w}}}^2
  \]
  za $w = 1-z$, to pa je omejeno.
  Če pa je $r$ blizu $0$, pa je $1 - 2r \cos (2 \pi \tau) + r^2 \ge (1-r)^2
  \approx 1$ in je $\abs{\tau Q_r(\tau)}$ spet omejen.
  Torej je supremum res končen.

  Ocenimo
  \[
	\abs{I_1} \le \int_{\abs{\tau} \le \delta} \abs{\tau Q_r(\tau)}
	\abs{\frac{f(t-\tau) - f(t)}{\tau}} d\tau.
  \]
  Oba člena tu sta omejena, torej je integral manjši od $2 M \delta$ za nek $M$,
  in konvergira k $0$ za $\delta \to 0$.

  Izračunamo lahko, da je
  \[
	Q_1(\tau) - Q_r(\tau) = \frac{1 + \cos w}{\sin w} P_r(\tau) \frac{1-r}{1+r}
  \]
  za $w = 2 \pi \tau$.
  Drugi integral lahko zapišemo kot
  \[
	I_2 = - \int_{\abs{\tau} \ge \delta} \tau (Q_1(\tau) - Q_{1 - \delta}(\tau))
	\left( \frac{f(t-\tau) - f(t)}{\tau} \right) d\tau,
  \]
  kar potem ocenimo na
  \[
	\abs{I_2} \le \sup \abs{\frac{f(t-\tau) - f(t)}{\tau}} \cdot
	\int_{\abs{\tau} \ge \delta} \abs{\frac{\tau (1 + \cos w)}{(2 - \delta) \sin
		w} P_{1-\delta}(\tau) \delta} d\tau.
  \]
  Ker je
  \[
	\int_{\abs{\tau} \ge \delta} P_{1-\delta}(\tau) d\tau
	\le \int_{\mathbb{T}} P_{1-\delta}(\tau) d\tau = 1,
  \]
  je $\abs{I_2}$ omejen s $C \delta$ in konvergira k $0$ za $\delta \to 0$.
\end{proof}

\vprasanje{Pokaži, da limita $f * Q_r(t)$ za $r \uparrow 1$ obstaja. Čemu je
  enaka?}

Rešitev Dirichletovega problema na enotskem disku želimo posplošiti za poljubno
(dovolj lepo) območje $\Omega$.
Poiskati želimo $P_\Omega$, da bo
\[
  u(z) = \int_{\partial \Omega} P_\Omega(z, w) f(w) dw
\]
rešitev $\Delta u = 0$, $u = f$ na $\partial \Omega$.
Upoštevamo Riemannov upodobitveni izrek, ki pravi, da za enostavno povezano
območje z gladkim robom obstaja biholomorfizem $\phi: D \to \Omega$.
Tega lahko razširimo do funkcije $\cl{D} \to \cl{\Omega}$, ki slika rob v rob
difeomorfno.
Pišimo $\tilde{f} = f \circ \phi: \partial D \to \R$.
Če parametriziramo rob $\Omega$ s $\phi(e^{2 \pi i t})$, zgornja formula dobi
obliko
\[
  u(z) = \int_0^1 P_\Omega(z, \phi(e^{2 \pi i t})) f(e^{2 \pi i t})
  \abs{\phi'(e^{2 \pi i t})} 2 \pi dt,
\]
funkcijo $\tilde{f}$ pa razširimo na $D$ kot
\[
  \tilde{u}(r e^{2 \pi i t}) = \int_0^1 P_r(e^{2 \pi i t} e^{-2 \pi i \tau})
  \tilde{f}(e^{2 \pi i \tau}) d\tau.
\]
Zahtevamo $\tilde{u} = u \circ \phi$, kar nam da
\[
  u(z) = \int_0^1 P(\phi^{-1}(z)\,\konj{\phi^{-1}(\phi(e^{2 \pi i t}))})
  f(\phi(e^{2 \pi i \tau})) d\tau
\]
torej
\[
  P_\Omega(z,w) \abs{\phi'(\phi^{-1}(w))} 2 \pi = P(\phi^{-1}(z)
  \konj{\phi^{-1}(w)}),
\]
iz česar dobimo
\[
  P_\Omega(z,w) = \frac{\abs{\psi'(w)}}{2 \pi} \frac{1 - \abs{\psi(z)
	  \konj{\psi(w)}}^2}{\abs{1 - \konj{\psi(z) \psi(w)}}^2}
  = \frac{\abs{\psi'(w)}}{2 \pi}
  \frac{1 - \abs{\psi(z)}^2}{\abs{\psi(w) - \psi(z)}^2}.
\]

\vprasanje{Izpelji predpis za jedrno funkcijo $P_\Omega$.}
