\naslov{Harmonične funkcije}

\begin{trditev}[verižno pravilo]
  Za $f, g \in \zvodvi{1}$ velja
  \begin{gather*}
	\partial_{\konj{z}} g\circ f
	= \partial_w g(f) \cdot \partial_{\konj{z}} f
	+ \partial_{\konj{w}} g(f) \cdot \partial_{\konj{z}} \konj{f}, \\
	\partial_z g \circ f
	= \partial_w g(f) \cdot \partial_{z} g
	+ \partial_{\konj{w}} g(f) \cdot \partial_z \konj{f}.
  \end{gather*}
\end{trditev}

\begin{definicija}
  Naj bo $\Omega \odpp \R^2$ in $u \in \zvodv{2}{\Omega}$.
  Funkcija $u$ je \pojem{harmonična} na $\Omega$, če je $\Delta u = 0$.
  Funkcija je \pojem{subharmonična}, če je $\Delta u \ge 0$.
\end{definicija}

\begin{definicija}
  Funkcija $u$ zadošča \pojem{lastnosti povprečne vrednosti} (LPV), če za vsak
  $z_0 \in \Omega$ in $r > 0$, pri katerih je $\cl{K}(z_0, r) \subseteq \Omega$,
  velja
  \[
	u(z_0) = \int_0^1 u(z_0 + r e^{2 \pi i t}) dt.
  \]
\end{definicija}

\begin{izrek}
  Funkcija $u \in \zvodv{2}{\Omega}$ je harmonična na $\Omega$ natanko tedaj, ko
  tam zadošča LPV.
\end{izrek}

\begin{proof}
  Naj bo $z_0 \in \Omega$ ter $r > 0$ dovolj majhen.
  Definirajmo $\varphi: [0,r] \to \C$ s predpisom
  \[
	\varphi(s) = \int_0^1 u(z_0 + s e^{2 \pi i t}) dt.
  \]
  Funkcija $\varphi$ je zvezna na $[0,r]$, diferenciabilna na $(0,r)$ in velja
  $\varphi(0) = u(z_0)$.
  Izračunamo lahko
  \[
	\varphi'(s)
	= \int_0^1 \partial_s u(x_0 + s \cos 2 \pi t, y_0 + s \sin 2 \pi t) dt
	= \int_0^1 \sk{\grad u(\vec{r}(t)), (\cos 2 \pi t, \sin 2 \pi t)} dt
  \]
  za $\vec{r}(t) = z_0 + s e^{2 \pi i t}$.
  Drugi vektor v skalarnem produktu je enak normali $\vec{n}(\vec{r}(t))$, torej
  integriramo smerni odvod
  \[
	\varphi'(s)
	= \int_0^1 \frac{\partial u}{\partial \vec{n}} (\vec{r}(t)) dt
	= \inv{2 \pi s} \iint_{K(z_0, s)} \Delta u dS
  \]
  po Greenovi formuli, kjer konstanta pred integralom pride iz spremembe
  spremenljivke.
  Obe smeri implikacije enostavno sledita.
\end{proof}

\vprasanje{Dokaži: funkcija je harmonična natanko tedaj, ko zadošča lastnosti
  povprečne vrednosti.}

\begin{trditev}
  Naj bosta $u, v \in \zvodv{2}{\Omega}$ realni.
  Če je $u + i v$ holomorfna, sta $u$ in $v$ harmonični.
\end{trditev}

\begin{trditev}
  Naj bo $\Omega \subseteq \R^2$ enostavno povezana odprta množica.
  Če je $u$ realna harmonična funkcija, obstaja holomorfna funkcija $f$ na
  $\Omega$, da je $u = \Real f$.
\end{trditev}

\begin{posledica}
  Če je $u$ harmonična in $F$ holomorfna, je $u \circ F$ harmonična.
\end{posledica}

\begin{proof}
  Vemo, da je lokalno $u = \Real f$.
  Sledi $u\circ F = \Real f \circ F = \Real (f \circ F)$.
\end{proof}

\vprasanje{Dokaži: če je $u$ harmonična in $F$ holomorfna, je $u \circ F$
  harmonična.}

\podnaslov{Dirichletov problem}

Označimo enotski disk z $D$ in njegov rob s $\mathbb{T} = \partial D$.
Funkcije na $\mathbb{T}$ lahko identificiramo z $1$-periodičnimi funkcijami na
$\R$ na očiten način.
Dirichletov problem je naslednji:
Če imamo dano funkcijo $f \in \zvezne{\mathbb{T}}$, ali lahko najdemo $u \in
\zvezne{\cl{D}}$, da bo $u$ harmonična na $D$ in da se bo ujemala z $f$ na
$\mathbb{T}$?

\begin{primer}
  Funkcijo $z^{-k}$ razširimo iz $\partial D$ na $D$ kot $\konj{z}^k$.
  Realni del je enak realnemu delu $z^k$, torej je funkcija res harmonična.
\end{primer}

\vprasanje{Kaj je Dirichletov problem na enotskem disku? Kako ga rešiš za
  $z^{-k}$?}

Označimo $e_m(t) = \exp(2 \pi i m t)$ za $m \in \Z$ in $t \in [0,1]$.
Pripadajoča razširitev funkcije $e_m$ na $D$ je funkcija $E_m: D \to \C$,
definirana kot
\[
  E_m(r e^{2 \pi i t}) =
  \begin{cases}
	r^m e^{2 \pi i m t} & m \in \N \\
	r^{-m} e^{2 \pi i m t} & m \in -\N
  \end{cases}
\]
Za $f = \sum_k c_k e_k$ dobimo razširitev $\sum_k c_k E_k$.
Ta prehod lahko zapišemo na eleganten način.
Funkcije $e_k$ tvorijo ortonormiran sistem: če je $f = \sum_k c_k e_k$, je
\[
  c_k = \int_0^1 f(\tau) \konj{e_k(\tau)} d\tau.
\]
Domnevamo, da je
\[
  F(r e^{2 \pi i t})
  = \sum_k \int_0^1 f(\tau) \konj{e_k(\tau)} r^{\abs{k}} e^{2 \pi i k t} d\tau
  = \int_0^1 f(\tau) \sum_k r^{\abs{k}} e^{2 \pi i k (t - \tau)} d\tau.
\]
Če vsoto v zadnjem integralu označimo z $G_r(t - \tau)$, je to enako $(f *
G_r)(t)$.
Z nekaj algebrajske manipulacije lahko ugotovimo, da je
\[
  G_r(s) = P_r(s) := \frac{1 - r^2}{1 - 2 r \cos (2 \pi s) + r^2}.
\]
Tej funkciji pravimo \pojem{Poissonovo jedro}.

\vprasanje{Izpelji predpis za rešitev Dirichletovega problema na enotskem disku.
  Kaj je Poissonovo jedro?}

\begin{lema}
  Za vsak $r \in \zo{0,1}$ je
  \[
	\int P_r(t) dt
	= \inv{\pi} \arctan\!\left( \frac{1+r}{1-r} \tan(\pi t) \right).
  \]
\end{lema}

\begin{trditev}
  Družina $\{P_r\}_r$ je približna enota.
  To pomeni:
  \begin{itemize}
  \item $\int_{-1/2}^{1/2} P_r = 1$
  \item $\int_{-1/2}^{1/2} \abs{P_r} \lesssim 1$ (je omejeno do konstante
	natančno)
  \item za vsak $\delta \in (0, \pol)$ velja
	\[
	  \lim_{r \uparrow 1} \int_{\delta < \abs{t} < \pol} \abs{P_r(t)} dt
	  = 0.
	\]
  \end{itemize}
\end{trditev}

\vprasanje{Definiraj približno enoto.}

\begin{izrek}
  Naj bo $\{P_r\}_r$ poljubna približna enota.
  Tedaj za vsako funkcijo $f \in \zvezne{\mathbb{T}}$ velja
  \begin{itemize}
  \item $P_r * f$ je zvezna na $\mathbb{T}$,
  \item $P_r * f \xrightarrow[r \to 1]{} f$ enakomerno na $\mathbb{T}$.
  \end{itemize}
\end{izrek}

\begin{proof}
  Prva točka:
  Izračunamo
  \[
	P_r * f(t) - P_r * f(t')
	= \int P_r(\tau) (f(t-\tau) - f(t'-\tau)) d\tau
  \]
  Ker je $f$ zvezna, je enakomerno zvezna, zato za $\varepsilon > 0$ obstaja
  $\delta > 0$, da iz $\abs{s_1 - s_2} < \delta$ sledi $\abs{f(s_i) - f(s_2)} <
  \varepsilon$.
  V primeru $\abs{t-t'} = \abs{(t-\tau) - (t'-\tau)} < \delta$ velja
  \[
	\abs{P_r*f(t) - P_r*f(t')}
	\le \int \abs{P_r(\tau)} \abs{f(t-\tau) - f(t'-\tau)} d\tau
	< \varepsilon \int \abs{P_r(\tau)} d\tau,
  \]
  ta integral pa je omejen.

  Druga točka:
  \[
	P_r * f(t) - f(t)
	= \int f(t-\tau) P_r(\tau) d\tau - f(t) \cdot 1
	= \int P_r(\tau) (f(t-\tau) - f(t)) d\tau
  \]
  po prvi lastnosti približne enote.
  Naj bo $\varepsilon > 0$.
  Obstaja $\delta > 0$, da za $\abs{\tau} < \delta$ velja $\abs{f(t-\tau) -
	f(t)} < \varepsilon$.
  Potem integral razdelimo na dva;
  \[
	\int_{\abs{\tau} < \delta} \abs{P_r(\tau)} \cdot \abs{f(t-\tau) - f(\tau)}
	d\tau
	< \varepsilon \int_{\abs{\tau} < \delta} \abs{P_r(\tau)} d\tau,
  \]
  kar je omejeno po drugi točki definicije, in
  \[
	\int_{\abs{\tau} > \delta} \abs{P_r(\tau)} \cdot \abs{f(t-\tau) - f(\tau)}
	< M \int_{\abs{\tau} > \delta} \abs{P_r(\tau)} d\tau,
  \]
  ker je razlika $f(t-\tau) - f(\tau)$ zvezna na kompaktu in zato omejena.
  Ta integral konvergira k $0$ za $r \to 1$.
\end{proof}
