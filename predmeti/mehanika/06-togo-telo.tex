\naslov{Togo telo}

\begin{definicija}
  Gibanje sistema materialnih točk je \pojem{togo}, če gibanje ohranja razdalje
  med točkami.
\end{definicija}

\begin{definicija}
  Telo je \pojem{togo}, če nima koles.
  Alternativno je telo togo, če je edini njegov način gibanja togo gibanje.
\end{definicija}

\begin{izrek}
  Izometrija je afina preslikava.
\end{izrek}

\begin{proof}
  Izberimo $\mb{P}_0$ in točke $\mb{A}, \mb{B}, \mb{C}$, ki niso kolinearne.
  Naj bo $\mb{P}$ poljubna točka.
  Za $\vec{r} = \mb{P} - \mb{P}_0$, $\vec{a} = \mb{A} - \mb{P}_0$, $\vec{b} =
  \mb{B} - \mb{P}_0$ in $\vec{c} = \mb{C} - \mb{P}_0$ velja
  \[
	\vec{r} = \alpha \vec{a} + \beta \vec{b} + \gamma \vec{c}.
  \]
  Izometrija naj slika $\mb{T}$ v $\mb{T}'$, poleg tega naj je
  \[
	\pvec{r}' = \alpha' \pvec{a}' + \beta' \pvec{b}' + \gamma' \pvec{c}'.
  \]
  Če enačbo za $\vec{r}$ skalarno množimo z $\vec{a}$, $\vec{b}$ oziroma
  $\vec{c}$, dobimo enačbe
  \begin{align*}
	\vec{r} \cdot \vec{a} &= \alpha \vec{a} \cdot \vec{a} + \beta \vec{b} \cdot \vec{a} + \gamma \vec{c} \cdot \vec{a} \\
	\vec{r} \cdot \vec{b} &= \alpha \vec{a} \cdot \vec{b} + \beta \vec{b} \cdot \vec{b} + \gamma \vec{c} \cdot \vec{b} \\
	\vec{r} \cdot \vec{c} &= \alpha \vec{a} \cdot \vec{c} + \beta \vec{b} \cdot \vec{c} + \gamma \vec{c} \cdot \vec{c}
  \end{align*}
  To zložimo v matriko $A$, da dobimo
  \[
	\begin{bmatrix}
	  \vec{r} \cdot \vec{a} \\
	  \vec{r} \cdot \vec{b} \\
	  \vec{r} \cdot \vec{c} \\
	\end{bmatrix}
	= A
	\begin{bmatrix}
	  \alpha \\ \beta \\ \gamma
	\end{bmatrix}
  \]
  in podobno za količine s črto.
  Izometrija ohranja tako razdalje kot kote, torej sta levi strani sistemov in
  matriki $A, A'$ enaki.
  Sledi torej $\alpha = \alpha'$, $\beta = \beta'$ in $\gamma = \gamma'$.
  Dobimo
  \[
	\mb{P}' = \mb{P}_0' + \pvec{r}' = \mb{P}_0' + A \vec{r}
	= \mb{P}_0' + A(\mb{P} - \mb{P}_0)
  \]
  in $A \in O(3)$.
\end{proof}

Togo gibanje lahko zapišemo v obliki
\[
  \mb{P}(t) = \mb{P}_0(t) + Q(t) (\mb{P}(t=0) -\mb{P}_0(t=0)).
\]
Predstavimo ga kot relativno gibanje, kjer je RKS položaj telesa, kot ga vidimo
ob času $t=0$, AKS pa položaj telesa ob času $t$.
RKS se giblje skupaj s telesom, zato mu pravimo \pojem{telesni KS}, absolutnemu
pa pravimo \pojem{prostorski KS}.

\vprasanje{Kako opišeš gibanje togega telesa?}

Za telo $B$ definiramo volumensko in masno mero in predpostavimo, da za $B' =
B(t)$ velja $m(B') = m(B)$.
Sledi
\[
  m(B') = \iiint_{B'} \rho' dV'
  = \iiint_{B} \rho'(\mb{P}'(\mb{P})) \abs{\det \frac{\partial
	  \mb{P}'}{\partial \mb{P}}} dV.
\]
Ker pa je $\mb{P}' = \mb{P}_0' + Q(\mb{P} - \mb{P}_0)$, torej je odvod enak $Q$
in njegova determinanta enaka $1$.
Torej
\[
  \iiint_{B} \rho'(\mb{P}'(\mb{P})) dV
  = m(B')
  = m(B)
  = \iiint_B \rho(\mb{P}) dV,
\]
torej (ker to velja tudi na delih telesa) $\rho'(\mb{P}'(\mb{P})) =
\rho(\mb{P})$.
Podobno izpeljemo, da je
\[
  \mb{P}_*' = \inv{m(B)} \iiint_{B'} \mb{P}' - \mb{P}_0' dm + \mb{P}_0'
\]
enak $\mb{P}_*' = \mb{P}_0' + Q(\mb{P}_* - \mb{P}_0)$ za $\mb{P}_*$, definiran s
podobnim integralom.
Podobno kot v diskretnem primeru lahko tudi tu izpeljemo
\[
  \dot{\mb{P}}_*' = \inv{m(B')} \iiint_{B'} \dot{\mb{P}}' dm.
\]

Vrtilno količino izračunamo kot
\begin{align*}
  \vec{l}'(\mb{P}_0')
  &= \iiint_{B'} (\mb{P}' - \mb{P}_0') \times \partial_t (\mb{P}' - \mb{P}_0') dm' \\
  &= \iiint_{B'} \pvec{\zeta}' \times (\pvec{\omega}' \times \pvec{\zeta}') dm' \\
  &= \iiint_{B'} \abs{\pvec{\zeta}'}^2 \pvec{\omega}' - (\pvec{\zeta}' \cdot \pvec{\omega}') . \pvec{\zeta}' dm' \\
  &= \iiint_{B'} \left(\abs{\pvec{\zeta}'}^2 I - \pvec{\zeta}' \otimes \pvec{\zeta}'\right) dm' \cdot \pvec{\omega}'.
\end{align*}
Iz tega dobimo definicijo prostorskega vztrajnostnega tenzorja
\[
  J'(\mb{P}_0') = \iiint_{B'} \left(\abs{\pvec{\zeta}'}^2 I - \pvec{\zeta}'
	\otimes \pvec{\zeta}'\right) dm'
  = \iiint_{B'} \abs{\mb{P}' - \mb{P}_0'}^2 I - (\mb{P}' - \mb{P}_0') \otimes
  (\mb{P}' - \mb{P}_0') dm'.
\]
Podobno definiramo telesni vztrajnostni tenzor, izračunamo lahko $J' = Q J Q^T$.
Vztrajnostni tenzor je simetričen, poleg tega pa ga lahko diagonaliziramo
\[
  J(\mb{P}_0') = \sum_{i=1}^3 J_i \vec{w}_i \otimes \vec{w}_i,
\]
kjer so $J_i$ \pojem{glavne} (lastne) vrednosti.

\begin{trditev}
  Vztrajnostni tenzor je nenegativen.
  Če $B$ ni tanka palica, je pozitivno definiten.
\end{trditev}

\begin{proof}
  Trdimo, da je $\vec{e} \cdot J \vec{e} \ge 0$ za enotski $\vec{e}$.
  Po Cauchy-Schwarzu velja
  \[
	\iiint_B \abs{\vec{\zeta}}^2 - (\vec{e} \cdot \vec{\zeta})^2 dm \ge 0
  \]
  natanko tedaj, ko je $\vec{e}$ vzporeden $\vec{\zeta}$, kar se zgodi le, če je
  $B$ tanka palica.
\end{proof}

\vprasanje{Definiraj vztrajnostni tenzor in dokaži, da je pozitivno definiten.}

\begin{trditev}
  Normala na ravnino zrcalne simetrije telesa je glavna smer vztrajnostnega
  tenzorja s polom na tej ravnini.
\end{trditev}

\begin{proof}
  Naj bo $\vec{e}$ ta normala.
  Trdimo $J(\mb{P}_0) \vec{e} = J \vec{e}$ za neko lastno vrednost $J$.
  To je ekvivalentno temu, da je deviator
  \[
	D \vec{e} = \vec{\zeta} \otimes \vec{\zeta} \vec{e}
  \]
  vzporeden $\vec{e}$.
  Naj bo $\vec{f} \bot \vec{e}$ in izračunajmo
  \[
	\vec{f} \cdot D \vec{e}
	= \iiint_B \vec{f} \cdot \vec{\zeta} . \vec{e} \cdot \vec{\zeta} dm
	= \iiint_B \vec{f} \cdot \pvec{\zeta}' . \vec{e} \cdot \pvec{\zeta}' dm
  \]
  za zrcalno sliko $\pvec{\zeta}'$.
  Izračunamo lahko, da velja $\vec{e} \cdot \pvec{\zeta}' = - \vec{e} \cdot
  \vec{\zeta}$ in $\vec{f} \cdot \pvec{\zeta}' = \vec{f} \cdot \vec{\zeta}$,
  torej velja $\vec{f} \cdot D \vec{e} = 0$.
\end{proof}

\begin{trditev}
  Naj ima $B$ dve ravnini simetrije z normalama $\vec{e}_1$ in $\vec{e}_2$.
  Potem ima $J(\mb{P}_0)$, kjer je $\mb{P}_0$ v preseku obeh ravnin, glavne
  smeri $\vec{e}_1, \vec{e_2}, \vec{e}_1 \times \vec{e}_2$.
\end{trditev}

\vprasanje{Kaj lahko poveš o vztrajnostnemu tenzorju, če ima eno ali dve ravnini
  zrcalne simetrije? Dokaži.}

Če razpišemo predpis za tenzorski produkt z uporabo $\mb{P} - \mb{P}_0 = (\mb{P}
- \mb{P}_*) + (\mb{P}_* - \mb{P}_0)$, dobimo
\[
  J(\mb{P}_0) = J(\mb{P}_*) + m \abs{\mb{P}_* - \mb{P}_0}^2 I - m (\mb{P}_* -
  \mb{P}_0) \otimes (\mb{P}_* - \mb{P}_0).
\]
Tej formuli pravimo \pojem{Steinerjev izrek}.

\vprasanje{Kaj pravi Steinerjev izrek?}

Za kinetično energijo razpišemo
\[
  T = \pol \iiint_{B'} \abs{\dot{\mb{P}}'}^2 dm',
\]
pri čemer upoštevamo $\dot{\mb{P}}' = \dot{\mb{P}}_*' + \pvec{\omega}' \times
(\mb{P}' - \mb{P}_*')$, ker se telo ne giblje v svojem sistemu in je zato $Q
\dot{\mb{P}} = 0$.
Na koncu dobimo
\[
  T = \pol m \abs{\dot{\mb{P}}_*'}^2 + \pol \vec{\omega} \cdot J(\mb{P}_*)
  \vec{\omega}.
\]

\vprasanje{Izpelji predpis za kinetično energijo togega telesa.}

Pri dinamiki togega telesa imamo tri principe:
\begin{itemize}
\item $m \ddot{\mb{P}}_*' = \vec{F}_{\text{zun}}'$ (enačba gibanja masnega
  središča)
\item $\partial_t \pvec{l}'(\mb{P}_*') = \pvec{N}'(\mb{P}_*')$ (princip o
  vrtilni količini)
\item $\pvec{N}'(\mb{P}_0') = \pvec{N}'(\mb{P}_*') + (\mb{P}_*' - \mb{P}_0')
  \times \vec{F}_{\text{zun}}'$
\end{itemize}

\begin{izrek}
  Naj bo $\mb{P}_0'$ točka trenutnega ali stalnega mirovanja togega telesa.
  Potem velja $\partial_t \pvec{l}'(\mb{P}_0') = \pvec{N}'(\mb{P}_0')$.
\end{izrek}

\begin{proof}
  Računamo
  \begin{align*}
	\pvec{l}'(\mb{P}_0')
	&= Q \vec{l}(\mb{P}_0) \\
	&= Q J(\mb{P}_0) \vec{\omega} \\
	&= Q(J(\mb{P}_*) + m \abs{\mb{P}_0 - \mb{P}_*}^2 I -
	  m (\mb{P}_0 - \mb{P}_*) \otimes (\mb{P}_0 - \mb{P}_*)) \vec{\omega} \\
	&= Q \vec{l}(\mb{P}_*) + m \abs{\mb{P}_0 - \mb{P}_*}^2 \pvec{\omega}
	  - m((\mb{P}_0 - \mb{P}_*) \cdot \vec{\omega}) . (\mb{P}_0' - \mb{P}_*') \\
	&= \pvec{l}'(\mb{P}_*') + m ((\mb{P}_0' - \mb{P}_*') \times \pvec{\omega}')
	  \times (\mb{P}_0' - \mb{P}_*') \\
	&= \pvec{l}'(\mb{P}_*') + m \dot{\mb{P}}_*' \times (\mb{P}_0 - \mb{P}_*).
  \end{align*}
  Zadnji sklep uporabi dejstvo $\dot{\mb{P}}_*' = \dot{\mb{P}}_0' +
  \pvec{\omega}' \times (\mb{P}_*' - \mb{P}_0')$, ki ga dobimo z odvajanjem
  $\mb{P}_*' = \mb{P}_0' + Q(\mb{P}_* - \mb{P}_0)$.
  Upoštevamo še dejstvo, da je $\dot{\mb{P}}_0' = 0$, ker je to točka trenutnega
  mirovanja.
  Zgornjo enakost nato še odvajamo do
  \begin{align*}
	\partial_t \pvec{l}'(\mb{P}_0')
	&= \pvec{N}'(\mb{P}_*') + m \ddot{\mb{P}}_*' \times (\mb{P}_0 - \mb{P}_*)
	  + m \dot{\mb{P}}_*' \times (- \dot{\mb{P}}_*') \\
	&= \pvec{N}'(\mb{P}_*') + (\mb{P}_*' - \mb{P}_0') \times \pvec{F}_{\text{zun}}' \\
	&= \pvec{N}'(\mb{P}_0').
  \end{align*}
\end{proof}

Navor okoli masnega središča ali navor okoli točke mirovanja lahko razpišemo v
\begin{align*}
  \pvec{N}'(\mb{P}_*')
  &= \partial_t \pvec{l}'(\mb{P}_*') \\
  &= \partial_t (Q J(\mb{P}_*) \vec{\omega}) \\
  &= \dot{Q} J(\mb{P}_*) \vec{\omega}
	+ Q J(\mb{P}_*) \dot{\vec{\omega}} \\
  &= \pvec{\omega}' \times Q J(\mb{P}_*) \vec{\omega}
	+ Q J(\mb{P}_*) \dot{\vec{\omega}} \\
  &= Q (\vec{\omega} \times J(\mb{P}_*) \vec{\omega}
	+ J(\mb{P}_*) \dot{\vec{\omega}}).
\end{align*}
Iz tega dobimo Eulerjeve dinamične enačbe
\[
  \vec{N}(\mb{P}_*) = J(\mb{P}_*) \dot{\vec{\omega}} + \vec{\omega} \times
  J(\mb{P}_*) \vec{\omega}.
\]
Enako dobimo za točko trenutnega mirovanja.

\vprasanje{Izpelji Eulerjeve dinamične enačbe. Dokaži, da delujejo tudi v točki
  mirovanja.}

V lastnem KS vztrajnostnega momenta je $J$ diagonalen.
Dobimo
\begin{gather*}
  J_1 \dot{\omega}_1 - \omega_2 \omega_3 (J_2 - J_3) = N_1 \\
  J_2 \dot{\omega}_2 - \omega_1 \omega_3 (J_3 - J_1) = N_2 \\
  J_3 \dot{\omega}_3 - \omega_1 \omega_2 (J_1 - J_2) = N_3
\end{gather*}
Podobno lahko naredimo za rotacijo okoli stalne osi $\vec{e}$
\begin{gather*}
  J_{13} \ddot{\varphi} - J_{23} \dot{\varphi}^2 = N_1 \\
  J_{23} \ddot{\varphi} + J_{13} \dot{\varphi}^2 = N_2 \\
  J_{33} \ddot{\varphi} = N_3.
\end{gather*}

\vprasanje{Izpelji Eulerjeve dinamične enačbe v lastnem koordinatnem sistemu in
  za rotacijo okoli stalne osi.}

Poznamo volumenske sile $(B_1, \vec{f}_1)$, za katere velja
\[
  \vec{F}_1 = \iiint_{B_1} \vec{f}_1 dV.
\]

\begin{trditev}
  Homogena volumenska sila je ekvivalentna točkovni sili s prijemališčem v
  masnem središču.
\end{trditev}

\begin{proof}
  Za silo je očitno, za navor pa velja
  \[
	\vec{N}(\mb{O})
	= \iiint_B (\mb{P} - \mb{O}) \times \vec{f} dm
	= \left( \iiint_B \mb{P} - \mb{O} dm \right) \times \vec{f}
	= (\mb{P}_* - \mb{O}) \times \vec{F}.
  \]
\end{proof}

Za odvod kinetične energije velja
\[
  \partial_t T
  = m \dot{\mb{P}}_*' \cdot \ddot{\mb{P}}_*' + \vec{\omega} \cdot J_*
  \dot{\vec{\omega}}
  = \dot{\mb{P}}_*' \cdot \pvec{F}' + \pvec{\omega}' \cdot \pvec{N}_*',
\]
kjer smo upoštevali Eulerjeve dinamične enačbe.
Recimo, da je $\vec{F}$ volumenska sila, in da velja $\pvec{f}' = - \grad u$.
Definiramo
\[
  U = \iiint_{B'} u(\mb{P}') dm'
\]
in računamo
\[
  \partial_t U
  = \iiint_{B'} \partial_t u(\mb{P}') dm'
  = \iiint_{B'} \grad u \cdot \dot{\mb{P}}' dm
  = - \iiint_B \pvec{f}' \cdot (\dot{\mb{P}}_*' + \pvec{\omega} \times (\mb{P}'
  - \mb{P}_*')) dm
\]
Sedaj ciklično zamenjamo člene mešanega produkta v drugem členu, do
\[
  \partial_t U
  = - \pvec{f}' \cdot \dot{\mb{P}}_*'
  - \iiint_{B'} \pvec{\omega} \cdot ((\mb{P}' - \mb{P}_*') \times \pvec{f}') dm
  = - \pvec{f}' \cdot \dot{\mb{P}}_*' - \pvec{\omega}' \cdot \pvec{N}_*'.
\]
Torej je $T + U$ konstantna.
To je izrek o energiji za togo telo.

\vprasanje{Izpelji izrek o energiji za togo telo.}

\begin{definicija}
  Telo $B_1$ se kotali po $B_2$, če imata telesi v dotikališču enako hitrost.
\end{definicija}

\podnaslov{Prosta vrtavka}

Vrtavka je prosta, če je $\pvec{F}' = 0$ in $\pvec{N}' = 0$.
Eulerjeve dinamične enačbe imajo tedaj obliko
\begin{gather*}
  J_1 \dot{\omega}_1 - \omega_2 \omega_3 (J_2 - J_3) = 0, \\
  J_2 \dot{\omega}_2 - \omega_1 \omega_3 (J_3 - J_1) = 0, \\
  J_3 \dot{\omega}_3 - \omega_1 \omega_2 (J_1 - J_2) = 0.
\end{gather*}
Enačba ima več rešitev:
\begin{itemize}
\item Trivialna: $\omega_1 = \omega_2 = \omega_3 = 0$
\item Enakomerno vrtenje okoli ene osi: $\omega_1 = \omega_2 = 0$, $\omega_3 = \text{konst.}$.
\item Netrivialne rešitve
\end{itemize}

Če je vrtavka simetrična, $J_1 = J_2$, je $\omega_3$ konstantna, in velja
\begin{gather*}
  \dot{\omega}_1 - \frac{\omega_3 (J_1 - J_3)}{J_1} \omega_2 = 0 \\
  \dot{\omega}_2 - \frac{\omega_3 (J_3 - J_1)}{J_1} \omega_1 = 0
\end{gather*}
Za $\Omega = \omega_3 (J_1 - J_3) / J_1$ je rešitev sistema
\begin{gather*}
  \omega_1 = A \cos(\Omega t - \delta) \\
  \omega_2 = - A \sin(\Omega t - \delta)
\end{gather*}
Vektor $\vec{\omega}$ torej precesira s kotno hitrostjo $\Omega$ okoli tretje
osi.

\vprasanje{Reši prosto simetrično vrtavko.}

Če se preselimo v koordinatni sistem, v katerem je $J$ diagonalen, velja $l_i =
J_i \omega_i$.
Velja $l^2 = l_1^2 + l_2^2 + l_3^2 = \text{konst.}$ in
\[
  T = \pol J_1 \omega_1^2 + \pol J_2 \omega_2^2 + \pol J_3 \omega_3^2
  = \frac{l_1^2}{2 J_1} + \frac{l_2^2}{2 J_2} + \frac{l_3^2}{2 J_3}.
\]
Iz tega dobimo
\[
  1 = \frac{l_1^2}{2 T J_1} + \frac{l_2^2}{2 T J_2} + \frac{l_3^2}{2 T J_3},
\]
čemur pravimo \pojem{Binetov elipsoid}.
Vektor $\vec{\omega}$ leži na preseku sfere $\abs{\vec{l}} = l$ in Binetovega
elipsoida.
Če uredimo $J_1 \le J_2 \le J_3$, dobimo $1 \le l^2 / (2 T J_1)$, torej za polos
velja $a_1 \le l$.
Podobno dobimo $a_3 \ge l$.
Če je rotacija le okoli največje ali najmanjše osi, se sfera in elipsoid
dotikata v točki; za malo zmoten elipsoid imamo v preseku dve krožnici.
Za srednjo polos pa majhna perturbacija elipsoida razklene točko v dve krivulji,
ki prideta daleč od točke.
Enakomerna rotacija okoli srednje osi je torej nestabilna.

\vprasanje{Analiziraj stabilnost enakomerne rotacije proste vrtavke.}

\podnaslov{Eulerjevi koti}

Rotacijo predstavimo kot kompozitum rotacij
\[
  R = R(\pvec{k}'', \psi) R(\pvec{i}', \theta) R(\vec{k}, \varphi).
\]

\begin{lema}
  Za $\pvec{f}' = R(\vec{e}, \varphi) \vec{f}$ velja $R(\pvec{f}, \psi)
  R(\vec{e}, \varphi) = R(\vec{e}, \varphi) R(\vec{f}, \varphi)$.
\end{lema}

\begin{trditev}
  Velja $R(\pvec{k}'', \psi) R(\pvec{i}', \theta) R(\vec{k}, \varphi) =
  R(\vec{k}, \varphi) R(\vec{i}, \theta) R(\vec{k}, \psi)$.
\end{trditev}

Dokaz za lemo in trditev je preprost račun.

\begin{trditev}
  Vsako rotacijo, ki ni rotacija okoli osi $\vec{k}$, lahko enolično napišemo z
  Eulerjevo rotacijo s koti $\theta, \varphi, \psi \in [0, 2 \pi)$.
\end{trditev}

\begin{proof}
  Naj bo $\vec{\varepsilon}_1, \vec{\varepsilon}_2, \vec{\varepsilon}_3$ rotiran
  koordinatni sistem.
  Za vozliščnico $\vec{k} \times \vec{\varepsilon}_3 = \vec{e}$ definiramo kote
  \begin{itemize}
  \item $\varphi$ je kot med $\vec{i}$ in $\vec{e}$,
  \item $\psi$ je kot med $\vec{e}$ in $\vec{\varepsilon}_1$,
  \item $\theta$ je kot med $\vec{k}$ in $\vec{\varepsilon}_3$.
  \end{itemize}
  Za dokaz enoličnosti pogledamo delovanje preslikave
  \[
	R(\vec{i}, \theta_1) = R(\vec{k}, \varphi_2 - \varphi_1) R(\vec{i},
	\theta_2) R(\vec{k}, \psi_2 - \psi_1)
  \]
  na vektorjih $\vec{i}$ in $\vec{k}$.
\end{proof}

\vprasanje{Dokaži, da se lahko vsako rotacijo zapiše na enoličen način z
  Eulerjevimi rotacijami.}

\begin{trditev}
  Vektor kotne hitrosti Eulerjeve rotacije je $\pvec{\omega}' = \dot{\varphi}
  \vec{k} + \dot{\theta} \pvec{i}' + \dot{\psi} \pvec{k}''$.
\end{trditev}

\begin{proof}
  Označimo $R = R(\vec{k}, \varphi) R(\vec{i}, \theta) R(\vec{k}, \psi) = R_1
  R_2 R_3$.
  Po definiciji kotne hitrosti je
  \begin{align*}
	\dot{R} \vec{a}
	&= \pvec{\omega}' \times R \vec{a} \\
	&= (\dot{R}_1 R_2 R_3 + R_1 \dot{R}_2 R_3 + R_1 R_2 \dot{R}_3) \vec{a} \\
	&= \pvec{\omega}_1' \times R_1 R_2 R_3 \vec{a}
	  + R_1 (\pvec{\omega}_2' \times R_2 R_3 \vec{a})
	  + R_1 R_2 (\pvec{\omega}_3' \times R_3 \vec{a}) \\
	&= (\pvec{\omega}_1' + R_1 \pvec{\omega}_2' + R_1 R_2 \pvec{\omega}_3')
	  \times R \vec{a}.
  \end{align*}
  Sledi
  \begin{align*}
	\pvec{\omega}'
	&= \pvec{\omega}_1' + R_1 \pvec{\omega}_2' + R_1 R_2 \pvec{\omega}_3' \\
	&= \dot{\varphi} \vec{k} + R_1 \dot{\theta} \vec{i} + R_2' R_1 \dot{\phi} \vec{k} \\
	&= \dot{\varphi} \vec{k} + \dot{\theta} \pvec{i}' + \dot{\psi} \pvec{k}''.
  \end{align*}
\end{proof}

\vprasanje{Kako izraziš vektor kotne hitrost Eulerjeve rotacije? Dokaži.}

Če razpišemo $\pvec{k}''$, $\pvec{i}'$ in $\pvec{j}'$, dobimo
\[
  \pvec{\omega}'
  = (\dot{\theta} \cos \varphi + \dot{\psi} \sin \varphi \sin \theta) \vec{i}
  + (\dot{\theta} \sin \varphi - \dot{\psi} \cos \varphi \sin \theta) \vec{j}
  + (\dot{\varphi} + \dot{\psi} \cos \theta) \vec{k},
\]
po drugi strani pa je $\pvec{\omega}' = Q \vec{\omega}$.
Za rotacijo $Q$ velja
\[
  \vec{\omega}(Q) \times \vec{a} = Q^T \dot{Q} \vec{a} = - \dot{Q}^T Q \vec{a}
  = - \pvec{\omega}'(Q^T) \times Q^T Q \vec{a} = - \pvec{\omega}'(Q^T) \times \vec{a},
\]
torej $\vec{\omega}(Q) = - \pvec{\omega}'(Q^T)$.
Sledi
\[
  \vec{\omega}
  = (\dot{\theta} \cos \psi + \dot{\varphi} \sin \theta \sin \psi) \vec{i}
  + (- \dot{\theta} \sin \psi + \dot{\varphi} \sin \theta \cos \psi) \vec{j}
  + (\dot{\psi} + \dot{\varphi} \cos \theta) \vec{k}
\]
oziroma
\[
  \pvec{\omega}'
  = (\dot{\theta} \cos \psi + \dot{\varphi} \sin \theta \sin \psi) \vec{\varepsilon}_1
  + (- \dot{\theta} \sin \psi + \dot{\varphi} \sin \theta \cos \psi) \vec{\varepsilon}_2
  + (\dot{\psi} + \dot{\varphi} \cos \theta) \vec{\varepsilon}_3,
\]
kjer je $\vec{\varepsilon}_1 = \pvec{i}'''$ in podobno za drugi komponenti.

\vprasanje{Izpelji predpis za $\pvec{\omega}'$ v rotirani bazi.}

\begin{trditev}
  Ne obstaja parametrizacija $SO(3)$, da je $\pvec{\omega}' = \dot{\vec{x}}$.
\end{trditev}

\begin{proof}
  Če bi obstajala taka parametrizacija $\vec{x} = \vec{x}(\vec{y}) =
  \vec{x}(\varphi, \theta, \psi)$, potem velja
  \[
	\pvec{\omega}' = \dot{\vec{x}} = \frac{\partial \vec{x}}{\partial \vec{y}}
	\dot{\vec{y}} =
	\begin{bmatrix}
	  \sin \theta \sin \psi & \cos \psi & 0 \\
	  \sin \theta \cos \psi & - \sin \psi & 0 \\
	  \cos \theta & 0 & 1
	\end{bmatrix}
	\dot{\vec{y}}.
  \]
  Računamo lahko
  \begin{align*}
	\frac{\partial^2 x_1}{\partial y_1 \partial y_2} = \cos \theta \sin \psi
	&& \frac{\partial^2 x_1}{\partial y_2 \partial y_1} = 0.
  \end{align*}
  To ne mora biti res.
\end{proof}
