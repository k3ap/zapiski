\naslov{PremoÄŤrtno gibanje}

\begin{definicija}
  Gibanje je \pojem{premoÄŤrtno}, ÄŤe ima pospeĹˇek konstantno smer.
\end{definicija}

Primer takega gibanja je poĹˇevni met.
Opazimo, da lahko vedno izberemo koordinatni sistem, v katerem tir poti leĹľi na
premici:
ÄŚe je $\vec{a} = a \vec{e}$, kjer je $\vec{e}$ konstanten vektor, velja
\[

\]
Izberemo lahko sistem, kjer je $\vec{v}_0$ enak $\vec{0}$, in bo torej $\vec{v}$
vzporeden $\vec{e}$.

ÄŚe gibanje poteka pod vplivom konzervativne sile, lahko zapiĹˇemo potencial $U$,
in velja izrek o energiji
\[
  \pol m \dot{x}^2 + U(x) = E_0.
\]
Od tod izpeljemo
\[
  \dot{x} = \pm \sqrt{\frac{2}{m} \left( E_0 - U(x) \right) }.
\]
EnaÄŤbo z loÄŤljivimi spremenljivkami tedaj integriramo in dobimo
\[
  \pm \int_{x_0}^{x} \frac{dx}{\sqrt{ \frac{2}{m} \left( E_0 - U(x) \right) } }
  = \int_{t_0}^t dt = t - t_0.
\]
Dobimo funkcijo $t = t(x)$.
ÄŚe se na poti ne ustavimo, po izreku o inverzni preslikavi obstaja funkcija $x
= x(t)$.
Pravimo, da je premoÄŤrtno gibanje \pojem{integrabilno}.

ÄŚe v kvalitativni analizi ugotovimo, da je neko gibanje periodiÄŤno med toÄŤkama
$a$ in $b$, lahko periodo izraÄŤunamo kot
\begin{align*}
  T &= \int_{x_0}^b \frac{dx}{\sqrt{ \frac{2}{m} \left( E_0 - U(x) \right) }}
  - \int_a^b \frac{dx}{\sqrt{ \frac{2}{m} \left( E_0 - U(x) \right) }}
  + \int_a^{x_0} \frac{dx}{\sqrt{ \frac{2}{m} \left( E_0 - U(x) \right) }} \\
  &= \sqrt{2 m} \int_a^b \frac{dx}{ \sqrt{ E_0 - U(x) } }.
\end{align*}
Ker je $E = U(x)$ v krajiĹˇÄŤih, je to posploĹˇen integral.
Situacija v obeh krajiĹˇÄŤih je simetriÄŤna, torej preverimo le za levo krajiĹˇÄŤe,
da integral res konvergira.
V prvem koraku razvijemo preslikavo $U$ v Taylorjev polinom prve stopnje v toÄŤki
$a$, kjer se pojavi vrednost odvoda $U$ v neki toÄŤki $\xi$ blizu $a$.
Ker je odvod zvezen, obstaja tak $\delta > 0$, da za $\xi \in [a, a + \delta)$
velja $2 \partial_x U(a) < \partial_x U(\xi) < \pol \partial_x U(a)$, torej
\[
  \int_a^{a + \delta} \frac{dx}{\sqrt{E_0 - U(x)}}
  = \int_a^{a + \delta} \frac{dx}{\sqrt{ - \partial_x U(\xi) (x-a) }}
  \le \int_a^{a + \delta} \inv{\sqrt{ -\pol \partial_x U(a) }} \inv{\sqrt{x -
	  a}} dx
  < \infty.
\]

\vprasanje{Izpelji izraz za periodo premoÄŤrtnega potencialnega gibanja.}

\begin{lema}
  Za $a < b$ velja
  \[
	\int_a^b \frac{dx}{\sqrt{(b-x) (x-a)}} = \pi.
  \]
\end{lema}

\begin{proof}
  Uvedemo novo spremenljivko $x = \pol (a+b) + \pol (b-a) z$, s ÄŤimer se
  integral spremeni v
  \[
	\int_{-1}^1 \frac{dz}{\sqrt{(1-z)(1+z)}}
	= \pi.
  \]
\end{proof}

\begin{primer}
  Oglejmo si harmoniÄŤni oscilator, ki deluje pod potencialom $U = \pol k x^2$.
  Tedaj velja $F = - \partial_x U = -kx$, torej $m\ddot{x} = -kx$.
  ReĹˇitev tega sistema je $x = A \cos \omega t + B \sin \omega t$ za
  $\omega = \sqrt{k/m}$.
  Iz tega lahko kar direktno preberemo $T = 2 \pi / \omega$.
  Posebnost harmoniÄŤnega oscilatorja je, da je $T$ neodvisen od $E_0$.
  Takemu gibanju pravimo \pojem{izohroniÄŤno}, harmoniÄŤni potencial je edini
  primer izohroniÄŤnega potenciala, ki je simetriÄŤen glede na svoj minimum.
\end{primer}

ÄŚe ima potencial lokalni minimum v $x_0$, lahko za doloÄŤanje potenciala
uporabimo harmoniÄŤno aproksimacijo.
ZapiĹˇemo
\[

  U(x_0) (x - x_0)^2,
\]
kar je harmoniÄŤni potencial s periodo
\[
  T = 2 \pi \sqrt{\frac{m}{\partial_{x^2} U(x_0)}}.
\]
Ta aproksimacija je dobra, ÄŤe velja $E_0 - U(x) \ll 1$.

\vprasanje{Izpelji harmoniÄŤno aproksimacijo.}

\[

toÄŤki $a$ in $b$ in ima srediĹˇÄŤe na njuni zveznici.
ÄŚe raÄŤunamo dalje, dobimo

\vprasanje{Izpelji libracijsko aproksimacijo.}

\begin{trditev}

  \frac{T}{m}$ na poti med robnima toÄŤkama energijskega nivoja.
\end{trditev}

\begin{proof}
  Velja $A = 2 \sqrt{\frac{m}{2}} \int_a^b \sqrt{E_0 - U} dx$, torej
  \[
	\frac{dA}{dE_0} = 2 \sqrt{\frac{m}{2}} \left(
	  b' \sqrt{E_0 - U(b)} - a' \sqrt{E_0 - U(a)} + \int_a^b
	  \frac{dx}{2\sqrt{E_0 - U}}

  \]
  Ker je $U(a) = U(b) = E_0$, sta prva dva ÄŤlena v oklepaju enaka $0$, torej
  \[
	\frac{dA}{dE_0} = \sqrt{\frac{m}{2}} \int_a^b \frac{dx}{\sqrt{E_0-U}}
	= \frac{m}{T}.
  \]
\end{proof}
