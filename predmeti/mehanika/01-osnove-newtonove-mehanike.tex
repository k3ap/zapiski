\naslov{Osnove Newtonove mehanike}

\begin{definicija}
  \pojem{Afin prostor} $\affin$ nad vektorskim prostorom $V$ je množica z
  binarno operacijo $+: \affin \times V \to \affin$, za katero velja:
  \begin{itemize}
  \item Za poljuben $\mb{A} \in \affin$ ter $a, b \in V$ velja $(\mb{A} +
    a) + b = \mb{A} + (a + b)$
  \item Za poljubna $\mb{A}, \mb{B} \in \affin$ obstaja natanko določen $a \in
    V$, da je $\mb{B} = \mb{A} + a$.
  \end{itemize}
  \pojem{Dimenzija} afinega prostora je enaka dimenziji vektorskega prostora
  $V$.
\end{definicija}

\begin{definicija}
  Naj bo $\affin$ afin prostor nad vektorskih prostorom $V$.
  Definiramo operacijo odštevanja $\affin \times \affin \to V$ s
  predpisom
  \[
    \mb{B} - \mb{A} = a \nt \mb{B} = \mb{A} + a.
  \]
\end{definicija}

\begin{trditev}
  V afinem prostoru veljajo naslednje zveze:
  \begin{itemize}
  \item $\mb{A} - \mb{A} = 0$.
  \item $(\mb{A} - \mb{B}) + (\mb{B} - \mb{A}) = 0$.
  \item $(\mb{A} - \mb{B}) + (\mb{B} - \mb{C}) + (\mb{C} - \mb{A}) = 0$.
  \item $(\mb{A} - \mb{B}) + a = (\mb{A} + a) - \mb{B}$.
  \item $(\mb{A} - \mb{B}) + \mb{C} = (\mb{C} - \mb{B}) + \mb{A}$.
  \end{itemize}
\end{trditev}

\begin{definicija}
  Preslikava $g : \affin \to \affin'$ med afinima prostoroma je
  \pojem{afina}, če obstaja $dg \in L(V, V')$, da za vsaka $\mb{A}, \mb{B} \in
  \affin$ velja $g(\mb{A}) - g(\mb{B}) = dg(\mb{A} - \mb{B})$.
\end{definicija}

Za afino preslikavo $g$ si lahko izberemo \pojem{pol} $\mb{O}$, ter izpeljemo
\[
  g(\mb{A}) = g(\mb{O}) + dg(\mb{A} - \mb{O}).
\]
Vrednosti funkcije seveda niso odvisne od izbire pola.

\vprasanje{Definiraj afin prostor in afino preslikavo.}

\newcommand{\cas}{\mathfrak{t}}
\begin{definicija}
  \pojem{Galilejeva struktura} je trojica $\mathcal{G} = (\affin,
  \cas, \rho)$, kjer je $\affin$ štirirazsežni afin prostor nad $V$, $\cas \in
  L(V, \R)$ in $\rho$ ekvlidska metrika na $\jedro \cas$, porojena z normo
  $\norm{\cdot}$.
  Funkciji $\cas$ pravimo \pojem{časovnost}, elementom $\affin$ pa pravimo
  \pojem{dogodki}.
  Pretečeni čas med dogodkoma $\mb{A}$ in $\mb{B}$ označimo s $\cas(\mb{A}, \mb{B})$.
  Dogodka sta \pojem{istočasna}, če je $\cas(\mb{A}, \mb{B}) = 0$.
  Za istočasne dogodke lahko definiramo razdaljo $\rho(\mb{A}, \mb{B}) = \norm{\mb{B} - \mb{A}}$
  (uporabimo isto oznako kot za metriko v $\jedro\cas$).
\end{definicija}

\begin{definicija}
  Galilejevi strukturi $\mathcal{G} = (\affin, \cas, \rho)$ in $\mathcal{G}' =
  (\affin', \cas', \rho')$ sta \pojem{ekvivalentni}, če obstaja afina bijekcija
  $g: \affin \to \affin'$, ki ohranja časovnost in razdaljo med istočasnimi
  dogodki;
  \begin{align*}
    \cas'(g(\mb{A}) - g(\mb{B})) = \cas(\mb{A}, \mb{B}),
    &&
       \rho'(g(\mb{A}), g(\mb{B})) = \rho(\mb{A}, \mb{B}).
  \end{align*}
  Taki transformaciji pravimo \pojem{Galilejeva transformacija}.
\end{definicija}

\vprasanje{Definiraj Galilejevo strukturo in Galilejeve transformacije.}

\newcommand{\galileo}{\R\times\mathbb{E}}
Modelni primer je naravna Galilejeva struktura na $\affin = \galileo$, kjer je
$\mathbb{E}$ trirazsežni Evklidski prostor.
Za elemente $A_i = (t_i, \mb{P}_i) \in \affin$ naravne strukture velja
\begin{itemize}
\item $\cas(A_1 - A_2) = t_1 - t_2$,
\item $\rho(A_1, A_2) = \norm{\mb{P}_1 - \mb{P}_2}$.
\end{itemize}

\begin{definicija}
  \pojem{Koordinatni sistem} na $\affin$ je bijekcija $\phi : \affin \to
  \galileo$ s komponentami $\phi(A) = (\tau \phi(A), \pi \phi(A))$, in pri
  kateri je $\tau \circ \phi$ linearna preslikava.
\end{definicija}

\begin{opomba}
  Če sta $\phi$ in $\phi'$ koordinatna sistema, je preslikava $\phi' \circ
  \phi^{-1}: \galileo \to \galileo$ bijekcija.
\end{opomba}

\vprasanje{Kaj je koordinantni sistem?}

\begin{izrek}
  Galilejeva transformacija $g: \galileo \to \galileo$ je oblike
  \[
    g(t, \mb{P}) = (t_0' + t, \mb{P}_0' + \vec{c} t + Q(\mb{P} -
    \mb{P}_0)),
  \]
  kjer je $Q \in O(3)$ ortogonalna transformacija.
\end{izrek}

\begin{proof}
  Ker je $g$ afina preslikava, jo lahko zapišemo kot
  \[
    g(t, \mb{P}) = g(t_0, \mb{P}_0) + dg(t - t_0, \mb{P} - \mb{P}_0),
  \]
  kjer je $dg \in L(\R^4, \R^4)$.
  Če označimo $g(t_0, \mb{P}_0) = (t_0', \mb{P}_0')$, in zapišemo $dg$ kot
  bločno matriko, dobimo
  \[
    g(t, \mb{P})
    =
    (t_0', \mb{P}_0')
    +
    \begin{bmatrix}
      \alpha & \vec{a}^T \\
      \vec{c} & Q
    \end{bmatrix}
    \begin{bmatrix}
      t - t_0 \\ \mb{P} - \mb{P}_0
    \end{bmatrix}
    =
    (t_0', \mb{P}_0')
    +
    \begin{bmatrix}
      \alpha (t - t_0) + \vec{a} \cdot (\mb{P} - \mb{P}_0) \\
      (t - t_0).\vec{c} + Q (\mb{P} - \mb{P}_0)
    \end{bmatrix}.
  \]
  Za dogodka $(t_1, \mb{P}_1)$ in $(t_2, \mb{P}_2)$ zahtevamo
  \[
    t_2 - t_1 = \tau(g(t_2, \mb{P}_2) - g(t_1, \mb{P}_1)).
  \]
  Če razvijemo desno stran zahteve po izpeljani formuli, dobimo pogoj
  \[
    t_2 - t_1 = \alpha (t_2 - t_1) + \vec{a} \cdot (\mb{P}_2 - \mb{P}_1).
  \]
  Iz tega sledi $\alpha = 1$ in $\vec{a} = \vec{0}$.
  Drug pogoj je, da se mora razdalja med istočasnimi dogodki ohranjati.
  Iz spodnjega dela bločne matrike dobimo pogoj
  \[
    \norm{\mb{P}_2 - \mb{P}_1} = \norm{ Q (\mb{P}_2 - \mb{P}_1)},
  \]
  torej mora biti $Q$ ortogonalna.
\end{proof}

\vprasanje{Kakšno obliko imajo Galilejeve transformacije $\galileo \to
  \galileo$? Dokaži.}

Če definiramo $\vec{v} = \dot{\mb{P}}$ in $\vec{a} = \dot{\vec{v}}$, lahko
opazujemo, kako se ti količini obnašata pri Galilejevi transformaciji.
V koordinatnem sistemu $\phi'(t', \mb{P}')$ velja $\pvec{v}' = \partial_{t'}
\mb{P}' = \dot{\mb{P}}'$ in $\pvec{a}' = \dot{\pvec{v}}'$.
Izpeljemo $\pvec{v}' = \vec{c} + Q \dot{\mb{P}}(t' - t_0') = \vec{c} + Q
\dot{\mb{P}}(t)$ in $\pvec{a}' = Q \ddot{\mb{P}}(t)$.

\newcommand{\umb}[1]{\underline{\mb{#1}}}

Za sistem materialnih točk $\mathcal{P} = \{\mb{P}_1, \ldots, \mb{P}_n\}$ lahko
definiramo
\begin{gather*}
  \umb{P} = (\mb{P}_1, \ldots, \mb{P}_n) \\
  \umb{P}_0' = (\mb{P}_0', \ldots, \mb{P}_0') \\
  \underline{\vec{c}} = (\vec{c}, \ldots, \vec{c}) \\
  \umb{P}' = (\mb{P}_1', \ldots, \mb{P}_n') = \umb{P}_0' + \underline{\vec{c}} t
  + \underline{Q} (\umb{P} - \umb{P}_0)
\end{gather*}
Gibanje lahko tedaj zapišemo s tremi principi.
\begin{itemize}
\item \emph{Princip determiniranosti}: Trajektorija sistema materialnih točk
  $\mathcal{P}$ je v danem koordinantnem sistemu natanko določena z začetnim
  položajem in hitrostjo.
  To pomeni, da obstaja funkcija interakcije $\underline{\vec{f}}$, da velja
  \[
    \ddot{\umb{P}} = \underline{\vec{f}}(t, \umb{P}, \dot{\umb{P}}).
  \]
\item \emph{Princip relativnosti}: Obstaja tak razred koordinatnih sistemov, v
  katerem je funkcija interakcije invariantna na Galilejeve transformacije.
  Temu razredu pravimo \pojem{razred inercialnih koordinatnih sistemov}.
  To pomeni, da je funkcija interakcije invariantna v tem razredu,
  \[
    \ddot{\umb{P}}' = \underline{\vec{f}}(t', \umb{P}', \dot{\umb{P}}').
  \]
\item \emph{Princip o sorazmernosti}: Obstajajo pozitivne konstante
  $\alpha_{ij}$, da za vsako interakcijo med materialnimi točkami sistema
  $\mathcal{P} = (\mb{P}_1, \ldots, \mb{P}_n)$ velja
  \[
    \vec{f}_i = - \sum_{j \ne i} \alpha_{ji} \vec{f}_{j}.
  \]
  Te konstante so enake za vse možne interakcije v sistemu.
\end{itemize}

\vprasanje{Kateri so principi gibanja?}

Z ozirom na princip relativnosti izpeljemo $\underline{Q} \ddot{\umb{P}} =
\underline{Q} \underline{\vec{f}}(t, \umb{P}, \dot{\umb{P}})$.
Če v to enakost vstavimo vrednosti $t' = t_0' + t$, $\vec{c} = \vec{0}$, $Q = I$
ter $\mb{P}_0' = \mb{P}_0$, dobimo $\underline{\vec{f}}(t_0' + t, \umb{P},
\dot{\umb{P}}) = \underline{\vec{f}}(t, \umb{P}, \umb{P}_0)$, kar mora veljati
za vsak $t_0'$.
Sledi, da funkcija $\vec{f}$ ne more biti eksplicitno odvisna od časa.
Tej ugotovitvi pravimo \pojem{homogenost časa}.

Če sedaj vstavimo $\vec{c} = \vec{0}$, $Q = I$ in $\mb{P}_0' = \mb{P}_0 +
\vec{a}$, kjer je $\vec{a}$ poljuben vektor (in ne pospešek), izpeljemo
$\mb{P}' = \mb{P} + \vec{a}$, in sledi $\underline{\vec{f}}(\umb{P} +
\underline{\vec{a}}, \dot{\umb{P}}) = \underline{\vec{f}}(\umb{P},
\dot{\umb{P}})$, torej $\vec{f}$ ne more biti odvisna od absolutnih položajev.
Seveda je še vedno lahko odvisna od relativnih položajev (v tem primeru se
$\vec{a}$ odšeteje).
Tej lastnosti pravimo \pojem{homogenost prostora}.

S poljubno izbiro vektorja $\vec{c}$ in $Q = I$ lahko podobno izpeljemo, da je
$\underline{\vec{f}}$ lahko odvisna le od relativnih hitrosti, čemur pravimo
\pojem{homogenost prostora hitrosti}.

Če nenazadnje relaksiramo še pogoj na $Q$, dobimo
\[
  \underline{\vec{f}}(Q(\mb{P}_i - \mb{P}_j), Q(\dot{\mb{P}}_i, \dot{\mb{P}}_j))
  = \underline{Q} \underline{\vec{f}}(\mb{P}_i - \mb{P}_j, \dot{\mb{P}}_i -
  \dot{\mb{P}}_j).
\]
Funkcijam, ki zadoščajo temu pogoju, pravimo \pojem{izotropične funkcije}.

V posebnem primeru za $n = 1$ je $\vec{f}$ konstantna funkcija (ker ne more biti
odvisna od ničesar).
Ker za vsak $Q \in O(3)$ velja $\vec{f} = Q \vec{f}$, mora biti $\vec{f} =
\vec{0}$.
Torej se prosta materialna točka v inercialnem koordinantem sistemu premika
premočrtno s konstantno hitrostjo.
To je ena od implikacij v prvem Newtonovem zakonu.

\vprasanje{Izpelji homogenost časa in faznega prostora iz principov gibanja.}

\begin{definicija}
  Interakcija $\vec{f}$ je \pojem{parska}, če lahko zapišemo
  \[
    \vec{f}_i = \sum_{j \ne i} \vec{f}_{ji}(\mb{P}_i - \mb{P}_i, \dot{\mb{P}}_j
    - \dot{\mb{P}}_i)
  \]
  za vse indekse $i$.
\end{definicija}

\begin{definicija}
  Interakcija $\vec{f}$ je \pojem{lokalna}, če je parska in če velja
  \[
    \lim_{\mb{P}_i - \mb{P}_j \to \infty} \vec{f}_{ji} = \vec{0}.
  \]
\end{definicija}

\vprasanje{Definiraj parske in lokalne interakcije.}

\begin{lema}
  Za števila $\alpha_{ij}$ iz principa sorazmernosti velja
  \begin{itemize}
  \item $\alpha_{ij} \alpha_{ji} = 1$,
  \item $\alpha_{ij} \alpha_{jk} \alpha_{kj} = 1$.
  \end{itemize}
\end{lema}

\begin{proof}
  Prva točka:
  Izberemo si take interakcije $\vec{f}_k$, ki so parske in lokalne in ki so
  neodvisne od relativnih hitrosti.
  Vse točke razen $i$ in $j$ pošljemo v neskončnost, da je njihov vpliv ničeln.
  Tedaj velja $\vec{f}_i = - \alpha_{ji} \vec{f}_j$ in $\vec{f}_j = -\alpha_{ij}
  \vec{f}_i$, torej $\vec{f}_i = \alpha_{ij} \alpha_{ji} \vec{f}_i$.

  Druga točka:
  Izberemo si indekse $i, j, k$ in podobno kot prej pošljemo druge točke v
  neskončnost.
  Ob predpostavki parske in lokalne interakcije tako dobimo
  \begin{gather*}
    \vec{f}_i = - \alpha_{ji} \vec{f}_j - \alpha_{ki} \vec{f}_k, \\
    \vec{f}_j = - \alpha_{ij} \vec{f}_i - \alpha_{kj} \vec{f}_k.
  \end{gather*}
  Če vstavimo drugo enačbo v prvo,
  \[
    \vec{f}_i = \alpha_{ji} \alpha_{ij} \vec{f}_i + \alpha_{ji} \alpha_{kj}
    \vec{f}_k - \alpha_{ki} \vec{f}_k,
  \]
  nam člen na levi in prvi člen na desni po prvi točki odpadeta.
  Dobljeno enačbo še pomnožimo z $\alpha_{ik}$ in nam ostane
  \[
    \vec{f}_k = \alpha_{ji} \alpha_{kj} \alpha_{ik} \vec{f}_k.
  \]
\end{proof}

\begin{lema}
  Naj za pozitivna števila $\alpha_{ij}$ velja ugotovitev prejšnje leme.
  Potem obstajajo števila $m_i$, da je $\alpha_{ji} = \nicefrac{m_j}{m_i}$.
\end{lema}

\begin{proof}
  Števila $\alpha_{ij}$ so definirana le za $i \ne j$.
  Definicijo lahko razširimo, da je $\alpha_{ii} = 1$.
  Definiramo $l_{ij} = \log \alpha_{ij}$.
  Velja $l_{ii} = 0$ in $l_{ij} = - l_{ji}$, poleg tega pa tudi $l_{ij} + l_{jk}
  + l_{ki} = 0$.

  Izberemo si indeks $i_0$, ki nam bo definiral enoto mase.
  Velja $l_{i_0 j} + l_{jk} + l_{k i_0} = 0$, kar odštejemo od prejšnje vsote
  treh členov in dobimo
  \[
    l_{ij} - l_{i_0 j} + l_{ki} - l_{k i_0} = 0.
  \]
  Od tu izpeljemo, da za poljubna $j$ in $k$ velja
  \[
    l_{ij} - l_{i_0 j} = l_{ik} - l_{i_0 k},
  \]
  torej je $n_{i i_0} = l_{ij} - l_{i_0 j}$ dobro definirana količina.
  Opazimo, da za $i = j$ velja $n_{i i_0} = l_{i i_0}$.

  Definiramo $m_i = \exp n_{i i_0}$.
  Sledi
  \[
    \log \alpha_{ij} = l_{ij} = l_{i_0 j} + n_{i i_0} = - l_{ji} + n_{i i_0} =
    -n_{j i_0} + n_{i i_0} = \log m_i - \log m_j = \log \frac{m_i}{m_j}.
  \]
\end{proof}

\begin{opomba}
  Številom $m_i$ pravimo \pojem{inercijske mase}.
\end{opomba}

\vprasanje{Kaj so inercijske mase? Dokaži, da res obstajajo.}

Produktu $m \vec{f} = \vec{F}$ pravimo \pojem{sila}.
Iz parskosti sledi
\[
  \vec{F}_i = \sum_{j \ne i} \vec{F}_{ji}(\mb{P}_i - \mb{P}_j, \dot{\mb{P}}_i -
  \dot{\mb{P}}_j).
\]
Naj velja $\sum_{i \ne j} \vec{F}_{ji} = \vec{0}$.
Predpostavimo, da so sile lokalne, in fiksiramo indeksa $k \ne l$.
Če vsa ostala telesa pošljemo v neskončnost, ostane
\[
  \vec{F}_{kl} + \vec{F}_{lk} = \vec{0}.
\]
S tem smo dokazali tretji Newtonov zakon.

\begin{trditev}[tretji Newtonov zakon]
  Če so vse sile parske in lokalne, velja $\vec{F}_{kl} = - \vec{F}_{lk}$.
\end{trditev}

Za nadaljevanje potrebujemo še dodaten princip gibanja, ki ga imenujemo
\emph{princip o masi}.
Pravi, da je inercijska masa enaka v vseh koordinatnih sistemih.

\vprasanje{Kaj je princip o masi?}

Najpreprostejši primer sile je gravitacija.
Med točkama $(m_1, \mb{P}_1)$ in $(m_2, \mb{P}_2)$ deluje sila
\[
  \vec{F}_{21} =  \frac{\kappa M_1 M_2}{\abs{\mb{P}_1 - \mb{P}_2}^2}
  \frac{\mb{P}_2 - \mb{P}_1}{\abs{\mb{P}_2 - \mb{P}_1}}.
\]
Številoma $M_1$ in $M_2$ pravimo \pojem{gravitacijski masi}.
Z eksperimentiranjem je Newton ugotovil, da so pravzaprav enake inercijskim
masam.

\begin{definicija}
  Zunanja sila $\vec{F} = \vec{F}(t, \mb{P}, \dot{\mb{P}})$ je
  \pojem{potencialna}, če obstaja potencial $U$, da je $\vec{F} = -
  \vec{\nabla}_{\mb{P}} U$.
\end{definicija}

\begin{definicija}
  \pojem{Delo} sile $\vec{F}$ pri gibanju materialne točke od $\mb{P}_1$ do
  $\mb{P}_2$ je krivuljni integral
  \[
	A = \int_{\mb{P}_1}^{\mb{P}_2} \vec{F} \cdot d\mb{P}
	= \int_{t_1}^{t_2} \vec{F} \cdot \dot{\mb{P}} dt,
  \]
  kjer smo pot parametrizirali s $\mb{P}(t)$.
  Produktu $\vec{F} \cdot \dot{\mb{P}}$ pravimo \pojem{moč}.
\end{definicija}

\begin{definicija}
  \pojem{Kinetična energija} $T$ je enaka $\pol m \abs{\dot{\mb{P}}}^2$.
\end{definicija}

Za rezultanto vseh sil $\vec{F}$ na telo $m$ lahko izpeljemo
\[
  A = \int_{t_1}^{t_2} \vec{F} \cdot \dot{\mb{P}} dt
  = \int_{t_1}^{t_2} m \ddot{\mb{P}} \cdot \dot{\mb{P}} dt
  = m \int_{t_1}^{t_2} \partial_t \left(\pol \dot{\mb{P}} \cdot \dot{\mb{P}}\right) dt
  = T_2 - T_1,
\]
kar lahko zapišemo v izrek.

\begin{izrek}[izrek o delu]
  Delo rezultante vseh sil je enako razliki kinetične energije telesa.
\end{izrek}

\vprasanje{Povej in dokaži izrek o delu.}

\begin{definicija}
  Sila je \pojem{konzervativna} v danem razredu inercialnih koordinatnih
  sistemov, če obstaja inercialni koordinatni sistem, v katerem je $\vec{F}$
  potencialna in odvisna samo od položaja.
\end{definicija}

Tedaj je $\vec{F}$ potencialna, torej velja $\vec{F} = - \grad U$ za nek
potencial $U$, ki mu pravimo \pojem{potencialna energija}.
Velja
\[
  A = \int_{t_1}^{t_2} \vec{F} \cdot \dot{\mb{P}} dt
  = - \int_{t_1}^{t_2} \grad U \cdot \dot{\mb{P}} dt
  = U(\mb{P}_1) - U(\mb{P}_2).
\]
Vidimo, da je delo odvisno le od začetnega in končnega položaja.
Sledi $T_2 - T_1 = U_1 - U_2$, torej je $T_1 + U_1 = T_2 + U_2 = E_0$ konstantna
vrednost.

\begin{izrek}[izrek o energiji]
  Če je rezultanta vseh sil konzervativna, je vsota kinetične in potencialne
  energije konstanta gibanja.
\end{izrek}

\vprasanje{Povej in dokaži izrek o energiji.}
