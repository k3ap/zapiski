\naslov{Variacijski račun}

\begin{definicija}
  Naj bosta $U$ in $V$ Banachova prostora in $P: U \to V$ operator.
  Naj bo $u \in U$ točka v prostoru.
  \pojem{Gateauxov odvod} $P$ v $u$ in v smeri $v \in U$ je podan s predpisom
  \[
	D_u^G P (v) = \left. \frac{d}{dt} \right|_{t=0} P(u + tv)
	= \lim_{t \to 0} \frac{P(u + tv) - P(u)}{t}.
  \]
  Ta odvod se imenuje tudi \pojem{smerni} ali \pojem{šibki} odvod.
\end{definicija}

\begin{definicija}
  Naj bosta $U$ in $V$ Banachova prostora in $P: U \to V$ operator.
  \pojem{Frechetov} ali \pojem{krepki} odvod $P$ v točki $u$ je omejen linearen
  operator $\mathcal{A} : U \to V$, za katerega velja $P(u + tv) = P(u) +
  \mathcal{A}(v) + o(u, v)$ in
  \[
	\lim_{\norm{v} \to 0} \frac{\norm{o(u, v)}}{\norm{v}} = 0.
  \]
  Pišemo $\mathcal{A} = D_u^F P$.
\end{definicija}

\vprasanje{Definiraj šibki in krepki odvod. Kako se še imenujeta?}

\begin{trditev}
  Če je $P$ v $u \in U$ Frechetovo odvedljiv, je tam tudi Gateauxovo odvedljiv.
  Tedaj sta odvoda enaka.
\end{trditev}

\begin{proof}
  Za vsak $v \in U$ velja
  \[
	P(u + tv) = Pu + D_u^F P(tv) + o(u, tv)
	= Pu + t D_u^F P(v) + o(u, tv),
  \]
  iz česar pride
  \[
	\frac{P(u+tv)-Pu}{t} = D_u^F P(v) + \frac{o(u, tv)}{t}.
  \]
  Brez škode za splošnost predpostavimo $\norm{v} = 1$.
  Ker Frechetov odvod obstaja, velja
  \[
	\lim_{t \to 0} \frac{P(u+tv) - Pu}{t}
	= D_u^F P(v) + \lim_{\norm{tv} \to 0} \frac{o(u,tv)}{\norm{tv}}
	= D_u^F P(v).
  \]
\end{proof}

\begin{trditev}
  Če obstaja $D_u^G P$ in je limita
  \[
	\lim_{t \to 0} \frac{P(u+tv)-Pu}{t} = D_u^G P(v)
  \]
  enakomerna glede na $v$ na enotski sferi $S \subseteq U$, potem obstaja tudi
  $D_u^F P$.
\end{trditev}

\begin{proof}
  Enakomernost pomeni, da za vsak $\varepsilon > 0$ obstaja $\delta > 0$, da
  $\abs{t} < \delta$ implicira
  \[
	\norm{\frac{P(u+tv) - Pu}{t} - D_u^G P(v)} < \varepsilon
  \]
  ne glede na $v \in S \subseteq U$, oziroma
  \[
	\norm{P(u+tv) - Pu - D_u^G P(tv)} < \varepsilon t.
  \]
  Označimo $h = tv$ in dobimo
  \[
	\norm{P(u+h) - Pu - D_u^GP(h)} < \varepsilon \norm{h}.
  \]
  Po definiciji limite potem velja
  \[
	\lim_{\norm{h} \to 0} \frac{\norm{P(u+h) - Pu - D_u^G P(h)}}{\norm{h}} = 0.
  \]
\end{proof}

\vprasanje{Povej zadostni pogoj za obstoj Frechetovega odvoda.}

\begin{trditev}
  Naj bo $u \in U$ lokalni minimum funkcionala $\mathcal{L} : U \to \R$, in naj
  bo $\mathcal{L}$ krepko odvedljiv.
  Potem velja $D_u \mathcal{L}(v) = 0$ za vsak $v$.
\end{trditev}

\begin{proof}
  Ker je v $u$ dosežen minimum, velja za vsak $v$ in vsak dovolj majhen $t$
  $\mathcal{L}(u + tv) > \mathcal{L}(u)$.
  Ker je $\mathcal{L}(u+tv) = \mathcal{L}(u) + D_u \mathcal{L}(tv) + o(tv)$,
  velja $D_u \mathcal{L}(tv) + o(tv) > 0$.
  Če je $t > 0$, je $t D_u \mathcal{L}(v) + o(tv) > 0$ in
  \[
	D_u \mathcal{L}(v) + \frac{o(tv)}{t} > 0.
  \]
  Ker drug člen limitira k $0$, je za dovolj majhen $t$ predznak izraza odvisen
  le od predznaka $D_u \mathcal{L}(v)$, ki mora torej biti pozitiven.
  Če enako naredimo za $t < 0$, dobimo, da mora biti predznak $D_u
  \mathcal{L}(v)$ negativen; torej je enak nič.
\end{proof}

\vprasanje{Dokaži, da je odvod funkcionala v lokalnem minimumu enak $0$.}

Če je
\[
  \mathcal{L} = \int_a^b L(x, u, u') dx,
\]
je odvod funkcionala enak
\[
  D_u \mathcal{L}(v)
  = \int_a^b \left( \frac{\partial L}{\partial u}(x) v(x) +
	\frac{\partial L}{\partial u'}(x) v'(x) \right) dx.
\]

\begin{definicija}
  Naj bo $U$ Banachov prostor funkcij $y: [a,b] \to \R$ z metriko, porojeno iz
  maksimum norme.
  Naj bosta $A, B \in \R$ konstanti.
  Prostor
  \[
	V = \{y \in U \such y(a) = A, y(b) = B\}.
  \]
  imenujemo \pojem{prostor dopustnih funkcij}.
\end{definicija}

Nekoliko splošneje: če so $l_i : U \to \R$ omejeni linearni funkcionali, je
prostor dopustnih funkcij podan z
\[
  V = \{y \in U \such \forall i \velja l_i(y) = A_i\}.
\]

\begin{definicija}
  \pojem{Dopustna variacija} je vsaka funkcija $v \in U$, za katero velja $u +
  tv \in V$ za vsak $t$.
\end{definicija}

Za dopustne variacije velja $l_i(u+tv) = l_i(u) + t l_i(v)$.
Če je $u + tv \in V$, je $l_i(u+tv) = A_i$, in torej $l_i(v) = 0$ za vsak $i$.
Prostor dopustnih variacij je torej podan s predpisom
\[
  \text{Var} = \{ v \in U \such l_i(v) = 0 \forall i \} = \bigcap_i \jedro l_i.
\]
Prostor $\text{Var}$ je torej linearen podprostor v $U$, prostor $V$ pa je afin
podprostor v $U$, modeliran s prostorom $\text{Var}$.
Če se vrnemo k osnovnemu variacijskemu problemu, velja $l_1(u) = u(a)$ in
$l_2(u) = u(b)$.

\begin{definicija}
  \pojem{Testne funkcije} na intervalu $[a, b]$ so funkcije $\varphi:[a,b] \to
  \R$, za katere velja
  \begin{itemize}
  \item $\varphi \in \zvodvi{\infty}$,
  \item $\cl{\operatorname{supp} \varphi} \subsetneq [a,b]$.
  \end{itemize}
\end{definicija}

\vprasanje{Kaj so dopustne funkcije, dopustne variacije in testne funkcije? Kaj
  mora veljati za dopustne variacije?}

\begin{izrek}[Osnovni izrek variacijskega računa]
  Naj bo $f: [a,b] \to \R$ zvezna funkcija.
  Če za vsako testno funkcijo $\varphi$ velja
  \[
	\int_a^b f(x) \varphi(x) = 0,
  \]
  potem je $f(x) = 0$ na $[a,b]$.
\end{izrek}

\begin{proof}
  Denimo, da ni tako, da obstaja $x_0 \in (a, b)$, za katerega je $f(x_0) = c>
  0$.
  Zaradi zveznosti $f$ obstaja $\delta > 0$, da velja $f(x) > c/2$ za $x \in
  (x_0 - \delta, x_0 + \delta)$.
  Naj bo $\varphi$ testna funkcija, za katero je $\operatorname{supp} \varphi
  \subset (x_0 - \delta, x_0 + \delta)$ in $\varphi(x) > 0$ na
  $\operatorname{supp} \varphi$.
  Potem imamo
  \[
	\int_a^b f(x) \varphi(x) dx
	= \int_{x_0 - \delta}^{x_0 + \delta} f(x) \varphi(x) dx
	> \frac{c}{2} \int_{x_0 - \delta}^{x_0 + \delta} \varphi(x) dx > 0,
  \]
  kar je protislovno.
\end{proof}

\vprasanje{Povej in dokaži osnovni izrek variacijskega računa.}

Predpostavimo, da je funkcija $\partial_{u'} L(x, u, u')$ odvedljiva po $x$, in
z integracijo po delih izračunamo
\[
  \int_a^b \partial_{u'} L v' dx
  = - \int_a^b \frac{d}{dx} \left( \partial_{u'} L \right) v dx
  + \left. \partial_{u'} L v \right|_a^b.
\]
Odvod je torej enak
\[
  D_u \mathcal{L}(v) = \int_a^b \left( \partial_u L - \frac{d}{dx} \partial_{u'}
	L\right) v dx + \left. \partial_{u'} L v \right|_a^b
  = 0.
\]
Ker so testne funkcije tudi dopustne variacije, dobimo, da za vse testne
funkcije $v$ velja
\[
  D_u \mathcal{L}(v) = \int_a^b \left( \partial_u L - \frac{d}{dx} \partial_{u'}
	L\right) v dx
  = 0,
\]
oziroma, po osnovnem izreku variacijskega računa,
\[
  \partial_u L(x, \hat{u}(x), \hat{u}'(x)) - \frac{d}{dx} \partial_{u'} L(x,
  \hat{u}(x), \hat{u}'(x)) = 0.
\]
Temu pravimo Euler-Lagrangeova enačba.

\vprasanje{Izpelji Euler-Lagrangeovo enačbo.}

Če imamo samo en robni pogoj $u(a) = A$, bo za testne funkcije še vedno veljala
Euler-Lagrangeova enačba, za ostale dopustne variacije pa dobimo novo enačbo
\[
  \partial_{u'} L (b) = 0.
\]
Temu pravimo \pojem{dinamični pogoj}.
