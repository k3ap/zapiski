\podnaslov{Normalni sistem}

Dana je matrika $A \in \R^{m \times n}$ za $m > n$ in vektor $b \in \R^m$.

Velja $Ax - b \bot Az$ za vse $z \in \R^n$ natanko tedaj, ko je za vsak $z$
\[

\]
Sledi $A^T (Ax - b) = 0$ oziroma $A^T A x = A^T b$, ÄŤemur pravimo
\pojem{normalni sistem}.
Pri tem smo tiho predpostavili, da je $\rang A = n$, sicer sistem nebi bil
enoliÄŤno reĹˇljiv.

Velja $w^T A^T A w = \norm{Aw}_2^2 > 0$ za $w \ne 0$, torej je \pojem{Gramova
  matrika} $A^T A$ simetriÄŤna pozitivno definitna, in zanjo obstaja razcep
Choleskega.
Pri reĹˇevanju sistema seveda uporabimo ta razcep.

\podnaslov{QR razcep}

\begin{izrek}
  Naj bo $A \in \R^{m \times n}$ za $m \ge n$ polnega ranga.
  Potem obstaja enoliÄŤen razcep $A = QR$, kjer je $Q \in \R^{m \times n}$ z
  ortonormiranimi stolpci ($Q^T Q = I_n$) in $R \in \R^{n \times n}$ zgornje
  trikotna s pozitivnimi diagonalnimi elementi.
\end{izrek}

\begin{proof}
  ÄŚe bi veljalo $A = QR$, je $A^T A = R^T Q^T Q R = R^T R$.
  Matrika $A^T A$ je simetriÄŤna pozitivno definitna, torej je $R^T R$ njen
  razcep Choleskega, in velja $R = V^T$.
  Iz $A = QR$ sledi $Q = A R^{-1}$.
\end{proof}

\vprasanje{Povej in dokaĹľi izrek o obstoju in enoliÄŤnosti QR razcepa.}

ÄŚe poznamo $A = QR$, je $\im A = \im Q$.
V drugem primeru bo vsakrĹˇno delo stabilnejĹˇe, ker so stolpci ortonormirani.
Normalni sistem se tedaj prevede na $Rx = Q^T b$, ki ga lahko reĹˇimo s premo
substitucijo.

\podnaslov{Gram-Schmidtova ortogonalizacija}

Poznamo tri naÄŤine za izraÄŤun QR razcepa.
NajenostavnejĹˇi pristop je Gram-Schmidtova ortogonalizacija.

\]

\[

\]
Celoten postopek je prikazan v algoritmu~\ref{alg:qr-gram-schmitt}, ki ima
raÄŤunsko zahtevnost $2 mn^2$.

\begin{algorithm}
  \caption{QR razcep s klasiÄŤnim Gram-Schmidtovim postopkom}
  \label{alg:qr-gram-schmitt}
  \begin{algorithmic}
	\For{$k = 1, \ldots, n$}
	\State $q_k = a_k$
	\For{$i=1, \ldots, k-1$}
	\State $r_{ik} = q_i^T a_k$
	\State $q_k = q_k - r_{ik} q_i$
	\EndFor
	\State $r_{kk} = \norm{q_k}_2$
	\State $q_k = \inv{r_{kk}} q_k$
	\EndFor
  \end{algorithmic}
\end{algorithm}

\vprasanje{Izpelji in zapiĹˇi klasiÄŤni Gram-Schmidtov postopek za izraÄŤun QR
  razcepa.}

V algoritmu naredimo Ĺˇe popravek, ki bo poveÄŤal stabilnost; pri raÄŤunanju
$r_{ik}$ namesto formule $r_{ik} = q_i^T a_k$ uporabimo $r_{ik} = q_i^T q_k$.
Novemu postopku pravimo \pojem{modificiran Gram-Schmidt}, in je v teoriji
ekvivalenten klasiÄŤnemu.
Modificiran postopek moramo tudi bolj pametno uporabiti; izraÄŤunamo QR razcep

  \end{bmatrix}

  \end{bmatrix}

NajboljĹˇo reĹˇitev dobimo, ko je $Rx = z$.

\podnaslov{Givensove rotacije}

ÄŚe je $c = \cos \varphi$ in $s = \sin \varphi$, matrika
\[

  =
  \begin{bmatrix}
	1 &  &  &  &  &  &  &  &  &  &  \\
	  & \ddots &  &  &  &  &  &  &  &  &  \\
	  &  & 1 &  &  &  &  &  &  &  &  \\
	  &  &  & c &  &  &  & s &  &  &  \\
	  &  &  &  & 1 &  &  &  &  &  &  \\
	  &  &  &  &  & \ddots &  &  &  &  &  \\
	  &  &  &  &  &  & 1 &  &  &  &  \\
	  &  &  & -s &  &  &  & c &  &  &  \\
	  &  &  &  &  &  &  &  & 1 &  &  \\
	  &  &  &  &  &  &  &  &  & \ddots &  \\
	  &  &  &  &  &  &  &  &  &  & 1
  \end{bmatrix},
\]
ki ima elemente na diagonali in v stolpcih in vrsticah $i, k$, predstavlja

ÄŚe te rotacije ustrezno kombiniramo, dobimo QR razcep, kakor je prikazano v

$Q$, za raÄŤunanje $Q$ pa porabimo Ĺˇe $6 m^2 n - 3 mn^2$ operacij.

\begin{algorithm}

  \label{alg:qr-givens}
  \begin{algorithmic}
	\State $Q = I_m$
	\For{$i=1, \ldots, n$}
	\For{$k=i+1, \ldots, m$}
	\State $r = \sqrt{a_{ii}^2 + a_{ki}^2}$
	\State $c = \nicefrac{a_{ii}}{r}$
	\State $s = \nicefrac{a_{ik}}{r}$
	\State IzraÄŤunaj
	\begin{gather*}

	  \begin{bmatrix}
		c & s \\ -s & c
	  \end{bmatrix}
	  \cdot
	  A([i, k], i:n)
	  \\
	  b([i, k]) =
	  \begin{bmatrix}
		c & s \\ -s & c
	  \end{bmatrix}
	  \cdot b([i, k])
	  \\
	  Q([i, k], :) =
	  \begin{bmatrix}
		c & s \\ -s & c
	  \end{bmatrix}
	  \cdot Q([i, k], :)
	\end{gather*}
	\EndFor
	\EndFor
	\State $Q = Q^T$
  \end{algorithmic}
\end{algorithm}

\podnaslov{Householderjeva zrcaljenja}

Vzemimo $w \in \R^m$, ki je razliÄŤen od $0$, in definirajmo
\[

\]

vrednostjo $-1$.

$w$.

Z izbiro $w = x \mp \norm{x}_2 e_1$ se $x$ preslika v
\[
  P
  \cdot
  \begin{bmatrix}
	x_1 \\ x_2 \\ \vdots \\ x_n

\]

\begin{algorithm}

  \label{alg:qr-householder}

	\State DoloÄŤi $w_i \in \R^{m-i+1}$ iz $A(i:m, i)$

	\State $Q(i:m, :) = P_i Q(i:m, i)$
	\EndFor
	\State $Q = Q^T$
  \end{algorithmic}
\end{algorithm}
