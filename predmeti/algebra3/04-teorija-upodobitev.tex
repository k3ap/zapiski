\naslov{Teorija upodobitev}

\begin{definicija}
  \pojem{$R$-algebra} je levi $R$-modul $A$, ki je hkrati kolobar z enoto, in
  velja $r(a_1 a_2) = (r a_1) a_2 = a_1 (r a_2)$.
\end{definicija}

\begin{primer}
  Če je $R$ komutativen kolobar z enico in $G$ grupa, je prosti $R$-modul nad
  $G$, ki ga označimo z $RG$, $R$-algebra, ki ji pravimo \pojem{grupna algebra}.
\end{primer}

\begin{definicija}
  Naj bo $A$ $R$-algebra in $V$ $R$-modul.
  \pojem{Upodobitev algebre $A$ nad $V$} je homomorfizem algeber $\rho: A
  \to \End_R(V)$.
\end{definicija}

\begin{opomba}
  Če je $V$ prost $R$-modul, in $V = R^n$, je $\End_R(V) \cong M_n(R)$.
  Številu $n$ pravimo \pojem{stopnja upodobitve}.
\end{opomba}

\begin{definicija}
  Naj bo $R$ komutativen kolobar z enico, $G$ grupa in $V$ $R$-modul.
  \pojem{Upodobitev grupe $G$ nad $V$} je homomorfizem algeber $\rho: RG
  \to \End_R(V)$.
\end{definicija}

\begin{primer}
  Če je $G$ poljubna grupa, da je $\rho: G \to \GL_R(RG)$, definirana na bazi
  kot
  \[
	\rho(g): h \mapsto gh
  \]
  upodobitev.
  Pravimo ji \pojem{leva regularna upodobitev grupe $G$}.
  Je stopnje $\abs{G}$.
\end{primer}

\begin{primer}
  Če je $G$ grupa in $R$ komutativen kolobar z enico, lahko definiramo množico
  $\operatorname{fun}(G, R)$ vseh funkcij $G \to R$.
  Tu seštevanje in množenje s skalarjem definiramo po točkah, da dobimo kolobar.
  Potem je $\rho: G \to \GL_R(\operatorname{fun}(G,R))$, definirana z
  \[
	\rho(g): \varphi \mapsto \left( h \mapsto \varphi(hg) \right),
  \]
  \pojem{desna regularna upodobitev grupe $G$}.
\end{primer}

\vprasanje{Kaj sta leva in desna regularna upodobitev grupe?}

\begin{definicija}
  Naj bosta $\rho_1: G \to \GL_R(V_1)$ in $\rho_2: G \to \GL_R(V_2)$ upodobitvi
  grupe $G$.
  $R$-linearni preslikavi $T: V_1 \to V_2$, za katero velja $T \circ \rho_1(g) =
  \rho_2(g) \circ T$ za vsak $g \in G$, pravimo \pojem{spletična} med $\rho_1$
  in $\rho_2$.
  Upodobitvi $\rho_1$ in $\rho_2$ sta \pojem{ekvivalentni}, če obstaja
  bijektivna spletična med njima.
\end{definicija}

\begin{opomba}
  Upodobitvi sta ekvivalentni natanko tedaj, ko sta $V_1$ in $V_2$ izomorfna kot
  $G$-modula.
\end{opomba}

\vprasanje{Kaj je spletična? Kdaj sta upodobitvi ekvivalentni?}

\begin{definicija}
  Podprostor $W \subseteq V$ je \pojem{$G$-invarianten}, če velja $\rho(g) w \in
  W$ za vsak $g \in G$ in $w \in W$.
\end{definicija}

V tem primeru lahko zožimo upodobitev na $G \to \GL_R(W)$, temu pravimo
\pojem{podupodobitev}.

\begin{vo}{Pokaži, da je leva regularna upodobitev grupe podupodobitev desne.}
  Pokazati moramo, da je $RG$ $G$-podmodul modula $\operatorname{fun}(G,R)$.
  Za to poiščemo injektivno spletično $T: RG \to \operatorname{fun}(G,R)$.
  Pravi predpis je
  \[
	T(g) : h \mapsto
	\begin{cases}
	  1 & g = h^{-1} \\
	  0 & g \ne h^{-1}
	\end{cases}
  \]
\end{vo}

Če sta $\rho_1: G \to \GL_R(V_1)$ in $\rho_2: G \to \GL_R(V_2)$ upodobitvi, je
$\rho_1 \oplus \rho_2: G \to \GL_R(V_1 \oplus V_2)$, definirana z $(v_1, v_2)
\mapsto (\rho_1(g) v_1, \rho_2(g) v_2)$, njuna \pojem{direktna vsota}.
Podobno lahko definiramo \pojem{tenzorski produkt} $\rho_1 \otimes \rho_2: G \to
\GL_R(V_1 \otimes_R V_2)$ kot
\[
  \rho_1 \otimes \rho_2(g) : (v_1 \otimes v_2) \mapsto \rho_1(g) v_1 \otimes
  \rho_2(g) v_2.
\]
Poznamo tudi Hom-konstrukcijo:
$\Hom_R(V_1, V_2)$ je $R$-modul, in postane $G$-modul z
\[
  g \cdot \varphi(v_1) = \rho_2(g) \varphi(\rho_1(g)^{-1} v_1).
\]
S tem dobimo upodobitev $\Hom(\rho_1, \rho_2): G \to \GL_R(\Hom_R(V_1, V_2))$.

\vprasanje{Opiši direktno vsoto, tenzorski produkt in Hom-konstrukcijo
  upodobitev.}

\podnaslov{Polenostavni moduli}

Naj bo $A$ kolobar z enico (ne nujno komutativen).

\begin{definicija}
  Naj bo $V$ $A$-modul.
  Pravimo, da je $V$ \pojem{enostaven}, če ima natanko dva podmodula.
  Pravimo, da je $V$ \pojem{polenostaven}, če je končna direktna vsota
  enostavnih modulov.
\end{definicija}

\begin{definicija}
  \pojem{Kompozicijska vrsta} $A$-modula $V$ je končno zaporedje $A$-modulov
  \[
	\{0\} = V_0 \subseteq V_1 \subseteq V_2 \subseteq \cdots \subseteq V_n = V,
  \]
  pri čemer so zaporedni kvocienti $V_{i+1} / V_i$ enostavni $A$-moduli.
\end{definicija}

\begin{lema}
  Naj bo $U$ $A$-modul in $U = S_1 + \cdots + S_n$, kjer so $S_i$ enostavni.
  Naj bo $V$ podmodul $U$.
  Potem obstaja $\{i_1, \ldots, i_k\} \subseteq \{1, \ldots, n\}$, da je $U = V
  \oplus S_{i_1} \oplus \cdots \oplus S_{i_k}$.
\end{lema}

\begin{proof}
  Izberimo največjo podmnožico indeksov, da bo $W = V \oplus S_{i_1} \oplus
  \cdots \oplus S_{i_k}$ direktna vsota.
  Recimo, da $W \ne U$, torej $S_j \nsubseteq W$ za nek $j$.
  Presek $S_j \presek W$ je podmodul v $S_j$, torej je $S_j \presek W = \{0\}$,
  in izbrana množica ni bila maksimalna.
\end{proof}

\vprasanje{Dokaži: če je $U$ direktna vsota enostavnih modulov, lahko za vsak
  podmodul $V$ zapišemo $U$ kot direktno vsoto $V$ in nekih enostavnih modulov.}

\begin{trditev}
  Naj bo $U$ $A$-modul.
  Naslednje trditve so ekvivalentne:
  \begin{itemize}
  \item $U$ je končna direktna vsota enostavnih $A$-modulov,
  \item $U$ je direktna vsota končno mnogo enostavnih podmodulov,
  \item $U$ ima končno kompozicijsko vrsto in vsak podmodul v $U$ ima direkten
	komplement.
  \end{itemize}
\end{trditev}

\begin{proof}
  Implikacija iz 1 v 2 je očitna.
  Implikacija iz 2 v 3 takoj sledi iz leme.
  Iz 3 v 1:
  Najprej dokažimo, da se tretja lastnost deduje na podmodule.
  Če je $W \le U$, ima $W$ končno kompozicijsko vrsto, ki jo dobimo s sekanjem
  vrste za $U$ z $W$.
  Če je $Z \le W$, je $Z \le U$ in je $U = Z \oplus X$, torej $W = Z \oplus (W
  \presek X)$.

  Sedaj uporabimo indukcijo po dolžini kompozicijske vrste $l$.
  Za $l = 0$ je $U$ enostaven in velja.
  Prehod na $l+1$: $U_l$ je podmodul, torej obstaja komplement $U = U_l \oplus
  X$.
  Ta podmodul ima kompozicijsko vrsto dolžine $l$, prav tako ima $X \cong U/U_l$
  tudi kompozicijsko vrsto dolžine kvečjemu $l$.
  Lastnost je dedna na $U_l$ in $X$, ki sta po indukcijski predpostavki vsoti
  enostavnih modulov.
\end{proof}

\vprasanje{Recimo, da ima modul $U$ končno kompozicijsko vrsto, in da ima vsak
  podmodul v $U$ direkten komplement. Pokaži, da je $U$ polenostaven.}

\begin{izrek}[Maschke]
  Naj bo $G$ končna grupa in $F$ polje.
  Predpostavimo, da karakteristika $F$ ne deli reda grupe.
  Naj bo $\rho: G \to \GL_F(V)$ upodobitev grupe $G$, kjer $V$
  končnodimenzionalen.
  Naj bo $W$ $G$-podmodul $V$.
  Potem obstaja $G$-podmodul $V$, da je $V = W \oplus X$.
\end{izrek}

\begin{proof}
  Naj bo $\pi: V \to W$ projektor.
  Velja $V = W \oplus \jedro \pi$, to jedro pa ni nujno $G$-invariantno.
  Definiramo $\pi': V \to V$ kot
  \[
	\pi' = \inv{\abs{G}} \sum_{g \in G} \rho(g) \pi \rho(g)^{-1}.
  \]
  Velja $\im \pi' \subseteq W$ ter $\pi'(w) = w$ za $w \in W$, torej je $V = W
  \oplus \jedro \pi'$.
  Sedaj $\pi'$ je homomorfizem $G$-modulov, in je jedro $G$-invariantno.
\end{proof}

\begin{posledica}
  Naj bo $G$ končna grupa, $F$ polje, ter $\operatorname{char} F \nmid \abs{G}$.
  Potem je vsak končnorazsežen $G$-modul polenostaven.
\end{posledica}

\vprasanje{Povej in dokaži Maschkejev izrek.}

\begin{definicija}
  Kolobar $A$ je \pojem{polenostaven}, če je vsak $A$-modul polenostaven.
\end{definicija}

\begin{primer}
  Če je $F$ polje, $G$ končna grupa in karakteristika $F$ ne deli $\abs{G}$, je
  po Maschkejevem izreku $FG$ polenostaven kolobar.
\end{primer}

\begin{lema}[Schur]
  Naj bo $A$ $F$-algebra ter $S_1, S_2$ enostavna $A$-modula.
  \begin{itemize}
  \item Če $S_1 \ncong S_2$, je $\Hom(S_1, S_2) = \{0\}$,
  \item če $S_1 \cong S_2$, je $\End_A(S_1)$ obseg,
  \item če je $F$ algebraično zaprto polje, je $\End_A(S_1) \cong F$.
  \end{itemize}
\end{lema}

\begin{proof}
  Prva točka: naj bo $\varphi: S_1 \to S_2$ homomorfizem.
  Jedro $\jedro \varphi$ je podmodul v $S_1$, torej je bodisi $\{0\}$ bodisi
  $S_1$.
  Podobno je $\im \varphi$ podmodul v $S_2$, torej je bodisi $\{0\}$ bodisi
  $S_2$.
  Če je $\varphi \ne 0$, je izomorfizem.
  Podobno za drugo točko.

  Tretja točka: naj bo $\varphi: S_1 \to S_1$ endomorfizem.
  To je linearna preslikava, torej ima lastno vrednost $\lambda \in F$.
  Potem $\varphi - \lambda \id$ ni injektiven, in je po drugi točki enak $0$, iz
  česar sledi $\varphi = \lambda \id$.
\end{proof}

\vprasanje{Povej in dokaži Schurovo lemo.}

\begin{lema}
  Naj bo $A$ kolobar.
  Če gledamo $A$ kot $A$-modul, potem je $\End_A(A) \cong A^{\text{op}}$.
\end{lema}

\begin{lema}
  Naj bo $U$ $A$-modul, $U = U_1 \oplus \cdots \oplus U_r$, kjer so $U_i$
  podmoduli v $U$.
  Potem je $\End_A(U)$ izomorfen kolobarju matrik $[\varphi_{ij}]_{ij}$ za
  $\varphi_{ij} \in \Hom_A(U_j, U_i)$.
\end{lema}

\begin{proof}
  Skica.
  Preslikava $\varphi: U \to U$ je določena s slikami $u_j \in U_j$.
  Razpišemo lahko
  \[
	\varphi(u_j) = \underbrace{\varphi_{1j}(u_j)}_{\in U_1} + \cdots +
	\underbrace{\varphi_{rj}(u_j)}_{\in U_r},
  \]
  pri čemer se izkaže $\varphi_{ij} \in \Hom_A(U_j, U_i)$.
  Ta predpis določa matriko, podobno dobimo v drugo smer.
  Preveriti je treba še, da komponiranje ustreza produktu matrik.
\end{proof}

\begin{izrek}[Artin-Wedderburn]
  Naj bo $A$ končnorazsežna algebra nad poljem $F$, za katero velja, da je vsak
  končnogeneriran $A$-modul polenostaven.
  Recimo, da je $A$ kot $A$-modul oblike
  \[
	A = S_1^{n_1} \oplus \cdots \oplus S_r^{n_r},
  \]
  kjer so $S_i$ paroma neizomorfni enostavni $A$-moduli in potenciranje
  $S_i^{n_i}$ označuje $n_i$-kratno direktno vsoto.
  Potem je $A$ kot $F$-algebra izomorfna $A \cong M_{n_1}(D_1) \oplus \cdots
  \oplus M_{n_r}(D_r)$, kjer so $D_i = \End_A(S_i)^{\text{op}}$ obsegi.
  V posebnem primeru, če je $F$ algebraično zaprto, je $D_i = F$.
\end{izrek}

\begin{proof}
  Velja $\End_A(A) = \End(S_1^{n_1} \oplus \cdots \oplus S_r^{n_r})$, elementi
  te množice so matrike $[\varphi_{ij}]_{ij}$, kjer $\varphi_{ij}: S_j^{n_j} \to
  S_i^{n_i}$.
  Če je $i \ne j$, je $\varphi_{ij} = 0$ po Schurovi lemi, torej so matrike
  diagonalne in je $\End_A(A) = \End_A(S_1^{n_1}) \oplus \cdots
  \oplus \End_A(S_r^{n_r})$.
  Elementi $\End_A(S_i^{n_i})$ so $n_i \times n_i$ matrike nad $\End_A(S_i)$,
  torej je
  \[
	\End_A(A) = M_{n_1}(\End_A(S_1)) \oplus \cdots \oplus M_{n_r}(\End_A(S_r)),
  \]
  hkrati pa velja $A = \End_A(A)^{\text{op}}$ kot algebra.
\end{proof}

\vprasanje{Povej in dokaži Artin-Wedderburnov izrek.}

\podnaslov{Uporaba pri upodobitvah grup}

\begin{definicija}
  Upodobitev $\rho: G \to \GL_F(V)$ je \pojem{nerazcepna}, če je $V$ enostaven
  $FG$-modul.
  Upodobitev je \pojem{povsem razcepna}, če je $V$ polenostaven $FG$-modul.
\end{definicija}

\begin{izrek}
  Naj bo $G$ končna grupa, $F$ polje in $\operatorname{char} F \nmid \abs{G}$.
  Vsaka upodobitev $G$ je povsem razcepna.
  Do ekvivalence natančno ima $G$ končno mnogo nerazcepnih upodobitev $\rho_1,
  \ldots, \rho_r$.
  Če je $F$ algebraično zaprto, je $n_1^2 + \cdots + n_r^2 = \abs{G}$ za $n_i =
  \deg \rho_i$.
\end{izrek}

Izrek sledi iz rezultatov prejšnjega razdelka.
