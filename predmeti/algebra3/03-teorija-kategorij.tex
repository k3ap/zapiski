\naslov{Teorija kategorij}

\begin{definicija}
  \pojem{Kategorija} je struktura $\kat$, sestavljena iz
  \begin{itemize}
  \item razreda objektov $\operatorname{Ob} \kat$,
  \item množice morfizmov $\kat(A,B)$ med vsakima objektoma $A, B$,
  \item preslikav $\circ : \kat(A,B) \times \kat(B,C) \to \kat(A,C)$, kjer so
	$A, B, C$ poljubni objekti kategorije $\kat$.
  \end{itemize}
  Zahtevamo, da so preslikave $\circ$ asociativne, in da za vsak objekt $A$
  obstaja morfizem $1_A \in \kat(A,A)$, da velja $f \circ 1_A = f$ za vsak $f
  \in \kat(A,B)$ ter $1_A \circ g = g$ za vsak $g \in \kat(B,A)$.
  Temu pogoju pravimo \pojem{unitalnost}.
\end{definicija}

\vprasanje{Definiraj kategorijo.}

\begin{definicija}
  Naj bo $\kat$ kategorija in $f \in \kat(A,B)$.
  Potem je $f$ \pojem{izomorfizem}, če obstaja morfizem $g \in \kat(B,A)$, da je
  $f \circ g = 1_B$ in $g \circ f = 1_A$.
  Pravimo, da sta $A$ in $B$ \pojem{izomorfna}, če med njima obstaja izomorfizem.
\end{definicija}

\begin{definicija}
  Naj bo $f \in \kat(A,B)$ morfizem.
  Potem je
  \begin{itemize}
  \item \pojem{prerez}, če obstaja $g \in \kat(B,A)$, da je $g \circ f = 1_A$,
  \item \pojem{retrakt}, če obstaja $g \in \kat(B,A)$, da je $f \circ g = 1_B$,
  \item \pojem{monomorfizem}, če za vsaka morfizma $g, h \in \kat(C,A)$ iz
	enakosti $f \circ g = f \circ h$ sledi $g = h$.
  \item \pojem{epimorfizem}, če za vsaka morfizma $g, h \in \kat(B,C)$ iz
	enakosti $g \circ f = h \circ f$ sledi $g = h$.
  \end{itemize}
\end{definicija}

\vprasanje{Definiraj prerez, retrakt, monomorfizem in epimorfizem.}

\begin{vo}{Povej primer epimorfizma, ki ni surjektiven.}
  Vložitev $\iota: \Z \to \Q$ je homomorfizem kolobarjev.
  Če je $g \circ \iota = h \circ \iota$, velja $g(n) = h(n)$ za vsako celo
  število $n$.
  Potem velja tudi
  \[
	g(\frac{a}{b}) = g(a) g(\inv{b})
	= h(a) g(\inv{b})
	= h(b \frac{a}{b}) g(\inv{b})
	= h(\frac{a}{b}).
  \]
\end{vo}

\begin{definicija}
  Naj bo $\kat$ kategorija.
  Objekt $Z$ je \pojem{začetni objekt}, če za vsak objekt $A$ velja
  $\abs{\kat(Z,A)} = 1$.
  Objekt $K$ je \pojem{končni objekt}, če za vsak objekt $A$ velja
  $\abs{\kat(A,K)} = 1$.
\end{definicija}

\begin{trditev}
  Poljubna začetna objekta dane kategorije sta izomorfna.
\end{trditev}

\begin{proof}
  Naj bosta $Z_1$ in $Z_2$ začetna objekta.
  Potem obstaja $f \in \kat(Z_1, Z_2)$ in $g \in \kat(Z_2, Z_1)$.
  Velja $f \circ g \in \kat(Z_2, Z_2)$; ker pa je $\abs{\kat(Z_2, Z_2)} = 1$, je
  $f \circ g = 1$.
  Podobno v drugo smer.
\end{proof}

\begin{opomba}
  Enako velja za končne objekte.
\end{opomba}

\vprasanje{Dokaži: poljubna začetna objekta v dani kategoriji sta izomorfna.}

\begin{definicija}
  Naj bo $\kat$ kategorija.
  \pojem{Nasprotna kategorija} $\kat^{\text{op}}$ je definirana na naslednji
  način:
  \begin{itemize}
  \item objekti so enaki objektom $\kat$
  \item če sta $A, B$ objekta, je $\kat^{\text{op}}(A,B) = \kat(B,A)$
  \item za $f \in \kat^{\text{op}}(A,B)$ in $g \in \kat^{\text{op}}(C,A)$ je
	kompozitum $f * g = g \circ f$.
  \end{itemize}
\end{definicija}

\vprasanje{Definiraj nasprotno kategorijo.}

\begin{definicija}
  Naj bosta $\kat$ in $\katD$ kategoriji.
  \pojem{Funktor} med kategorijama je predpis $F$, za katerega velja
  \begin{itemize}
  \item za vsak objekt $A$ v $\kat$ imamo natanko določen objekt $F(A)$ v
	$\katD$,
  \item za vsak $f \in \kat(A,B)$ imamo natanko določen morfizem $F(f) \in
	\katD(F(A), F(B))$,
  \item $F(1_A) = 1_{F(A)}$ za vsak objekt $A$ v $\kat$,
  \item $F(f \circ g) = F(f) \circ F(g)$
  \end{itemize}
  \pojem{Kofunktor} med $\kat$ in $\katD$ je funktor v nasprotni kategoriji.
\end{definicija}

\vprasanje{Definiraj funktorje in kofunktorje.}

\begin{definicija}
  Naj bosta $F, G$ funktorja iz $\kat$ v $\katD$.
  \pojem{Naravna transformacija} med $F$ in $G$ je nabor morfizmov $\mu_C: F(C)
  \to G(C)$ za objekt $C$ kategorije $\kat$, da za vsak morfizem $f: A \to B$ v
  kategoriji $\kat$ diagram
  \[\begin{tikzcd}
	  {F(A)} && {F(B)} \\
	  \\
	  {G(A)} && {G(B)}
	  \arrow["{F(f)}", from=1-1, to=1-3]
	  \arrow["{\mu_A}"', from=1-1, to=3-1]
	  \arrow["{\mu_B}", from=1-3, to=3-3]
	  \arrow["{G(f)}"', from=3-1, to=3-3]
	\end{tikzcd}\]
  komutira.
  Pravimo, da sta funktorja $F$ in $G$ \pojem{naravno izomorfna}, če obstaja
  naravna transformacija $\mu: F \to G$, za katero so vsi morfizmi $\mu_C$
  izomorfizmi.
\end{definicija}

\podnaslov{Univerzalne konstrukcije}

\begin{definicija}
  Naj bosta $A, B$ objekta kategorije $\kat$.
  \pojem{Produkt} objektov $A$ in $B$ je objekt $P$, skupaj z morfizmoma $p: P
  \to A$ in $q: P \to B$, da za poljuben objekt $X$ in poljubna morfizma $f: X
  \to A$ ter $g: X \to B$ obstaja natanko en morfizem $h: X \to P$, da naslednji
  diagram komutira:
  \[\begin{tikzcd}
	  A & P & B \\
	  & X
	  \arrow["p"', from=1-2, to=1-1]
	  \arrow["q", from=1-2, to=1-3]
	  \arrow["f", from=2-2, to=1-1]
	  \arrow["h", from=2-2, to=1-2]
	  \arrow["g"', from=2-2, to=1-3]
	\end{tikzcd}\]
\end{definicija}

\begin{definicija}
  \pojem{Koprodukt} objektov $A$ in $B$ je produkt v nasprotni kategoriji.
\end{definicija}

\vprasanje{Definiraj produkt in koprodukt.}

\begin{definicija}
  Kategorija je \pojem{konkretna}, če so objekti množice, morfizmi pa
  preslikave.
\end{definicija}

\begin{definicija}
  Naj bo $\kat$ konkretna kategorija in $X$ neprazna množica.
  \pojem{Prosti objekt} nad množico $X$ v kategoriji $\kat$ je objekt $F$,
  skupaj s preslikavo $\iota: X \to F$, da za vsak objekt $C$ in vsako
  preslikavo $f: X \to C$ obstaja natanko en morfizem $\tilde{f}: F \to C$, da
  naslednji diagram komutira:
  \[\begin{tikzcd}
	  X & C \\
	  F
	  \arrow["f", from=1-1, to=1-2]
	  \arrow["\iota", from=2-1, to=1-1]
	  \arrow["{\tilde{f}}"', from=2-1, to=1-2]
	\end{tikzcd}\]
\end{definicija}

\begin{opomba}
  Prost objekt, če obstaja, je do izomorfizma natanko enolično določen.
  Definiramo kategorijo $\hat{\kat}$ z objekti $X \xrightarrow{f} C$ in morfizmi
  \[\begin{tikzcd}
	  X & C \\
	  X & C
	  \arrow["f", from=1-1, to=1-2]
	  \arrow["\id"', from=1-1, to=2-1]
	  \arrow[dotted, from=1-2, to=2-2]
	  \arrow["g"', from=2-1, to=2-2]
	\end{tikzcd}\]
  kjer je diagram komutativen.
  Prosti objekt v $C$ je ravno začetni objekt kategorije $\hat{\kat}$.
\end{opomba}

\vprasanje{Definiraj prosti objekt in pokaži, da je do izomorfizma natanko
  enolično določen.}

\begin{vo}{Kaj so prosti objekti v kategoriji polj?}
  Prostih objektov ni.
  Če imamo $\iota: X \to F$ ter $f: X \to E$, ter imata $F$ in $E$ različni
  karakteristiki, potem med njima ni homomorfizma.
\end{vo}
