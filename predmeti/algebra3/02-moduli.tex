\naslov{Moduli}

\begin{definicija}
  Naj bo $R$ kolobar in $M$ neprazna mnoĹľica.
  \pojem{Levi $R$-modul} je $(M, +, \cdot)$, kjer sta $+ : M \times M \to M$ in
  $\cdot : R \times M \to M$.
  Pri tem zahtevamo
  \begin{itemize}
  \item $(M, +)$ je Abelova grupa
  \item $r(m_1 + m_2) = r m_1 + r m_2$
  \item $(r_1 + r_2) m = r_1 m + r_2 m$
  \item $(r_1 r_2) m = r_1 (r_2 m)$
  \item $1 \cdot m = m$
  \end{itemize}
\end{definicija}

Analogno lahko definiramo desne module.
Vsak levi $R$-modul je pravzaprav tudi desni $R^{\text{opp}}$ modul, kjer je
$R^{\text{opp}}$ kolobar z enakim seĹˇtevanjem kot v $R$ in mnoĹľenjem v obratnem
vrstnem redu.

\vprasanje{Definiraj modul.}

\begin{primer}
  Vektorski prostor nad poljem $F$ je levi (in desni) $F$-modul.
\end{primer}

\begin{primer}
  Abelova grupa (tu pisana aditivno) je $\Z$-modul za mnoĹľenje
  \[
	n \cdot g = \underbrace{g + g + \ldots + g}_{n}.
  \]
\end{primer}

\begin{primer}
  Naj bo $I$ levi ideal kolobarja $R$.
  Potem je $I$ levi $R$-modul.
\end{primer}

\begin{primer}
  ÄŚe je $V$ vektorski prostor nad poljem $F$ in $R$ mnoĹľica endomorfizmov $V \to
  V$, je $V$ levi $R$-modul za mnoĹľenje
  \[
	A \cdot v = A(v).
  \]
\end{primer}

\vprasanje{NaĹˇtej nekaj enostavnih primerov modulov.}

\begin{definicija}
  Naj bo $M$ $R$-modul.
  MnoĹľica $N \subseteq M$ je \pojem{podmodul} v $M$, ÄŤe je $N$ za podedovano
  seĹˇtevanje in mnoĹľenje s skalarjem tudi modul.
  Oznaka $N \le M$.
\end{definicija}

\begin{definicija}
  Naj bo $M$ $R$-modul in $X \subseteq M$ neprazna mnoĹľica.
  NajmanjĹˇemu podmodulu, ki vsebuje $X$, pravimo \pojem{podmodul, generiran z
	mnoĹľico $X$}.
  Oznaka $\sk{X}$.
  Modul je \pojem{konÄŤno generiran}, ÄŤe obstaja konÄŤna $X$, da je $M = \sk{X}$.
  ÄŚe je modul generiran z enim samim elementom, je \pojem{cikliÄŤni}.
\end{definicija}

\begin{definicija}
  Naj bosta $N, M$ $R$-modula.
  Preslikava $\varphi : M \to N$ je \pojem{homomorfizem} $R$-modulov, ÄŤe velja
  $\varphi(m_1 + m_2) = \varphi(m_1) + \varphi(m_2)$ in $\varphi(r m) = r
  \varphi(m)$.
  \pojem{Jedro} homomorfizma je mnoĹľica
  \[
	\jedro \varphi = \{m \in M \such \varphi(m) = 0\},
  \]
  \pojem{slika} pa
  \[
	\im \varphi = \{\varphi(m) \such m \in M\}.
  \]
\end{definicija}

\begin{definicija}
  Naj bo $M$ $R$-modul in $N \le M$.
  Za relacijo
  \[
	m_1 \sim m_2 \nt m_1 - m_2 \in N
  \]
  definiramo \pojem{kvocientni modul} $M/_\sim = M/_N$.
\end{definicija}

\begin{izrek}
  Za kvocientne module veljajo naslednje lastnosti:
  \begin{itemize}
  \item ÄŤe je $\varphi : M \to N$ homomorfizem, potem je $M/_{\jedro \varphi}
	\cong \im \varphi$,
  \item ÄŤe sta $N, K$ podmodula v $M$, potem je $(N+K)/_K \cong N/_{N \presek
	  K}$,
  \item velja
	\[
	  \frac{M/_N}{L/_N} \cong M/_L.
	\]
  \end{itemize}
\end{izrek}

\begin{definicija}
  Naj bodo $N_1, \ldots, N_k$ $R$-moduli.
  \pojem{Zunanji direktni produkt} je $R$-modul $N_1 \times N_2 \times \cdots
  \times N_k$ s seĹˇtevanjem in mnoĹľenjem s skalarjem po komponentah.
  Oznaka $N_1 \oplus N_2 \oplus \cdots \oplus N_k$.
\end{definicija}

\begin{opomba}
  ÄŚe imamo neskonÄŤno druĹľino modulov, lahko naredimo podobno konstrukcijo, ki ji
  pravimo \pojem{karteziÄŤni produkt}.
\end{opomba}

\begin{definicija}
  Naj bodo $N_1, \ldots, N_k \le M$.
  Vsoti $N_1 + \cdots + N_k$ pravimo \pojem{notranja direktna vsota}, ÄŤe za vsak
  $i$ velja $N_i \presek (N_1 + \cdots + N_{i-1} + N_{i+1} + \cdots + N_k) =
  \{0\}$.
\end{definicija}

\begin{opomba}
  Zunanja in notranja definicija sta ekvivalentni.
\end{opomba}

\begin{definicija}
  Naj bo $M$ $R$-modul in $X \subseteq M$.
  Pravimo, da je $X$ \pojem{baza} modula $M$, ÄŤe je $M = \sk{X}$ in ÄŤe za vsak
  $k \in \N$ in vse $x_1, \ldots, x_k \in X$ iz enakosti $r_1 x_1 + \cdots + r_n
  x_n = 0$ sledi $r_i = 0$.
\end{definicija}

\begin{primer}
  Obstajajo moduli, ki nimajo baz.
  Modul $M = \Z_n = \Z/_{n \Z}$ je $\Z$-modul.
  Recimo, da je nek $x + n\Z$ v bazi.
  Ampak $n(x + n \Z) = 0$, $n$ pa v $\Z$ ni enak $0$.
\end{primer}

\begin{definicija}
  Modul je \pojem{prost}, ÄŤe ima bazo.
\end{definicija}

\vprasanje{Kaj je baza modula? Podaj primer modula, ki ni prost.}

\begin{izrek}
  Naj bo $M$ $R$-modul.
  Naslednje trditve so ekvivalentne:
  \begin{itemize}
  \item $M$ je prost $R$-modul.
  \item Obstaja indeksna mnoĹľica $\Lambda$, da je $M$ kot $R$-modul izomorfen
	zunanji direktni vsoti kopij $R$-modula $R$:
	\[
	  M \cong \bigoplus_{\lambda \in \Lambda} R.
	\]
  \item Obstaja indeksna mnoĹľica $\Lambda$ in podmoduli $M_\lambda \le M$,
	$M_\lambda \cong R$, da je
	\[
	  M = \bigoplus_{\lambda \in \Lambda} M_\lambda.
	\]
  \end{itemize}
\end{izrek}

\begin{proof}
  Druga in tretja toÄŤka sta oÄŤitno ekvivalentni.
  Naj bo $X$ baza modula $M$.
  Trdimo $M = \bigoplus_{x \in X} \sk{x}$.
  Ker je $M = \sk{X}$, je $M$ vsota podmodulov $\sk{x}$.
  Vzemimo $z \in \sk{x} \presek \sum_{y \ne x} \sk{y}$.
  Velja $z = rx = r_1 y_1 + \cdots + r_k y_k$, iz definicije baze pa sledi $r =
  r_i = 0$.
  Preveriti moramo Ĺˇe, da je $\sk{x} \cong R$.
  Ustrezen izomorfizem $R \to \sk{x}$ bo $\varphi(r) = rx$.

  Sedaj dokaĹľimo, da iz druge toÄŤke sledi prva.
  Imamo izomorfizem
  \[
	\varphi : \bigoplus_{\lambda \in \Lambda} R \to M.
  \]
  Izberimo $e_\lambda$ kot $\Lambda$-terico, ki ima na mestu $\lambda$ enico,
  drugje pa $0$, in definirajmo $x_\lambda = \varphi(e_{\lambda})$.
  Preverimo lahko, da je $X = \{x_\lambda \such \lambda \in \Lambda\}$ baza.
\end{proof}

\vprasanje{Karakteriziraj prostost modula in dokaĹľi karakterizacijo.}

\begin{posledica}
  Vsak $R$-modul je kvocient nekega prostega $R$-modula.
\end{posledica}

\begin{proof}
  Naj bo $M$ $R$-modul.
  Definiramo
  \[
	\varphi: \oplus_{m \in M} R \to M
  \]
  tako, da za element $e_m$ direktne vsote, ki ima na $m$-tem mestu enico,
  drugje pa niÄŤ, predpiĹˇemo $\varphi(e_m) = m$.
  Ti elementi tvorijo bazo, torej je s tem natanko doloÄŤen epimorfizem
  $R$-modulov.
\end{proof}

\vprasanje{PokaĹľi, da je vsak modul kvocient prostega modula.}

\begin{primer}
  Podmodul prostega modula ni nujno prost.
  $\Z_8$ je prost $\Z_8$-modul, njegov podmodul $2 \Z_8 = \{0, 2, 4, 6\}$ pa ni
  prost.
\end{primer}

\vprasanje{Podaj primer podmodula prostega modula, ki ni sam prost.}

\begin{izrek}[univerzalna lastnost]
  $R$-modul $M$ je prost natanko tedaj, ko obstaja mnoĹľica $X$ in preslikava
  $\iota: X \to M$, da velja: za vsak $R$-modul $N$ in vsako preslikavo $\kappa:
  X \to N$ obstaja natanko en homomorfizem $R$-modulov $f: M \to N$, da velja $f
  \circ \iota = \kappa$.
\end{izrek}

\begin{proof}
  V desno: $X$ naj bo baza $M$, $\iota$ pa vloĹľitev, torej
  \begin{gather*}
	M = \oplus_{x \in X} R, \\
	\iota(x) = e_x.
  \end{gather*}
  Naj bo $N$ poljuben $R$-modul in $\kappa : X \to N$ poljubna preslikava.
  Recimo, da obstaja $f$, kot zahtevamo.
  Potem mora veljati $f \circ \iota = \kappa$, torej $f(e_x) = \kappa(x)$.
  Ker je ${\{e_x\}}_x$ baza, to enoliÄŤno doloÄŤi $f$, ÄŤe le obstaja.
  Definiramo ga kot
  \[
	f\!\left( \sum_{x \in X} a_x e_x \right) = \sum_{x \in X} a_x \kappa(x),
  \]
  kar ustreza zahtevi.

  V levo:
  Dokazujemo, da je
  \[
	M \cong \oplus_{x \in X} R.
  \]
  Definiramo $\kappa$, kar nam po univerzalni lastnosti poda homomorfizem $f$,
  da diagram
  \[\begin{tikzcd}
	  X & {\oplus_x R} \\
	  M
	  \arrow["\kappa", from=1-1, to=1-2]
	  \arrow["\iota"', from=1-1, to=2-1]
	  \arrow["g"{description}, curve={height=6pt}, dotted, from=1-2, to=2-1]
	  \arrow["f"{description}, curve={height=6pt}, dotted, from=2-1, to=1-2]
	\end{tikzcd}\]
  komutira.
  Pokazali smo, da prosti moduli imajo univerzalno lastnost, torej obstaja
  natanko en homomorfizem $R$-modulov $g$ kot zgoraj, torej $g \circ \kappa =
  \iota$.
  Pokazati Ĺľelimo, da sta $f$ in $g$ inverzni.
  Velja $f \circ g \circ \kappa = f \circ \iota = \kappa = \id \circ \kappa$.
  Oglejmo si torej diagram
  \[\begin{tikzcd}
	  X & {\oplus_x R} \\
	  {\oplus_x R}
	  \arrow["\kappa", from=1-1, to=1-2]
	  \arrow["\kappa"', from=1-1, to=2-1]
	  \arrow["{f \circ g}"{description}, curve={height=6pt}, from=1-2, to=2-1]
	  \arrow["{\id}"{description}, curve={height=6pt}, dotted, from=2-1, to=1-2]
	\end{tikzcd}\]
  Za oba homomorfizma diagram komutira, torej velja $f \circ g = \id$.
  Podobno v drugo smer.
\end{proof}

\vprasanje{Povej in dokaĹľi izrek o univerzalni lastnosti prostih modulov.}

\begin{primer}
  Naj bo $F$ polje in $R$ mnoĹľica endomorfizmov
  \[
	R = \End\!\left( \oplus_{n \in \N} F \right),
  \]
  torej neskonÄŤnih kvadratnih matrik elementov iz $F$.
  Kot $R$-modul je $R$ prost z bazo $\{\id\}$.
  Imamo pa tudi bazo iz dveh elementov:
  \begin{align*}
	B_1 =
	\begin{bmatrix}
	  1 & 0 & 0 & 0 & 0 & \cdots \\
	  0 & 0 & 1 & 0 & 0 & \cdots \\
	  0 & 0 & 0 & 0 & 0 & \cdots \\
	  \vdots & \vdots & \vdots & \vdots & \vdots & \ddots
	\end{bmatrix}
	&& B_2 =
	   \begin{bmatrix}
		 0 & 1 & 0 & 0 & 0 & \cdots \\
		 0 & 0 & 0 & 1 & 0 & \cdots \\
		 0 & 0 & 0 & 0 & 0 & \cdots \\
		 \vdots & \vdots & \vdots & \vdots & \vdots & \ddots
	   \end{bmatrix}
  \end{align*}
  Vsaka matrika $A$ je $R$-linearna kombinacija
  \[
	A = [a_1, a_3, a_5, \ldots] B_1 + [a_2, a_4, a_6, \ldots] B_2,
  \]
  torej je $\{B_1, B_2\}$ res baza (preverimo lahko, da sta res linearno
  neodvisni).
\end{primer}

\begin{definicija}
  Kolobar $R$ ima \pojem{lastnost enoliÄŤnega ranga}, ÄŤe imajo vse baze
  poljubnega prostega $R$-modula isto kardinalnost.
\end{definicija}

\vprasanje{Povej primer kolobarja, ki nima lastnosti enoliÄŤnega ranga.}

\begin{primer}
  Polja in obsegi imajo lastnost enoliÄŤnega ranga.
\end{primer}

\begin{definicija}
  Naj bo $R$ kolobar, $I \edinka R$ in $M$ $R$-modul.
  Potem je
  \[
	IM = \{ \sum_{i=1}^k u_i m_i \such u_i \in I, m_i \in M, k \in \N \}
  \]
  podmodul v $M$.
\end{definicija}

\begin{lema}
  ÄŚe je $M$ prost $R$-modul z bazo $X$ in $I \triangleleft R$ razliÄŤen od $R$,
  potem je $M/IM$ prost $R/I$-modul z bazo $X + IM$.
\end{lema}

\begin{proof}
  OÄŤitno je $\sk{X + IM} = M/IM$, ker je $\sk{X} = M$.
  Linearna neodvisnost:
  \begin{align*}
	& \sum_i (r_i + I) (x_i + IM) = 0 + IM \\
	\nt& \sum_i (r_i x_i + IM) = 0 + IM \\
	\nt& \sum_i r_i x_i \in IM.
  \end{align*}
  Velja
  \[
	\sum_i r_i x_i = \sum_j u_j m_j = \sum_j u_j \left( \sum_k s_k x_k \right)
	= \sum_{j,k} \underbrace{u_j s_k}_{\in I} x_k.
  \]
  To sta dva razvoja po bazi $X$, torej $r_i = \sum_j u_j s_i \in I$, in $r_i +
  I = 0 + I$.
\end{proof}

\vprasanje{DokaĹľi: ÄŤe je $M$ prost $R$-modul z idealom $I$, je $M/IM$ prost
  $R/I$-modul.}

\begin{lema}
  Ob oznakah prejĹˇnje leme je $\abs{X} = \abs{X + IM}$.
\end{lema}

\begin{proof}
  Recimo, da je $x + IM = x' + IM$.
  Potem je $x - x' \in IM$, torej
  \[
	x - x' = \sum_j u_j x_j.
  \]
  ÄŚe je kakĹˇen $u_j = 1$, je $I = R$, kar po predpostavki ne velja.
  Torej so vsi razliÄŤni od $1$, in zaradi enoliÄŤnosti razvoja po bazi velja $x =
  x'$.
\end{proof}

\begin{izrek}
  Veljata naslednji toÄŤki o lastnosti enoliÄŤnega ranga:
  \begin{itemize}
  \item Naj bo $R$ kolobar.
	Recimo, da obstaja pravi ideal $I \edinka R$, da ima $R/I$ lastnost
	enoliÄŤnega ranga.
	Potem ima $R$ lastnost enoliÄŤnega ranga.
  \item Vsak komutativen kolobar ima lastnost enoliÄŤnega ranga.
  \end{itemize}
\end{izrek}

\begin{proof}
  Prva toÄŤka:
  Naj bo $M$ poljuben prost $R$-modul z bazama $X$ in $Y$.
  Vemo, da je $M/IM$ prost $R/I$-modul z bazama $X + IM$ in $Y + IM$.
  Ker ima $R/I$ lastnost enoliÄŤnega ranga, je $\abs{X + IM} = \abs{Y + IM}$,
  torej je po lemi $\abs{X} = \abs{Y}$.

  Druga toÄŤka:
  Naj bo $R$ komutativen kolobar.
  ÄŚe je $R$ polje, ima lastnost enoliÄŤnega ranga.
  ÄŚe pa ni polje, ima nek pravi ideal, ki je vsebovan v nekem maksimalnem idealu
  $I$.
  Potem je $R/I$ polje, torej ima lastnost enoliÄŤnega ranga.
  Po prvi toÄŤki ima tako $R$ lastnost enoliÄŤnega ranga.
\end{proof}

\podnaslov{Projektivni moduli}

\begin{definicija}
  $R$-modul $P$ je \pojem{projektiven}, ÄŤe za vsak homomorfizem $R$-modulov $f:
  P \to M$ in vsak epimorfizem $R$-modulov $g: M' \to M$ obstaja homomorfizem
  $R$-modulov $h: P \to M'$, da naslednji diagram komutira:
  \[\begin{tikzcd}
	  P \\
	  M & {M'}
	  \arrow["f"', from=1-1, to=2-1]
	  \arrow["h", dotted, from=1-1, to=2-2]
	  \arrow["g", two heads, from=2-2, to=2-1]
	\end{tikzcd}\]
\end{definicija}

\vprasanje{Kaj je projektiven modul?}

\begin{trditev}
  Vsak prost $R$-modul je projektiven.
\end{trditev}

\begin{proof}
  Naj bo $F$ prost modul z bazo $X$.
  Naj bo $\iota : X \to M$ vloĹľitev, $f$ homomorfizem $F \to M$ in $g: M' \to M$
  epimorfizem.
  Definiramo $\varphi: X \to M'$ tako, da $\varphi(x)$ izberemo nek element
  $M'$, ki se z $g$ slika v $f(\iota(x))$.
  Ker je $F$ prost, obstaja natanko en homomorfizem $h: F \to M'$, da je
  $\varphi(x) = h(\iota(x))$ za $x \in X$.
  Ta ustreza zahtevi projektivnosti.
\end{proof}

\vprasanje{DokaĹľi: vsak prost $R$-modul je projektiven.}

\begin{izrek}
  Za $R$-modul $P$ so ekvivalentne naslednje trditve:
  \begin{itemize}
  \item $P$ je projektiven
  \item za vsak epimorfizem $\varphi: M \to P$ je $M = P \oplus \ker \varphi$
  \item obstaja $R$-modul $M$, da je $P \oplus M$ prost $R$-modul
  \end{itemize}
\end{izrek}

\begin{proof}
  1 v 2:
  Ker je $P$ projektiven, obstaja homomorfizem $R$-modulov $\psi: P \to M$, da
  je $\varphi \circ \psi = \id$.
  Iz tega sledi injektivnost $\psi$, torej je $\im \psi \cong P$.
  Dokazali bomo $M = \jedro \varphi \oplus \im \psi$.
  Naj bo $x \in \jedro \varphi \presek \im \psi$, torej $x = \psi(y)$.
  Sledi $\varphi(\psi(y)) = 0$, torej je $x = 0$.
  Naj bo $m \in M$.
  ZapiĹˇemo
  \[
	m = \underbrace{\psi(\varphi(m))}_{\in \im \psi}
	+ \underbrace{(m - \psi(\varphi(m)))}_{\in \ker \varphi}.
  \]

  2 v 3:
  Vsak modul je kvocient prostega modula, torej obstaja nek prost $F$, in
  epimorfizem $f: F \to P$.
  Po predpostavki je $F = \ker \varphi \oplus P$.

  3 v 1:
  Obstaja $R$-modul $N$, da je $P \oplus N$ prost.
  Ta je projektiven, torej za projekcijo na prvo komponento $\pi_1 : P \oplus N
  \to P$ obstaja $\tilde{h}: P \to M'$, da naslednji diagram komutira:
  \[\begin{tikzcd}
	  & {P \oplus N} \\
	  {M'} & M
	  \arrow["{\tilde{h}}"', from=1-2, to=2-1]
	  \arrow["{f \circ \pi_1}", from=1-2, to=2-2]
	  \arrow[two heads, from=2-1, to=2-2]
	\end{tikzcd}\]
  Definiramo $h: P \to M'$ kot $h(p) = \tilde{h}(p,0)$.
\end{proof}

  $\Z_2$ in $Z_3$ sta $\Z_6$-modula, $\Z_6 = \Z_2 \oplus \Z_3$ pa je prost

\podnaslov{Tenzorski produkt modulov}

V razdelku gledamo le module nad komutativnimi kolobarji z enico.

\begin{definicija}
  Naj bosta $M, N$ $R$-modula.
  Naj bo $F$ prost $R$-modul nad mnoĹľico $M \times N$ in $T$ podmodul v $F$,
  generiran z naslednjimi elementi:
  \begin{itemize}
  \item $(m_1 + m_2, n) - (m_1, n) - (m_2, n)$
  \item $(m, n_1 + n_2) - (m, n_1) - (m, n_2)$
  \item $(rm, n) - r(m, n)$
  \item $(m, rn) - r(m, n)$
  \end{itemize}
  Potem je \pojem{tenzorski produkt} $R$-modulov $M, N$ $R$-modul
  \[
	M \otimes_R N = F/T.
  \]
\end{definicija}

\vprasanje{Definiraj tenzorski produkt.}

Za elementarne tenzorje veljajo naslednje lastnosti:
\begin{itemize}
\item $(m_1 + m_2) \otimes n = m_1 \otimes n + m_2 \otimes n$
\item $m \otimes (n_1 + n_2) = m \otimes n_1 + m \otimes n_2$
\item $(rm) \otimes n = r(m \otimes n)$
\item $m \otimes (rn) = r (m \times n)$
\end{itemize}

\begin{opomba}
  Preslikava $\beta: M \times N \to M \otimes_R N$, definirana z $\beta(m,n) = m
  \otimes n$, je bilinearna.
\end{opomba}

\begin{izrek}[univerzalna lastnost]
  Naj bodo $M, N, K$ $R$-moduli.
  Za vsako bilinearno preslikavo $\gamma : M \times N \to K$ obstaja natanko en
  homomorfizem $R$-modulov $f: M \otimes_R N \to K$, da naslednji diagram
  komutira:
  \[\begin{tikzcd}
	  {M \times N} & K \\
	  {M \otimes_R N}
	  \arrow["\gamma", from=1-1, to=1-2]
	  \arrow["\beta"', from=1-1, to=2-1]
	  \arrow["f"', dashed, from=2-1, to=1-2]
	\end{tikzcd}\]
\end{izrek}

\begin{proof}
  Naj bosta $F, T$ kot v definiciji tenzorskega produkta.
  Ker je $F$ prost nad $M \times N$, obstaja natanko en homomorfizem $\tilde{f}:
  F \to K$, da diagram
  \[\begin{tikzcd}
	  {M \times N} & K \\
	  F
	  \arrow["\gamma", from=1-1, to=1-2]
	  \arrow["\iota"', from=1-1, to=2-1]
	  \arrow["{\tilde{f}}"', dashed, from=2-1, to=1-2]
	\end{tikzcd}\]
  komutira.
  Velja $\tilde{f}(m, n) = \gamma(m, n)$.
  Trdimo, da je $T \subseteq \jedro \tilde{f}$; velja
  \[
	\tilde{f}((m_1 + m_2, n) - (m_1, n) - (m_2, n))
	= \gamma(m_1 + m_2, n) - \gamma(m_1, n) - \gamma(m_2, n)
	= 0,
  \]
  ker je $\gamma$ bilinearna.
  Podobno pokaĹľemo, da je $\tilde{f}$-slika ostalih generatorjev $0$, torej
  $\tilde{f}$ inducira natanko en homomorfizem $f: F/T \to K$.
\end{proof}

\vprasanje{DokaĹľi, da za tenzorski produkt velja univerzalna lastnost.}

\begin{izrek}
  Naj bodo $M, N, P$ $R$-moduli.
  Naj bo $\beta: M \times N \to P$ poljubna bilinearna preslikava.
  Recimo, da za vsak $R$-modul $K$ in vsako bilinearno preslikavo $\gamma: M
  \times N \to K$ obstaja natanko en homomorfizem $R$-modulov $f: P \to K$, da
  naslednji diagram komutira:
  \[\begin{tikzcd}
	  {M \times N} & K \\
	  P
	  \arrow["\gamma", from=1-1, to=1-2]
	  \arrow["\beta"', from=1-1, to=2-1]
	  \arrow["f"', dashed, from=2-1, to=1-2]
	\end{tikzcd}\]
  Potem sta $P$ in $M \otimes_R N$ izomorfna.
\end{izrek}

\begin{proof}
  Za $\alpha(m, n) = m \otimes n$ po prejĹˇnjem izreku obstaja natanko en $g: M
  \otimes_R N \to R$, da naslednji diagram komutira:
  \[\begin{tikzcd}
	  {M \times N} & P \\
	  {M \otimes_R N}
	  \arrow["\beta", from=1-1, to=1-2]
	  \arrow["\alpha"', from=1-1, to=2-1]
	  \arrow["g"', dashed, from=2-1, to=1-2]
	\end{tikzcd}\]
  Po predpostavki obstaja natanko en $f: P \to M \otimes_R N$, da je $f\circ
  \beta = \alpha$.
  Preverimo lahko, da sta $f$ in $g$ inverzna.
\end{proof}

\begin{trditev}
  Naj bo $M$ $R$-modul.
  Potem je $R \otimes_R M \cong M$.
\end{trditev}

\begin{proof}
  Definiramo $\beta: R \times M \to M$ s predpisom $\beta(r,m) = rm$.
  To je bilinearna preslikava, torej po univerzalni lastnosti inducira
  homomorfizem $R$-modulov $R \otimes_R M \to M$, ki slika $r \otimes m \mapsto
  rm$.
  V drugo smer podamo homomorfizem $M \to R \otimes_R M$ s predpisom $m \mapsto
  1 \otimes m$.
  Preverimo lahko, da sta preslikavi inverzni.
\end{proof}

\vprasanje{Kaj je $R \otimes_R M$? DokaĹľi.}

\begin{izrek}
  Naj bo $M$ prost $R$-modul nad $X = \{m_i \such i \in I\}$ in $N$ prost
  $R$-modul nad $Y = \{n_j \such j \in J\}$.
  Potem je $M \otimes_R N$ tudi prost $R$-modul nad $Z = \{m_i \otimes n_j \such
  i \in I, j \in J\}$.
\end{izrek}

\begin{proof}
  MnoĹľica $Z$ oÄŤitno generira $M \otimes_R N$.
  Naj bo
  \[
	\sum_{ij} r_{ij} (m_i \otimes_R n_j) = 0.
  \]
  Naredimo homomorfizem $f_k: N \to R$ za vsak $k \in J$, s predpisom $f_k(n_j)
  = \delta_{kj}$.
  Preslikava je definirana na bazi, lahko jo razĹˇirimo do homomorfizma $N \to
  R$.
  Oglejmo si preslikavo $\id_M \otimes_R f_k : M \otimes_R N \to M \otimes_R R$.
  Slika zgornje linearne kombinacije s to preslikavo je
  \[
	0 = \sum_{ij} r_{ij} (m_i \otimes \delta_{kj}) = \sum_i r_{ik} m_i \otimes 1.
  \]
  Ker je $M \otimes_R R \cong M$, velja $\sum_i r_{ik} m_i = 0$ (slikamo z
  izomorfizmom), torej je $r_{ik} = 0$ za vsak $i$.
  To lahko naredimo za poljuben $k$.
\end{proof}

\vprasanje{PokaĹľi, da je tenzorski produkt prostih modulov prost.}

\begin{trditev}
  Naj bodo $A, B, C$ $R$-moduli.
  Potem velja:
  \begin{itemize}
  \item $A \otimes_R B \cong B \otimes_R A$
  \item $A \otimes_R (B \oplus C) \cong (A \otimes_R B) \oplus (A \otimes_R C)$
  \item $A \otimes_R (B \otimes_R C) \cong (A \otimes_R B) \otimes_R C$
  \end{itemize}
\end{trditev}

\podnaslov{SkrÄŤitev in razĹˇiritev skalarjev}

Naj bo $\varphi : S \to R$ homomorfizem komutativnih kolobarjev.
ÄŚe je $M$ $R$-modul, postane tudi $S$-modul preko predpisa
\[
  s \circ m = \varphi(s) \cdot m.
\]
Predpis je dobro definiran, temu pravimo \pojem{skrÄŤitev skalarjev}.
Podobno lahko naredimo tudi v obratni smeri:
ÄŤe je $M$ $S$-modul, je po ravno ugotovljenem $R$ tudi $S$-modul, in je $R
\oplus_R M$ $R$-modul z mnoĹľenjem
\[
  r(r' \otimes m) = rr' \otimes m.
\]
Temu pravimo \pojem{razĹˇiritev skalarjev}.

\vprasanje{Kaj je razĹˇiritev in kaj skrÄŤitev skalarjev?}

\begin{trditev}
  Naj bo $\varphi: R \to S$ homomorfizem komutativnih kolobarjev.
  Naj bo $M$ $R$-modul in $N$ $S$-modul.
  Z uporabo skrÄŤitve oz.~razĹˇiritve skalarjev lahko definiramo Abelovi grupi
  $\operatorname{Hom}_R(M,N)$ in $\operatorname{Hom}_S(S \otimes_R M, N)$, ki
  sta izomorfni.
\end{trditev}

\begin{izrek}[Dualnost med $\operatorname{Hom}$ in $\otimes$]
  Naj bodo $M, N, P$ $R$-moduli za komutativen kolobar $R$.
  Potem sta $\operatorname{Hom}_R(M \otimes_R N, P)$ in $\operatorname{Hom}_R(M,
  \operatorname{Hom}_R(N, P))$ izomorfna kot $R$-modula.
\end{izrek}

\podnaslov{Eksaktna zaporedja modulov}

\begin{definicija}
  Naj bodo $A, B, C$ $R$-moduli.
  Imejmo homomorfizme $R$-modulov
  \[\begin{tikzcd}
	  A & B & C
	  \arrow["f", from=1-1, to=1-2]
	  \arrow["g", from=1-2, to=1-3]
	\end{tikzcd}\]
  To zaporedje je \pojem{eksaktno} pri $B$, ÄŤe velja $\im f = \jedro g$.
\end{definicija}

\begin{vo}{Kaj je eksaktno zaporedje? Karakteriziraj injektivnost in
	surjektivnost z eksaktnostjo.}
  Za definicijo glej zgoraj.
  Oglejmo si zaporedje
  \[\begin{tikzcd}
	  0 & A & B & C & 0
	  \arrow[from=1-1, to=1-2]
	  \arrow["f", from=1-2, to=1-3]
	  \arrow["g", from=1-3, to=1-4]
	  \arrow[from=1-4, to=1-5]
	\end{tikzcd}\]
  To zaporedje je eksaktno pri $A$ natanko tedaj, ko je $f$ injektivna, in
  eksaktno pri $C$ natanko tedaj, ko je $g$ surjektivna.
\end{vo}

Za zaporedje
\[\begin{tikzcd}
	0 & A & B & C & 0
	\arrow[from=1-1, to=1-2]
	\arrow["f", from=1-2, to=1-3]
	\arrow["g", from=1-3, to=1-4]
	\arrow[from=1-4, to=1-5]
  \end{tikzcd}\]
pravimo, da je \pojem{kratko eksaktno}, ÄŤe je eksaktno pri $A$, $B$ in $C$.
IzkaĹľe se, da so vsa kratka eksaktna zaporedja oblike
\[\begin{tikzcd}
	0 & B & A & {A/B} & 0
	\arrow[from=1-1, to=1-2]
	\arrow[hook, from=1-2, to=1-3]
	\arrow[from=1-3, to=1-4]
	\arrow[from=1-4, to=1-5]
  \end{tikzcd}\]
