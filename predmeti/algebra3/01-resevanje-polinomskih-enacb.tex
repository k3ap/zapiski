\naslov{Reševanje polinomskih enačb}

Vemo, da za polinomsko enačbo $x^n + a_{n-1} x^{n-1} + \cdots + a_0 = 0$ obstaja
razširitev polja $\F$, v kateri je enačba razcepna.
Rešitev polinomske enačbe se izraža z radikali, če se da zapisati rešitve enačbe
s pomočjo računskih operacij in korenov.

\begin{definicija}
  Naj bo $K$ polje.
  Razširitvi oblike $K(\sqrt[n]{a})/K$ za $a \in K$ pravimo \pojem{radikalska
	razširitev} polja $K$.
\end{definicija}

Polinomska enačba $p(X) = 0$ je rešljiva z radikali natanko tedaj, ko rešitve
živijo v neki razširitvi $E/F$, pri čemer obstaja končna veriga razširitev $F
\subseteq E_0 \subseteq \ldots \subseteq E_k = E$, kjer so zaporedne razširitve
radikalske.
Problem se prevede na vprašanje iz teorije grup.
Od tu naprej predpostavimo, da so vse razširitve končne.

\begin{definicija}
  Naj bo $E/F$ končna razširitev.
  To je \pojem{normalna razširitev}, če za vsak nerazcepen polinom $p(X) \in
  F[X]$ velja ena od naslednjih možnosti:
  \begin{itemize}
  \item $p$ nima ničle v $E$
  \item $p$ ima vse ničle v $E$
  \end{itemize}
  Ekvivalentno: vsak nerazcepen polinom $p(X) \in F[X]$ z vsaj eno ničlo v $E$
  razpade v linearne faktorje v $E[X]$.
\end{definicija}

\begin{primer}
  $\Q(\sqrt[3]{2})$ nad $\Q$ ni normalna.
  Polinom $p(X) = X^3 - 2$ ne razpade.
\end{primer}

\begin{primer}
  $\Q(\sqrt{2})$ je normalna.
\end{primer}

\vprasanje{Definiraj radikalske in normalne razširitve. Povej primer normalne
  razširitve in razširitve, ki ni normalna.}

\begin{izrek}
  Naj bo $E/F$ končna razširitev.
  Potem je ta razširitev normalna natanko tedaj, ko je $E$ razpadno polje nekega
  polinoma s koeficienti iz $F$.
\end{izrek}

\begin{proof}
  V desno:
  Recimo, da je $E$ normalna.
  Naj bo $p_i(X)$ minimalni polinom $a_i$.
  Definiramo $p(X) = p_1(X) \ldots p_k(X)$.
  Zaradi normalnosti so vse ničle polinoma $p_i(X)$ v $E$, torej so vse ničle
  polinoma $p(X)$ v $E$.
  Velja $F(p) \subseteq E$, ampak $a_1, \ldots, a_k \in F(p)$, torej $F(p)
  \supseteq F(a_1, \ldots, a_k) = E$.

  V levo:
  Recimo, da je $E = F(p)$.
  Vzemimo poljuben nerazcepen $q(X) \in F[X]$, ki ima neko ničlo $a \in E$.
  Naj bo $b$ poljubna ničla $q$.
  Ničli $a$ in $b$ imata isti minimalni polinom, $q$, torej je $F(a) \cong
  F(b)$.
  Izomorfizem $\delta$ lahko razširimo na izomorfizem razpadnih polj $\tau$.
  Predpostavimo lahko, da je $\delta(a) = b$, torej tudi $\tau(a) = b$.
  Ničla $a$ je racionalen izraz, odvisen od $a_1, \ldots, a_k$, torej je tudi
  $b$ racionalen izraz, odvisen od $a_1, \ldots, a_k$, in posledično $b \in E$.
\end{proof}

\vprasanje{Karakteriziraj normalnost za končne razširitve. Dokaži
  karakterizacijo.}

\begin{definicija}
  Polinom $p(X) \in F[X]$ je \pojem{separabilen}, če ima same enostavne ničle.
  Končna razširitev $E/F$ je \pojem{separabilna}, če je minimalni polinom
  vsakega elementa $a \in E$ separabilen.
\end{definicija}

\begin{opomba}
  V karakteristiki $0$ je vsaka končna razširitev separabilna.
\end{opomba}

\begin{izrek}[Primitivni element]
  Vsaka separabilna razširitev je enostavna.
\end{izrek}

\begin{proof}
  Če je $F$ končno polje, je tudi $E$ končno in je $(E^*, \cdot)$ ciklična
  grupa, generirana z $a$.
  Velja $E = F(a_1, \ldots, a_n)$, brez škode za splošnost $a_i \ne 0$, torej
  obstajajo $n_i$, da je $a_i = a^{n_i}$.
  Torej je $E = F(a)$.

  Če pa je $F$ neskončno polje, zapišimo $E = F(a_1, \ldots, a_n)$, ker je
  razširitev končna.
  Poleg tega so $a_i$ algebraični.
  Ker velja $F(a_1, \ldots, a_n) = F(a_1, \ldots, a_{n-2})(a_{n-1}, a_n)$, izrek
  zadošča dokazati za $n = 2$.

  Naj bo torej $E = F(b, c)$ ter $p(X)$ in $q(X)$ minimalna polinoma $b$ in $c$.
  Označimo z $E_1$ razpadno polje polinoma $p(X) q(X)$ in z $b = b_1, \ldots,
  b_r$, $c = c_1, \ldots, c_s$ njune ničle v tem polju.
  Izberimo $\lambda \in F$ tako, da $\lambda \ne (b_j - b){(c - c_k)}^{-1}$ za vse
  $j$ in $k$.
  Tak $\lambda$ res obstaja, ker je polje neskončno.
  Sedaj definiramo $a = b + \lambda c$.
  Očitno je $F(a) \subseteq F(b, c)$; trdimo, da velja tudi vsebovanost v drugi
  smeri.
  Vpeljimo $f(X) = p(a - \lambda X) \in F(a)[X]$.
  Velja $f(c) = p(a - \lambda c) = p(b) = 0$, torej je $c$ hkrati ničla $f(X)$
  in $q(X) \in F[X] \subseteq F(a)[X]$.
  Naj bo $\tilde{q}(X) \in F(a)[X]$ minimalni polinom elementa $c$ nad poljem
  $F(a)$.
  Velja $\tilde{q} | f$ in $\tilde{q} | q$.

  Možne ničle polinoma $\tilde{q}$ so skupne ničle polinomov $q$ in $f$.
  Vemo, da imata skupno ničlo $c$; katere od $c_i$ pa so še ničle $f$?
  Če je $f(c_j) = p(a - \lambda c_j) = 0$, potem je $a - \lambda c_j = b_k$ za
  nek $k$, torej $b + \lambda c - \lambda c_j = b_k$ in posledično $\lambda =
  (b_j - b){(c - c_k)}^{-1}$, kar je po predpostavki nemogoče.
  Torej je $c$ edina ničla $\tilde{q}$.
  Ker je to minimalni polinom in polje $F$ separabilno, je ničla enostavna in
  zato $\tilde{q}(X) = X-c \in F(a)[X]$, torej $c \in F(a)$.
\end{proof}

\begin{primer}
  Razširitev $E = F_2(\sqrt{t})$ je algebraična razširitev, ki ni separabilna.
  Minimalni polinom za $\sqrt{t}$ je $x^2 - t = {(x + \sqrt{t})}^2$.
\end{primer}

\vprasanje{Kdaj je razširitev separabilna? Podaj primer končne razširitve, ki ni
  separabilna. Povej izrek o primitivnem elementu in ga dokaži za končna polja.}

\begin{definicija}
  \pojem{Galoisova grupa razširitve} $E/F$ je grupa
  \[
	\Gal(E/F) = \{\sigma \in \Aut(E) \such \left. \sigma \right|_F =
	\operatorname{id}_F\}.
  \]
\end{definicija}

\begin{lema}
  Naj bo $E/F$ razširitev, $a \in E$ ničla $p(X) \in F[X]$ ter $\sigma \in
  \Gal(E/F)$.
  Potem je $\sigma(a)$ tudi ničla $p(X)$.
\end{lema}

\begin{proof}
  Uporabimo $\sigma$ na zapisu $p(X) = a_n X^n + \cdots + a_0$.
\end{proof}

\begin{definicija}
  Naj bo $p(X) \in F[X]$ polinom.
  \pojem{Galoisova grupa} polinoma $p$ je $\Gal(p) = \Gal(F(p)/F)$.
\end{definicija}

\vprasanje{Definiraj Galoisovo grupo razširitve in polinoma.}

\begin{definicija}
  Normalnim separabilnim razširitvam pravimo \pojem{Galoisove razširitve}.
\end{definicija}

\begin{trditev}
  Naj bo $E/F$ končna Galoisova razširitev.
  Potem je $\abs{\Gal(E/F)} = [E:F]$.
\end{trditev}

\begin{proof}
  Po izreku o primitivnem elementu obstaja $a \in E$, da je $E = F(a)$.
  Naj bo $p(X) \in F[X]$ minimalni polinom $a$.
  Zaradi normalnosti so vse ničle $p$ v $E$, torej je za $\sigma \in \Gal(E/F)$
  slika $\sigma(a)$ lahko katerakoli ničla $p$.
  Ničel je ravno toliko kot je stopnja razširitve, ker ima $p$ paroma različne
  ničle.
  Torej je za $\sigma$ natanko toliko možnosti.
\end{proof}

\begin{opomba}
  Če je $E/F$ končna separabilna razširitev, je $\abs{\Gal(E/F)} \le [E:F]$.
\end{opomba}

\vprasanje{Kolikšen je red Galoisove grupe končne Galoisove razširitve? Dokaži.}

\begin{lema}
  Naj bo $F \subseteq L \subseteq E$.
  \begin{itemize}
  \item Če je $E/F$ končna, je tudi $E/L$ končna.
  \item Če je $E/F$ normalna, je tudi $E/L$ normalna.
  \item Če je $E/F$ separabilna, je tudi $E/L$ separabilna.
  \end{itemize}
\end{lema}

\begin{proof}
  Prva točka je enostavna, dokaz tretje pa je podoben dokazu druge, torej
  dokažemo le to.
  Naj bo $p(X) \in L[X]$ nerazcepen z ničlo $a \in E$ in
  naj bo $q(X) \in F[X]$ minimalen polinom za $a$ nad $F$.
  Trdimo, da $p(X)$ deli $q(X)$.
  Vzemimo največji skupni delitelj $d(X) \in L[X]$ polinomov $p(X)$ in $q(X)$.
  Ker imata skupno ničlo $a$, $d$ ni konstanten, $d(a) = 0$ in $d$ deli $p$.
  Ker je $p$ nerazcepen, mora biti $d(X) = p(X)$, torej $p$ res deli $q$.
  Vsaka ničla $b$ polinoma $p$ je torej tudi ničla $q$.
  Ker je $E$ normalna, je $b \in E$.
\end{proof}

\begin{izrek}[Fundamentalni izrek Galoisove teorije]
  Naj bo $E/F$ končna Galoisova razširitev.
  \begin{itemize}
  \item Predpisa $L \mapsto \Gal(E/L)$ in $G \mapsto E^G$ sta paroma inverzni
	preslikavi med vmesnimi polji razširitve in podgrupami Galoisove grupe.
  \item Za poljubni vmesni polji $F \subseteq L \subseteq M \subseteq E$ velja
	$[M:L] = \abs{\Gal(E/F) : Gal(E/M)}$.
  \item Za vmesno polje $F \subseteq L \subseteq E$ je $L/F$ normalna natanko
	tedaj, ko je $\Gal(E/L) \edinka \Gal(E/F)$.
	Velja še $\Gal(L/F) \cong \Gal(E/F) / \Gal(E/L)$.
  \end{itemize}
\end{izrek}

\begin{proof}
  Za prvo točko:
  Dokažimo, da za vsak $L$ velja
  \[
	E^{\Gal(E/L)} = L.
  \]
  Očitno je $L \subseteq E^{\Gal(E/L)}$.
  Trdimo, da za vsak $a \in E \brez L$ obstaja $\sigma \in \Gal(E/L)$, ki ga ne
  fiksira.
  Naj bo $q(X) \in L[X]$ minimalni polinom $a$.
  Ker $a \notin L$, je $\st q > 1$, torej ima $q$ od $a$ različno ničlo $b \in
  E$.
  Polji $L(a)$ in $L(b)$ sta izomorfni, izomorfizem lahko razširimo do
  izomorfizma razpadnih polj polinoma $q$; ta izomorfizem fiksira elemente $L$
  in slika $a$ v $b$.

  Sedaj pokažimo $\Gal(E/E^G) = G$.
  Očitno je $G \subseteq \Gal(E/E^G)$, za drugo inkluzijo pa je dovolj dokazati
  (ker so vse grupe končne), da je $[E:E^G] \le \abs{G}$.
  Zaradi separabilnosti je $E = E^G(a)$.
  Naj bo $q(X) \in E^G[X]$ minimalni polinom $a$.
  Velja $\st q = [E:E^G]$.
  Definiramo
  \[
	p(X) = \prod_{\tau \in G} (X - \tau(a)) \in E[X].
  \]
  Trdimo, da je $p(X) \in E^G[X]$, za kar moramo pokazati, da vsak $\sigma \in
  G$ fiksira koeficiente polinoma.
  Označimo polinom slik koeficientov z
  \[
	\sigma(p(X)) = \prod_{\tau \in G} (X - \sigma(\tau(a))).
  \]
  Ko $\tau$ preteče cel $G$, tudi $\sigma \tau$ preteče cel $G$, torej
  $\sigma(p(X)) = p(X) \in E^G[X]$.
  Velja $p(a) = 0$, torej $q$ deli $p$ in zato $[E:E^G] = \st q \le \st p =
  \abs{G}$.

  Druga točka:
  Preprost račun
  \[
	[M:L]
	= \frac{[E:L]}{[E:M]} = \frac{\abs{\Gal(E/L)}}{\abs{\Gal(E/M)}}
	= \abs{\Gal(E/L) : \Gal(E/M)}.
  \]

  Tretja točka:
  Dokaz v več korakih.
  Prvo pokažimo, da je $L/F$ normalna natanko tedaj, ko za vsak $\sigma \in
  \Gal(E/F)$ velja $\sigma(L) = L$.
  V desno naj bo $\sigma \in \Gal(E/F)$.
  Velja $L = F(a_1, \ldots, a_k)$, označimo s $p_i$ minimalni polinom $a_i$.
  Ker je razširitev normalna, so vse ničle $p_i$ v $L$, torej se $a_i$ s
  $\sigma$ slika v neko ničlo $p_i$.
  Torej je $\sigma(a_i) \in L$ za vse $i$, drugo inkluzijo pa dobimo z inverzom
  $\sigma^{-1}$.
  Podobno v drugo smer.

  Sedaj pokažimo, da za $F \subseteq L \subseteq E$ in poljuben $\sigma \in
  \Gal(E/F)$ velja $\Gal(E/\sigma(L)) = \sigma \Gal(E/F) \sigma^{-1}$.
  To velja zato, ker je $\tau \in \Gal(E/\sigma(L))$ natanko tedaj, ko je
  $\left. \tau \right|_{\sigma(L)} = \operatorname{id}$, oziroma
  $\tau(\sigma(x)) = \sigma(x)$ za vsak $x \in L$.
  Če množimo z leve s $\sigma^{-1}$, dobimo $\sigma^{-1} \tau \sigma (x) = x$ za
  $x \in L$, kar je ekvivalentno zahtevi $\sigma^{-1} \tau \sigma \in \Gal(E/L)$.

  V tretjem koraku opazimo, da je desni del ekvivalence v prvem koraku enaka
  zahtevi, da za vsak $\sigma \in \Gal(E/F)$ velja $\Gal(E/F) =
  \Gal(E/\sigma(L))$, oziroma $\sigma \Gal(E/L) \sigma^{-1} = \Gal(E/L)$, kar pa
  je po definiciji enako $\Gal(E/L) \edinka \Gal(E/F)$.

  V zadnjem koraku recimo, da je $L/F$ normalna, in definirajmo $\phi :
  \Gal(E/F) \to \Gal(L/F)$ kot
  \[
	\phi(\sigma) = \left. \sigma \right|_L.
  \]
  Po prvem koraku dokaza prejšnje točke zožena preslikava res slika $L$ v $L$,
  iz izreka o izomorfizmu torej dobimo $\Gal(E/F) / \Gal(E/L) \cong \Gal(L/F)$.
\end{proof}

\vprasanje{Povej in dokaži fundamentalni izrek Galoisove teorije.}

\podnaslov{Rešljive grupe}

\begin{definicija}
  Grupa $G$ je \pojem{rešljiva}, če obstaja končno zaporedje podgrup
  \[
	\{e\} = G_0 \subseteq G_1 \subseteq \cdots \subseteq G_k = G,
  \]
  da je za vsak $i$ $G_i \edinka G_{i+1}$ ter $G_{i+1} / G_i$ Abelova grupa.
\end{definicija}

\begin{vo}{Kaj je rešljiva grupa? Povej nekaj pozitivnih in negativnih primerov.}
  Abelove grupe so rešljive, enostavne nekomutativne grupe pa niso.
  Za $H = \sk{(1 2 3)}$ je $\{\id\} \edinka H \edinka S_3$, za $K = \sk{(1 2) (3
	4), (1 4)(3 2)}$ pa $\{\id\} \edinka K \edinka A_4 \edinka S_4$, preverimo
  lahko, da sta tako $S_3$ kot $S_4$ rešljivi.
\end{vo}

\begin{trditev}
  Vsaka podgrupa rešljive grupe je rešljiva.
\end{trditev}

\begin{proof}
  Naj bo $G$ rešljiva, $\{e\} = G_0 \edinka \cdots \edinka G_k = G$, in naj bo
  $H \le G$.
  Potem je $G_i \presek H \edinka G_{i+1} \presek H$, za kvocient pa velja
  \[
	\frac{G_{i+1} \presek H}{G_i \presek H} = \frac{G_{i+1} \presek H}{G_i
	  \presek G_{i+1} \presek H} \cong \frac{(G_{i+1} \presek H) \cdot G_i}{G_i}
	\le \frac{G_{i+1}}{G_i},
  \]
  kjer smo uporabili drugi izrek o izomorfizmu.
  Grupa $G_{i+1}/G_i$ je Abelova, torej je
  \[
	\{e\} = G_0 \presek H \edinka \cdots \edinka G_k \presek H = H.
  \]
\end{proof}

\vprasanje{Pokaži, da je podgrupa rešljive grupe rešljiva.}

\begin{trditev}
  Kvocient rešljive grupe je rešljiv.
\end{trditev}

\begin{proof}
  Naj bo $G$ rešljiva in $N \edinka G$.
  Preverimo lahko, da velja
  \[
	\{e\} \edinka \frac{G_0 N}{N} \edinka \cdots \edinka \frac{G_k N}{N} =
	\frac{G}{N}.
  \]
  Z uporabo tretjega in drugega izreka o izomorfizmu izračunamo
  \[
	\frac{G_{i+1} N / N}{G_i N / N} \cong \frac{G_{i+1} N}{G_i N}
	= \frac{G_{i+1} G_i N}{G_i N} \cong \frac{G_{i+1}}{G_{i+1} \presek G_i N}
	\cong \frac{G_{i+1} / G_i}{(G_{i+1} \presek G_i N) / G_i}.
  \]
  Kvocient Abelove grupe je Abelova grupa.
\end{proof}

\vprasanje{Pokaži, da je kvocient rešljive grupe rešljiv.}

\begin{trditev}
  Naj bo $N \edinka G$ in naj bosta $N$ in $G/N$ rešljivi.
  Potem je $G$ rešljiva grupa.
\end{trditev}

\begin{proof}
  Naj bosta $N_0 \edinka \cdots \edinka N_k$ in $G_0 / N \edinka \cdots G_l / N$
  zaporedji edink.
  Iz $G_i/N \edinka G_{i+1}/N$ sledi $G_i \edinka G_{i+1}$.
  Po tretjem izreku o izomorfizmu je $G_{i+1}/G_i$ Abelova (ker je
  $\frac{G_{i+1}/N}{G_i/N}$ Abelova).
  Sedaj imamo zaporedje
  \[
	N_0 \edinka N_i \edinka \cdots \edinka N_k = N = G_0 \edinka G_1 \edinka
	\cdots \edinka G_l = G.
  \]
\end{proof}

\podnaslov{Rešljivost polinomskih enačb z radikali}

Dan je polinom $p(X) \in F[X]$.
Rešljivost polinomske enačbe z $p(X) = 0$ z radikali je ekvivalentna obstoju
razširitve $E/F$, ki jo dobimo z zaporedjem radikalskih razširitev $F = E_0
\subseteq E_1 \subseteq \cdots \subseteq E_k = E$.

\begin{lema}
  Naj bo $F$ polje, ki vsebuje primitivni $n$-ti koren enote.
  Naj bo $a \in F$.
  Potem je Galoisova grupa polinoma $X^n - a$ ciklična.
\end{lema}

\begin{proof}
  Naj bo $\varepsilon$ primitivni $n$-ti koren enote.
  Če je $b$ neka ničla $X^n - a$, so vse ničle oblike $b, \varepsilon b, \ldots,
  \varepsilon^{n-1} b$.
  Potem je $F(X^n - a) = F(b, \varepsilon b, \ldots, \varepsilon^{n-1} b) =
  F(b)$, ker je $\varepsilon \in F$.
  Preslikava $\sigma \in \Gal(X^n - a)$ je določena s sliko $b$;
  \[
	b \mapsto \varepsilon^i b.
  \]
  S tem je določena injektivna preslikava $Gal(X^n - a) \to \Z_n$, ki je
  hkrati homomorfizem grup.
\end{proof}

\begin{izrek}
  Enačba $p(X) = 0$ je rešljiva z radikali natanko tedaj, ko je Galoisova grupa
  polinoma $p$ rešljiva grupa.
\end{izrek}

\begin{proof}
  Dokažemo le implikacijo v desno; v drugo smer je podobno.
  Enačba je rešljiva, torej obstajajo razširitve $F = E_0 \subseteq E_1
  \subseteq \cdots \subseteq E_k = E$, da je $E_{i+1} = E_i(a_i)$ in $a_i^{n_i}
  \in E_i$.
  Definiramo $n = n_0 \ldots n_{k-1}$.
  Naj bo $\varepsilon$ primitivni $n$-ti koren enote.
  Za ta dokaz se ne bomo ukvarjali s separabilnostjo; privzamemo npr.~da je
  karakteristika polja $0$.

  Po izreku o primitivnem elementu obstaja $a$, da je $E = F(a)$.
  Naj bo $g(X) \in F[X]$ minimalni polinom $a$ in $\Omega = F(g(X) (X^n - 1))$.
  To je normalna razširitev $F$, ki vsebuje $E$ in $\varepsilon$.
  Vzemimo še $\tilde{E}$ kot normalno zaprtje polja $E(\varepsilon)$ v $\Omega$
  tj.~najmanjša normalna razširitev $E(\varepsilon)$, ki je vsebovana v
  $\Omega$.

  Polje $\tilde{E}$ vsebuje vse $\sigma(E(\varepsilon))$ za $\sigma \in
  \Gal(\Omega / F)$, torej je enako
  \[
	\tilde{E} = F(\varepsilon, a_0, a_1, \ldots, a_{k-1}, \sigma_1(a_0), \ldots,
	\sigma_2(a_0), \ldots).
  \]
  Razširitev $\tilde{E}/F$ lahko naredimo po korakih, po vrstnem redu kot zgoraj.
  V vsaki vmesni razširitvi dodamo ničlo polinoma $X^{n_i} - b_i$ za nek $b_i$,
  torej so razširitve radikalske.
  Preverimo lahko, da so tudi normalne.

  Po Galoisovi korespondenci dobimo verigo podgrup
  \[
	\Gal(\tilde{E}/F) \supseteq \Gal(\tilde{E}/F(\varepsilon)) \supseteq \cdots
	\supseteq \Gal(\tilde{E}/\tilde{E}) = \{e\}.
  \]
  Zaradi normalnosti je to zaporedje edink.
  Kvocienti so po lemi ciklične grupe, torej je $\Gal(\tilde{E}/F)$ rešljiva.
  Grupa
  \[
	\frac{\Gal(\tilde{E}/F)}{\Gal(\tilde{E}/F(p))} \cong \Gal(F(p)/F) = \Gal(p)
  \]
  je kvocient rešljive grupe, torej je rešljiva.
\end{proof}

\vprasanje{Dokaži, da je rešljivost polinomske enačbe $p(X) = 0$ z radikali
  implicira rešljivost grupe $\Gal(p)$.}
