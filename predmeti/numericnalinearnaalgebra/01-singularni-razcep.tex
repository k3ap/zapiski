\naslov{Singularni razcep}

\begin{izrek}
  Za vsako matriko $A \in \R^{m \times n}$, kjer je $m \ge n$, obstaja razcep $A
  = U \Sigma V^T$, kjer je $U \in \R^{m \times m}$ ortogonalna, $V \in \R^{n
	\times n}$ ortogonalna in $\Sigma \in \R^{m \times n}$  oblike
  \[
	\Sigma =
	\begin{bmatrix}
	  \sigma_1 & & \\
			   & \ddots & \\
			   & & \sigma_n \\
	  \multicolumn{3}{c}{0}
	\end{bmatrix}
  \]
  za singularne vrednosti $\sigma_1 \ge \ldots \ge \sigma_n \ge 0$ matrike $A$.
\end{izrek}

\begin{proof}
  ÄŚe je $A = U \Sigma V^T$, potem je $A^T A = V \Sigma^T \Sigma V^T$.
  Ta matrika je simetriÄŤna nenegativno definitna, torej so njene lastne
  vrednosti realne in nenegativne ter jih lahko uredimo padajoÄŤe.
  Lastne vektorje lahko izberemo ortonormirane in jih zloĹľimo v stolpce $V$.

  Iz $AV = U \Sigma$ sledi $A v_i = \sigma_i u_i$ za stolpce $v_i$ in $u_i$.
  ÄŚe je $\sigma_i \ne 0$, lahko tak $u_i$ poiĹˇÄŤemo.
  ÄŚe je $\sigma_r > \sigma_{r+1} = \sigma_{r+2} = \cdots = \sigma_n = 0$, tako
  doloÄŤimo prvih $r$ stolpcev $U$.
  Ker je $v_j$ lastni vektor za $A^T A$, velja
  \[
	u_i^T u_j = \inv{\sigma_i \sigma_j} v_i^T A^T A v_j = \inv{\sigma_i
	  \sigma_j} \sigma_j^2 v_i^T v_j = \delta_{ij}.
  \]
  Sedaj imamo
  \begin{gather*}
	V =
	\begin{bmatrix}
	  V_1 & V_2
	\end{bmatrix},
	\\
	\Sigma =
	\begin{bmatrix}
	  S & 0 \\
	  0 & 0
	\end{bmatrix},
	\\
	U =
	\begin{bmatrix}
	  U_1 & U_2
	\end{bmatrix},
  \end{gather*}
  kjer smo $U_2$ doloÄŤili tako, da dopolnimo stolpce $U_1$ do ortonormirane baze
  prostora.

  Preverimo, da te matrike res tvorijo singularni razcep:
  \[
	U^T A V =
	\begin{bmatrix}
	  U_1^T A V_1 & U_1^T A V_2 \\
	  U_2^T A V_1 & U_2^T A V_2
	\end{bmatrix}.
  \]
  Velja $(AV_2)^T A V_2 = V_2^T A^T A V_2 = V_2^T \cdot 0 = 0$, ker so stolpci
  $V_2$ lastni vektorji za lastno vrednost $0$.
  Iz definicije $u_i$ pa vidimo, da je $U_1^T A V_1 = S$.
  Velja $A V_1 = [\sigma_1 u_1, \ldots, \sigma_r u_r]$, ti stolpci so pravokotni
  vrednostim $u_{r+j}$ po definiciji.
\end{proof}

ÄŚe je $m < n$, singularni razcep dobimo tako, da transponiramo razcep za $A^T$.
V dokazu je $r$ rang matrike $A$.
NumeriÄŤno je singularni razcep najboljĹˇe orodje za raÄŤunanje ranga; dobimo tudi
bazi za jedro in sliko matrike $A$ (stolpci $V_2$ in $U_1$).

\vprasanje{PokaĹľi, da za vsako matriko obstaja singularni razcep.}

ÄŚe je $A$ ranga $r < n$, potem za vektor $x$, ki reĹˇi predoloÄŤen sistem $A x =
b$, izberemo tistega, ki minimizira $\norm{Ax-b}_2$ (ta ni veÄŤ enoliÄŤno
doloÄŤen), in ki ima med takimi minimalno normo $\norm{x}_2$.

\begin{izrek}
  Naj bo $A \in \R^{m \times n}$, $A = U \Sigma V^T$ ranga $r$.
  Potem je reĹˇitev predoloÄŤenega sistema $Ax = b$ enaka
  \[
	x = \sum_{i=1}^r \frac{u_i^T b}{\sigma_i} v_i.
  \]
\end{izrek}

\begin{proof}
  ZapiĹˇimo $U = [U_1, U_2]$, $V = [V_1, V_2]$ in $\Sigma = \operatorname{diag}(S,
  0)$, kjer so prve komponente Ĺˇirine $r$, druge pa Ĺˇirine $n-r$.
  Potem je
  \[
	\norm{Ax-b}_2 = \norm{U \Sigma V^T x - b}_2
	= \norm{\Sigma V^T x - U^T b}_2
	= \norm{
	  \begin{bmatrix}
		S y_1 - U_1^T b \\
		-U_2^T b
	  \end{bmatrix}

  \]
  za $V^T x = y = (y_1, y_2)$.
  Minimum bo doseĹľen, ÄŤe je $S y_1 = U_1^T b$, pri ÄŤemer je $y_2$ lahko
  poljuben.

\end{proof}

\vprasanje{KakĹˇna je reĹˇitev problema najmanjĹˇih kvadratov $Ax=b$ za matriko $A

\begin{definicija}

	\times m}$, ki zadoĹˇÄŤa naslednjim toÄŤkam:

\end{definicija}

ÄŚe je rang matrike $A$ enak $r <n$, potem je

\vprasanje{Definiraj psevdoinverz. ÄŚemu je enak?}

\begin{izrek}
  Naj bo $A = U \Sigma V^T \in \R^{m \times n}$ in $\rang A = r$.
  Potem je psevdoinverz enak
  \[
	A^+ = \sum_{i=1}^r \inv{\sigma_i} v_i u_i^T
  \]
  oziroma $A^+ = V \Sigma^+ U^T$ za $\Sigma^+ = \operatorname{diag}(S^{-1}, 0)$.

\begin{proof}
  IĹˇÄŤemo $X \in \R^{m \times n}$, ki zadoĹˇÄŤa Moore-Penroseovim pogojem.
  Velja $X = V Y U^T$ za nek $Y$.

  desni blok $Y$ enak $0$; podobno iz drugih pogojev izpeljemo enakosti v

\end{proof}

\podnaslov{Aproksimacija z matrikami niĹľjega ranga}

\begin{izrek}[Eckart-Young-Mirskg]
  ÄŚe je $A = U \Sigma V^T$ in $\rang A = r$, je za $k < r$ matrika
  \[
	A_k = \sum_{i=1}^k \sigma_i u_i v_i^T
  \]
  tista, ki minimizira
  \begin{itemize}
  \item $\min_{\rang B = k} \norm{A - B}_2$,
  \item $\min_{\rang B = k} \norm{A - B}_F$.
  \end{itemize}
  Dodatno velja $\norm{A - A_k}_2 = \sigma_{k+1}$ in $\norm{A - A_k}_F =
  \sqrt{\sigma_{k+1}^2 + \cdots + \sigma_n^2}$.
\end{izrek}

\begin{proof}
  DokaĹľemo samo prvo toÄŤko.
  Naj bo $B$ matrika ranga $k$.
  Definiramo $V_{k+1} = [v_1, \ldots, v_{k+1}]$.
  Velja $\dim(\jedro B) = n-k$ in $\dim(\im V_{k+1}) = k+1$, torej obstaja
  vektor $w \ne 0$ v preseku.
  Brez Ĺˇkode za sploĹˇnost je $\norm{w}_2 = 1$.
  Velja
  \[
	\norm{A-B}_2 = \max_{\norm{x}_2 = 1} \norm{(A-B)x}_2
	\ge \norm{(A-B)w}_2 = \norm{Aw}_2 = \sqrt{w^T A^T A w} \ge \sigma_{k+1}.
  \]
  Po drugi strani za $A_k$ oÄŤitno velja $\norm{A - A_k}_2 = \sigma_{k+1}$.
\end{proof}

\vprasanje{Kaj je najboljĹˇa aproksimacija matrike z matriko niĹľjega ranga?
  DokaĹľi.}

ÄŚe je $\sigma_{k+1} < \sigma_k$, je iz dokaza razvidno, da je najboljĹˇa
aproksimacija enoliÄŤna.

\podnaslov{Regularizacija}

Pri Fredholmovi integralski enaÄŤbi prve vrste
\[
  \int_0^1 K(s,t) f(t) dt = g(s)
\]
sta podana jedro $K$ in funkcija $g$, izraÄŤunati pa Ĺľelimo $f$.
Vsako jedro lahko razvijemo v
\[

\]
kjer so $u_i$ in $v_i$ leve in desne singularne funkcije, $\sigma_1 \ge \sigma_2
\ge \sigma_3 \ge \cdots$ pa singularne vrednosti, velja $\lim \sigma_n = 0$.
Funkcije $u_i, v_i$ so ortonormirane za skalarni produkt
\[
  \sk{g,h} = \int_0^1 g(t) h(t) dt,
\]
poleg tega velja
\[
  \int_0^1 K(s,t) v_i(t) dt = \sigma_i u_i(s).
\]
Podobno kot za singularni razcep lahko reĹˇitev poiĹˇÄŤemo z
\[

\]
PriÄŤakujemo, da bodo prispevki poznih ÄŤlenov majhni, torej da $\sk{u_i, g}$
limita k $0$ hitreje kot $\sigma_i$.
ÄŚe namesto $g$ poznamo $g + \Delta g$, lahko izraÄŤunamo
\[
  \tilde{f} = \sum_{i=1}^\infty \frac{\sk{u_i, g}}{\sigma_i} v_i +

Pri numeriÄŤnem reĹˇevanju bomo skalarni produkt izraÄŤunali z neko kvadraturno

	\\
	\alpha_0 K(s_1, t_0) & \alpha_1 K(s_1, t_1) & \cdots & \alpha_n K(s_1, t_n)
	\\
	\vdots & \vdots & \ddots & \vdots \\
	\alpha_0 K(s_n, t_0) & \alpha_1 K(s_n, t_1) & \cdots & \alpha_n K(s_n, t_n)
  \end{bmatrix}

  \begin{bmatrix}
	f(t_1) \\ \vdots \\ f(t_n)
  \end{bmatrix}
