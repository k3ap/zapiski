\podnaslov{Neodvisnost slučajnih spremenljivk}

\begin{definicija}
  Slučajni spremenljivki $X$ in $Y$ sta \pojem{neodvisni}, če za vsaki Borelovi
  $A$ in $B$ velja $P(X \in A, Y \in B) = P(X \in A) P(Y \in B)$.
\end{definicija}

\begin{definicija}
  Slučajne spremenljivke $X_1, \ldots, X_n$ so \pojem{neodvisne}, če za vsak
  nabor Borelovih množic $A_1, \ldots, A_n$ velja
  \[
	P(\forall i \velja X_i \in A_i) = \prod_j P(X_j \in A_j).
  \]
\end{definicija}

\begin{definicija}
  Slučajne spremenljivke $\{X_i\}_{i \in I}$ so \pojem{neodvisne}, če so
  neodvisne vse končne poddružine.
\end{definicija}

\vprasanje{Definiraj neodvisnost slučajnih spremenljivk.}

\begin{izrek}
  Naj za diskretni slučajni spremenljivki $X$ in $Y$ velja $P(X = x, Y = y) =
  f(x) g(y)$ za funkciji $f : R(X) \to \R$ in $g: R(Y) \to \R$ ($R$ je zaloga
  vrednosti).
  Potem sta $X$ in $Y$ neodvisni.
\end{izrek}

\begin{proof}
  Po formuli za robne porazdelitve je
  \[
	P(X = x) = f(x) \sum_y g(y) = f(x) C_1.
  \]
  Podobno $P(Y = y) = C_2 g(y)$.
  Predpišemo
  \[
	P(X = x, Y = y) = P(X = x) P(Y = y) C_1^{-1} C_2^{-1}.
  \]
  Dokazati moramo še, da velja $C_1 C_2 = 1$.
  Seštejmo
  \begin{align*}
	1 &= \sum_{x, y} P(X = x, Y = y) \\
	&= \inv{C_1 C_2} \sum_{x, y} P(X = x) P(Y = y)  \\
	&= \inv{C_1 C_2} \left( \sum_x P(X = x) \right) \left( \sum_y P(Y = y)
	\right) \\
	&= \inv{C_1 C_2}.
  \end{align*}
\end{proof}
